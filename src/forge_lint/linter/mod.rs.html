18/// Trait representing a generic linter for analyzing and reporting issues in smart contract source
19/// code files. A linter can be implemented for any smart contract language supported by Foundry.
20///
21/// # Type Parameters
22///
23/// - `Language`: Represents the target programming language. Must implement the [`Language`] trait.
24/// - `Lint`: Represents the types of lints performed by the linter. Must implement the [`Lint`]
25///   trait.
26///
27/// # Required Methods
28///
29/// - `init`: Creates a new solar `Session` with the appropriate linter configuration.
30/// - `early_lint`: Scans the source files (using the AST) emitting a diagnostic for lints found.
31/// - `late_lint`: Scans the source files (using the HIR) emitting a diagnostic for lints found.
32///
33/// # Note:
34///
35/// - For `early_lint` and `late_lint`, the `ParsingContext` should have the sources pre-loaded.
36pub trait Linter: Send + Sync + Clone {
37    type Language: Language;
38    type Lint: Lint;
39
40    fn init(&self) -> Session;
41    fn early_lint<'sess>(&self, input: &[PathBuf], pcx: ParsingContext<'sess>);
42    fn late_lint<'sess>(&self, input: &[PathBuf], pcx: ParsingContext<'sess>);
43}
44
45pub trait Lint {
46    fn id(&self) -> &'static str;
47    fn severity(&self) -> Severity;
48    fn description(&self) -> &'static str;
49    fn help(&self) -> &'static str;
50}

52pub struct LintContext<'s> {
53    sess: &'s Session,
54    with_description: bool,
55    pub inline_config: InlineConfig,
56    active_lints: Vec<&'static str>,
57}

59impl<'s> LintContext<'s> {
60    pub fn new(
61        sess: &'s Session,
62        with_description: bool,
63        config: InlineConfig,
64        active_lints: Vec<&'static str>,
65    ) -> Self {
66        Self { sess, with_description, inline_config: config, active_lints }
67    }
68
69    pub fn session(&self) -> &'s Session {
70        self.sess
71    }
72
73    // Helper method to check if a lint id is enabled.
74    //
75    // For performance reasons, some passes check several lints at once. Thus, this method is
76    // required to avoid unintended warnings.
77    pub fn is_lint_enabled(&self, id: &'static str) -> bool {
78        self.active_lints.contains(&id)
79    }
80
81    /// Helper method to emit diagnostics easily from passes
82    pub fn emit<L: Lint>(&self, lint: &'static L, span: Span) {
83        if self.inline_config.is_disabled(span, lint.id()) || !self.is_lint_enabled(lint.id()) {
84            return;
85        }
86
87        let desc = if self.with_description { lint.description() } else { "" };
88        let diag: DiagBuilder<'_, ()> = self
89            .sess
90            .dcx
91            .diag(lint.severity().into(), desc)
92            .code(DiagId::new_str(lint.id()))
93            .span(MultiSpan::from_span(span))
94            .help(lint.help());
95
96        diag.emit();
97    }
98
99    /// Emit a diagnostic with a code fix proposal.
100    ///
101    /// For Diff snippets, if no span is provided, it will use the lint's span.
102    /// If unable to get code from the span, it will fall back to a Block snippet.
103    pub fn emit_with_fix<L: Lint>(&self, lint: &'static L, span: Span, snippet: Snippet) {
104        if self.inline_config.is_disabled(span, lint.id()) || !self.is_lint_enabled(lint.id()) {
105            return;
106        }
107
108        // Convert the snippet to ensure we have the appropriate type
109        let snippet = match snippet {
110            Snippet::Diff { desc, span: diff_span, add, trim_code } => {
111                // Use the provided span or fall back to the lint span
112                let target_span = diff_span.unwrap_or(span);
113
114                // Check if we can get the original code
115                if self.span_to_snippet(target_span).is_some() {
116                    Snippet::Diff { desc, span: Some(target_span), add, trim_code }
117                } else {
118                    // Fall back to Block if we can't get the original code
119                    Snippet::Block { desc, code: add }
120                }
121            }
122            // Block snippets remain unchanged
123            block => block,
124        };
125
126        let desc = if self.with_description { lint.description() } else { "" };
127        let diag: DiagBuilder<'_, ()> = self
128            .sess
129            .dcx
130            .diag(lint.severity().into(), desc)
131            .code(DiagId::new_str(lint.id()))
132            .span(MultiSpan::from_span(span))
133            .highlighted_note(snippet.to_note(self))
134            .help(lint.help());
135
136        diag.emit();
137    }
138
139    /// Gets the "raw" source code (snippet) of the given span.
140    pub fn span_to_snippet(&self, span: Span) -> Option<String> {
141        self.sess.source_map().span_to_snippet(span).ok()
142    }
143
144    /// Gets the number of leading whitespaces (indentation) of the line where the span begins.
145    pub fn get_span_indentation(&self, span: Span) -> usize {
146        if !span.is_dummy() {
147            // Get the line text and compute the indentation prior to the span's position.
148            let loc = self.sess.source_map().lookup_char_pos(span.lo());
149            if let Some(line_text) = loc.file.get_line(loc.line) {
150                let col_offset = loc.col.to_usize();
151                if col_offset <= line_text.len() {
152                    let prev_text = &line_text[..col_offset];
153                    return prev_text.len() - prev_text.trim().len();
154                }
155            }
156        }
157
158        0
159    }
160}

162#[derive(Debug, Clone, Eq, PartialEq)]
163pub enum Snippet {
164    /// A standalone block of code. Used for showing examples without suggesting a fix.
165    Block {
166        /// An optional description displayed above the code block.
167        desc: Option<&'static str>,
168        /// The source code to display. Multi-line strings should include newlines.
169        code: String,
170    },
171
172    /// A proposed code change, displayed as a diff. Used to suggest replacements, showing the code
173    /// to be removed (from `span`) and the code to be added (from `add`).
174    Diff {
175        /// An optional description displayed above the diff.
176        desc: Option<&'static str>,
177        /// The `Span` of the source code to be removed. Note that, if uninformed,
178        /// `fn emit_with_fix()` falls back to the lint span.
179        span: Option<Span>,
180        /// The fix.
181        add: String,
182        /// If `true`, the leading whitespaces of the first line will be trimmed from the whole
183        /// code block. Applies to both, the added and removed code. This is useful for
184        /// aligning the indentation of multi-line replacements.
185        trim_code: bool,
186    },
187}

189impl Snippet {
190    pub fn to_note(self, ctx: &LintContext<'_>) -> Vec<(DiagMsg, Style)> {
191        let mut output = if let Some(desc) = self.desc() {
192            vec![(DiagMsg::from(desc), Style::NoStyle), (DiagMsg::from("\n\n"), Style::NoStyle)]
193        } else {
194            vec![(DiagMsg::from(" \n"), Style::NoStyle)]
195        };
196
197        match self {
198            Self::Diff { span, add, trim_code: trim, .. } => {
199                // Get the original code from the span if provided and normalize its indentation
200                if let Some(span) = span
201                    && let Some(rmv) = ctx.span_to_snippet(span)
202                {
203                    let ind = ctx.get_span_indentation(span);
204                    let diag_msg = |line: &str, prefix: &str, style: Style| {
205                        let content = if trim { Self::trim_start_limited(line, ind) } else { line };
206                        (DiagMsg::from(format!("{prefix}{content}\n")), style)
207                    };
208                    output.extend(rmv.lines().map(|line| diag_msg(line, "- ", Style::Removal)));
209                    output.extend(add.lines().map(|line| diag_msg(line, "+ ", Style::Addition)));
210                } else {
211                    // Should never happen, but fall back to `Self::Block` behavior.
212                    output.extend(
213                        add.lines()
214                            .map(|line| (DiagMsg::from(format!("{line}\n")), Style::NoStyle)),
215                    );
216                }
217            }
218            Self::Block { code, .. } => {
219                output.extend(
220                    code.lines().map(|line| (DiagMsg::from(format!("{line}\n")), Style::NoStyle)),
221                );
222            }
223        }
224        output.push((DiagMsg::from("\n"), Style::NoStyle));
225        output
226    }
227
228    pub fn desc(&self) -> Option<&'static str> {
229        match self {
230            Self::Diff { desc, .. } => *desc,
231            Self::Block { desc, .. } => *desc,
232        }
233    }
234
235    /// Removes up to `max_chars` whitespaces from the start of the string.
236    fn trim_start_limited(s: &str, max_chars: usize) -> &str {
237        let (mut chars, mut byte_offset) = (0, 0);
238        for c in s.chars() {
239            if chars >= max_chars || !c.is_whitespace() {
240                break;
241            }
242            chars += 1;
243            byte_offset += c.len_utf8();
244        }
245
246        &s[byte_offset..]
247    }
