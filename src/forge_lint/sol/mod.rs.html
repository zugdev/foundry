95    fn include_lint(&self, lint: SolLint) -> bool {
96        self.severity.as_ref().is_none_or(|sev| sev.contains(&lint.severity()))
97            && self.lints_included.as_ref().is_none_or(|incl| incl.contains(&lint))
98            && !self.lints_excluded.as_ref().is_some_and(|excl| excl.contains(&lint))
99    }
100
101    fn process_source_ast<'ast>(
102        &self,
103        sess: &'ast Session,
104        ast: &'ast ast::SourceUnit<'ast>,
105        file: &SourceFile,
106        path: &Path,
107    ) -> Result<(), diagnostics::ErrorGuaranteed> {
108        // Declare all available passes and lints
109        let mut passes_and_lints = Vec::new();
110        passes_and_lints.extend(high::create_early_lint_passes());
111        passes_and_lints.extend(med::create_early_lint_passes());
112        passes_and_lints.extend(info::create_early_lint_passes());
113
114        // Do not apply 'gas' and 'codesize' severity rules on tests and scripts
115        if !self.path_config.is_test_or_script(path) {
116            passes_and_lints.extend(gas::create_early_lint_passes());
117            passes_and_lints.extend(codesize::create_early_lint_passes());
118        }
119
120        // Filter passes based on linter config
121        let (mut passes, lints): (Vec<Box<dyn EarlyLintPass<'_>>>, Vec<_>) = passes_and_lints
122            .into_iter()
123            .fold((Vec::new(), Vec::new()), |(mut passes, mut ids), (pass, lints)| {
124                let included_ids: Vec<_> = lints
125                    .iter()
126                    .filter_map(|lint| if self.include_lint(*lint) { Some(lint.id) } else { None })
127                    .collect();
128
129                if !included_ids.is_empty() {
130                    passes.push(pass);
131                    ids.extend(included_ids);
132                }
133
134                (passes, ids)
135            });
136
137        // Process the inline-config
138        let comments = Comments::new(file);
139        let inline_config = parse_inline_config(sess, &comments, InlineConfigSource::Ast(ast));
140
141        // Initialize and run the early lint visitor
142        let ctx = LintContext::new(sess, self.with_description, inline_config, lints);
143        let mut early_visitor = EarlyLintVisitor::new(&ctx, &mut passes);
144        _ = early_visitor.visit_source_unit(ast);
145        early_visitor.post_source_unit(ast);
146
147        Ok(())
148    }

150    fn process_source_hir<'hir>(
151        &self,
152        sess: &Session,
153        gcx: &solar_sema::ty::Gcx<'hir>,
154        source_id: hir::SourceId,
155        file: &'hir SourceFile,
156    ) -> Result<(), diagnostics::ErrorGuaranteed> {
157        // Declare all available passes and lints
158        let mut passes_and_lints = Vec::new();
159        passes_and_lints.extend(high::create_late_lint_passes());
160        passes_and_lints.extend(med::create_late_lint_passes());
161        passes_and_lints.extend(info::create_late_lint_passes());
162
163        // Do not apply 'gas' and 'codesize' severity rules on tests and scripts
164        if let FileName::Real(ref path) = file.name
165            && !self.path_config.is_test_or_script(path)
166        {
167            passes_and_lints.extend(gas::create_late_lint_passes());
168            passes_and_lints.extend(codesize::create_late_lint_passes());
169        }
170
171        // Filter passes based on config
172        let (mut passes, lints): (Vec<Box<dyn LateLintPass<'_>>>, Vec<_>) = passes_and_lints
173            .into_iter()
174            .fold((Vec::new(), Vec::new()), |(mut passes, mut ids), (pass, lints)| {
175                let included_ids: Vec<_> = lints
176                    .iter()
177                    .filter_map(|lint| if self.include_lint(*lint) { Some(lint.id) } else { None })
178                    .collect();
179
180                if !included_ids.is_empty() {
181                    passes.push(pass);
182                    ids.extend(included_ids);
183                }
184
185                (passes, ids)
186            });
187
188        // Process the inline-config
189        let comments = Comments::new(file);
190        let inline_config =
191            parse_inline_config(sess, &comments, InlineConfigSource::Hir((&gcx.hir, source_id)));
192
193        // Run late lint visitor
194        let ctx = LintContext::new(sess, self.with_description, inline_config, lints);
195        let mut late_visitor = LateLintVisitor::new(&ctx, &mut passes, &gcx.hir);
196
197        // Visit this specific source
198        _ = late_visitor.visit_nested_source(source_id);
199
200        Ok(())
201    }
202}

204impl Linter for SolidityLinter {
205    type Language = SolcLanguage;
206    type Lint = SolLint;
207
208    /// Build solar session based on the linter config
209    fn init(&self) -> Session {
210        let mut builder = Session::builder();
211        if self.with_json_emitter {
212            let map = Arc::<SourceMap>::default();
213            let json_emitter = JsonEmitter::new(Box::new(std::io::stderr()), map.clone())
214                .rustc_like(true)
215                .ui_testing(false);
216
217            builder = builder.dcx(DiagCtxt::new(Box::new(json_emitter))).source_map(map);
218        } else {
219            builder = builder.with_stderr_emitter();
220        };
221
222        // Create a single session for all files
223        let mut sess = builder.build();
224        sess.dcx = sess.dcx.set_flags(|flags| flags.track_diagnostics = false);
225        sess
226    }
227
228    /// Run AST-based lints.
229    ///
230    /// Note: the `ParsingContext` should already have the sources loaded.
231    fn early_lint<'sess>(&self, input: &[PathBuf], pcx: ParsingContext<'sess>) {
232        let sess = pcx.sess;
233        _ = sess.enter_parallel(|| -> Result<(), diagnostics::ErrorGuaranteed> {
234            // Parse the sources
235            let ast_arena = solar_sema::thread_local::ThreadLocal::new();
236            let ast_result = pcx.parse(&ast_arena);
237
238            // Process each source in parallel
239            ast_result.sources.iter().par_bridge().for_each(|source| {
240                if let (FileName::Real(path), Some(ast)) = (&source.file.name, &source.ast)
241                    && input.iter().any(|input_path| path.ends_with(input_path))
242                {
243                    _ = self.process_source_ast(sess, ast, &source.file, path)
244                }
245            });
246
247            Ok(())
248        });
249    }
250
251    /// Run HIR-based lints.
252    ///
253    /// Note: the `ParsingContext` should already have the sources loaded.
254    fn late_lint<'sess>(&self, input: &[PathBuf], pcx: ParsingContext<'sess>) {
255        let sess = pcx.sess;
256        _ = sess.enter_parallel(|| -> Result<(), diagnostics::ErrorGuaranteed> {
257            // Parse and lower to HIR
258            let hir_arena = solar_sema::thread_local::ThreadLocal::new();
259            let hir_result = pcx.parse_and_lower(&hir_arena);
260
261            if let Ok(Some(gcx_wrapper)) = hir_result {
262                let gcx = gcx_wrapper.get();
263
264                // Process each source in parallel
265                gcx.hir.sources_enumerated().par_bridge().for_each(|(source_id, source)| {
266                    if let FileName::Real(ref path) = source.file.name
267                        && input.iter().any(|input_path| path.ends_with(input_path))
268                    {
269                        _ = self.process_source_hir(sess, &gcx, source_id, &source.file);
270                    }
271                });
272            }
273
274            Ok(())
275        });
276    }
277}

279enum InlineConfigSource<'ast, 'hir> {
280    Ast(&'ast ast::SourceUnit<'ast>),
281    Hir((&'hir hir::Hir<'hir>, hir::SourceId)),
282}
283
284fn parse_inline_config<'ast, 'hir>(
285    sess: &Session,
286    comments: &Comments,
287    source: InlineConfigSource<'ast, 'hir>,
288) -> InlineConfig {
289    let items = comments.iter().filter_map(|comment| {
290        let mut item = comment.lines.first()?.as_str();
291        if let Some(prefix) = comment.prefix() {
292            item = item.strip_prefix(prefix).unwrap_or(item);
293        }
294        if let Some(suffix) = comment.suffix() {
295            item = item.strip_suffix(suffix).unwrap_or(item);
296        }
297        let item = item.trim_start().strip_prefix("forge-lint:")?.trim();
298        let span = comment.span;
299        match InlineConfigItem::parse(item, &ALL_REGISTERED_LINTS) {
300            Ok(item) => Some((span, item)),
301            Err(e) => {
302                sess.dcx.warn(e.to_string()).span(span).emit();
303                None
304            }
305        }
306    });
307
308    match source {
309        InlineConfigSource::Ast(ast) => InlineConfig::from_ast(items, ast, sess.source_map()),
310        InlineConfigSource::Hir((hir, id)) => {
311            InlineConfig::from_hir(items, hir, id, sess.source_map())
312        }
313    }

345impl<'a> TryFrom<&'a str> for SolLint {
346    type Error = SolLintError;
347
348    fn try_from(value: &'a str) -> Result<Self, Self::Error> {
349        for &lint in high::REGISTERED_LINTS {
350            if lint.id() == value {
351                return Ok(lint);
352            }
353        }
354
355        for &lint in med::REGISTERED_LINTS {
356            if lint.id() == value {
357                return Ok(lint);
358            }
359        }
360
361        for &lint in info::REGISTERED_LINTS {
362            if lint.id() == value {
363                return Ok(lint);
364            }
365        }
366
367        for &lint in gas::REGISTERED_LINTS {
368            if lint.id() == value {
369                return Ok(lint);
370            }
371        }
372
373        for &lint in codesize::REGISTERED_LINTS {
374            if lint.id() == value {
375                return Ok(lint);
376            }
377        }
378
379        Err(SolLintError::InvalidId(value.to_string()))
380    }
