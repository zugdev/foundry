26const METADATA_SUFFIX: &str = "metadata.json";
27const JSON_EXTENSION: &str = ".json";
28const FAVORABILITY_THRESHOLD: f64 = 0.3;
29const COVERAGE_MAP_SIZE: usize = 65536;

31/// Possible mutation strategies to apply on a call sequence.
32#[derive(Debug, Clone)]
33enum MutationType {
34    /// Splice original call sequence.
35    Splice,
36    /// Repeat selected call several times.
37    Repeat,
38    /// Interleave calls from two random call sequences.
39    Interleave,
40    /// Replace prefix of the original call sequence with new calls.
41    Prefix,
42    /// Replace suffix of the original call sequence with new calls.
43    Suffix,
44    /// ABI mutate random args of selected call in sequence.
45    Abi,
46}

48/// Holds Corpus information.
49#[derive(Serialize)]
50struct CorpusEntry {
51    // Unique corpus identifier.
52    uuid: Uuid,
53    // Total mutations of corpus as primary source.
54    total_mutations: usize,
55    // New coverage found as a result of mutating this corpus.
56    new_finds_produced: usize,
57    // Corpus call sequence.
58    #[serde(skip_serializing)]
59    tx_seq: Vec<BasicTxDetails>,
60    // Whether this corpus is favored, i.e. producing new finds more often than
61    // `FAVORABILITY_THRESHOLD`.
62    is_favored: bool,
63}
64
65impl CorpusEntry {
66    /// New corpus from given call sequence and corpus path to read uuid.
67    pub fn new(tx_seq: Vec<BasicTxDetails>, path: PathBuf) -> eyre::Result<Self> {
68        let uuid = if let Some(stem) = path.file_stem().and_then(|s| s.to_str()) {
69            Uuid::try_from(stem.strip_suffix(JSON_EXTENSION).unwrap_or(stem).to_string())?
70        } else {
71            Uuid::new_v4()
72        };
73        Ok(Self { uuid, total_mutations: 0, new_finds_produced: 0, tx_seq, is_favored: false })
74    }
75
76    /// New corpus with given call sequence and new uuid.
77    pub fn from_tx_seq(tx_seq: &[BasicTxDetails]) -> Self {
78        Self {
79            uuid: Uuid::new_v4(),
80            total_mutations: 0,
81            new_finds_produced: 0,
82            tx_seq: tx_seq.into(),
83            is_favored: false,
84        }
85    }
86}

88#[derive(Serialize, Default)]
89pub(crate) struct CorpusMetrics {
90    // Number of edges seen during the invariant run.
91    cumulative_edges_seen: usize,
92    // Number of features (new hitcount bin of previously hit edge) seen during the invariant run.
93    cumulative_features_seen: usize,
94    // Number of corpus entries.
95    corpus_count: usize,
96    // Number of corpus entries that are favored.
97    favored_items: usize,
98}
99
100impl fmt::Display for CorpusMetrics {
101    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
102        writeln!(f)?;
103        writeln!(f, "        - cumulative edges seen: {}", self.cumulative_edges_seen)?;
104        writeln!(f, "        - cumulative features seen: {}", self.cumulative_features_seen)?;
105        writeln!(f, "        - corpus count: {}", self.corpus_count)?;
106        write!(f, "        - favored items: {}", self.favored_items)?;
107        Ok(())
108    }
109}
110
111impl CorpusMetrics {
112    /// Records number of new edges or features explored during the campaign.
113    pub fn update_seen(&mut self, is_edge: bool) {
114        if is_edge {
115            self.cumulative_edges_seen += 1;
116        } else {
117            self.cumulative_features_seen += 1;
118        }
119    }
120
121    /// Updates campaign favored items.
122    pub fn update_favored(&mut self, is_favored: bool, corpus_favored: bool) {
123        if is_favored && !corpus_favored {
124            self.favored_items += 1;
125        } else if !is_favored && corpus_favored {
126            self.favored_items -= 1;
127        }
128    }
129}

131/// Fuzz corpus manager, used in coverage guided fuzzing mode by both stateless and stateful tests.
132pub(crate) struct CorpusManager {
133    // Fuzzed calls generator.
134    tx_generator: BoxedStrategy<BasicTxDetails>,
135    // Call sequence mutation strategy type generator.
136    mutation_generator: BoxedStrategy<MutationType>,
137    // Corpus configuration.
138    config: FuzzCorpusConfig,
139    // In-memory corpus, populated from persisted files and current runs.
140    // Mutation is performed on these.
141    in_memory_corpus: Vec<CorpusEntry>,
142    // Identifier of current mutated entry.
143    current_mutated: Option<Uuid>,
144    // Number of failed replays from persisted corpus.
145    failed_replays: usize,
146    // History of binned hitcount of edges seen during fuzzing.
147    history_map: Vec<u8>,
148    // Corpus metrics.
149    pub(crate) metrics: CorpusMetrics,
150}

152impl CorpusManager {
153    pub fn new(
154        config: &FuzzCorpusConfig,
155        func_name: &String,
156        tx_generator: BoxedStrategy<BasicTxDetails>,
157        executor: &Executor,
158        fuzzed_function: Option<&Function>,
159        fuzzed_contracts: Option<&FuzzRunIdentifiedContracts>,
160    ) -> eyre::Result<Self> {
161        let mut config = config.clone();
162        config.with_test_name(func_name);
163
164        let mutation_generator = prop_oneof![
165            Just(MutationType::Splice),
166            Just(MutationType::Repeat),
167            Just(MutationType::Interleave),
168            Just(MutationType::Prefix),
169            Just(MutationType::Suffix),
170            Just(MutationType::Abi),
171        ]
172        .boxed();
173        let mut history_map = vec![0u8; COVERAGE_MAP_SIZE];
174        let mut metrics = CorpusMetrics::default();
175        let mut in_memory_corpus = vec![];
176        let mut failed_replays = 0;
177
178        // Early return if corpus dir / coverage guided fuzzing not configured.
179        let Some(corpus_dir) = &config.corpus_dir else {
180            return Ok(Self {
181                tx_generator,
182                mutation_generator,
183                config,
184                in_memory_corpus,
185                current_mutated: None,
186                failed_replays,
187                history_map,
188                metrics,
189            });
190        };
191
192        // Ensure corpus dir for current test is created.
193        if !corpus_dir.is_dir() {
194            foundry_common::fs::create_dir_all(corpus_dir)?;
195        }
196
197        let can_replay_tx = |tx: &BasicTxDetails| -> bool {
198            fuzzed_contracts.is_some_and(|contracts| contracts.targets.lock().can_replay(tx))
199                || fuzzed_function.is_some_and(|function| {
200                    tx.call_details
201                        .calldata
202                        .get(..4)
203                        .is_some_and(|selector| function.selector() == selector)
204                })
205        };
206
207        'corpus_replay: for entry in std::fs::read_dir(corpus_dir)? {
208            let path = entry?.path();
209            if path.is_file()
210                && let Some(name) = path.file_name().and_then(|s| s.to_str())
211                && name.contains(METADATA_SUFFIX)
212            {
213                // Ignore metadata files
214                continue;
215            }
216
217            let read_corpus_result = match path.extension().and_then(|ext| ext.to_str()) {
218                Some("gz") => foundry_common::fs::read_json_gzip_file::<Vec<BasicTxDetails>>(&path),
219                _ => foundry_common::fs::read_json_file::<Vec<BasicTxDetails>>(&path),
220            };
221
222            let Ok(tx_seq) = read_corpus_result else {
223                trace!(target: "corpus", "failed to load corpus from {}", path.display());
224                continue;
225            };
226
227            if !tx_seq.is_empty() {
228                // Warm up history map from loaded sequences.
229                let mut executor = executor.clone();
230                for tx in &tx_seq {
231                    if can_replay_tx(tx) {
232                        let mut call_result = executor
233                            .call_raw(
234                                tx.sender,
235                                tx.call_details.target,
236                                tx.call_details.calldata.clone(),
237                                U256::ZERO,
238                            )
239                            .map_err(|e| eyre!(format!("Could not make raw evm call: {e}")))?;
240
241                        let (new_coverage, is_edge) =
242                            call_result.merge_edge_coverage(&mut history_map);
243                        if new_coverage {
244                            metrics.update_seen(is_edge);
245                        }
246
247                        // Commit only when running invariant / stateful tests.
248                        if fuzzed_contracts.is_some() {
249                            executor.commit(&mut call_result);
250                        }
251                    } else {
252                        failed_replays += 1;
253
254                        // If the only input for fuzzed function cannot be replied, then move to
255                        // next one without adding it in memory.
256                        if fuzzed_function.is_some() {
257                            continue 'corpus_replay;
258                        }
259                    }
260                }
261
262                metrics.corpus_count += 1;
263
264                trace!(
265                    target: "corpus",
266                    "load sequence with len {} from corpus file {}",
267                    tx_seq.len(),
268                    path.display()
269                );
270
271                // Populate in memory corpus with the sequence from corpus file.
272                in_memory_corpus.push(CorpusEntry::new(tx_seq, path)?);
273            }
274        }
275
276        Ok(Self {
277            tx_generator,
278            mutation_generator,
279            config,
280            in_memory_corpus,
281            current_mutated: None,
282            failed_replays,
283            history_map,
284            metrics,
285        })
286    }

288    /// Updates stats for the given call sequence, if new coverage produced.
289    /// Persists the call sequence (if corpus directory is configured and new coverage) and updates
290    /// in-memory corpus.
291    pub fn process_inputs(&mut self, inputs: &[BasicTxDetails], new_coverage: bool) {
292        // Early return if corpus dir / coverage guided fuzzing is not configured.
293        let Some(corpus_dir) = &self.config.corpus_dir else {
294            return;
295        };
296
297        // Update stats of current mutated primary corpus.
298        if let Some(uuid) = &self.current_mutated {
299            if let Some(corpus) =
300                self.in_memory_corpus.iter_mut().find(|corpus| corpus.uuid.eq(uuid))
301            {
302                corpus.total_mutations += 1;
303                if new_coverage {
304                    corpus.new_finds_produced += 1
305                }
306                let is_favored = (corpus.new_finds_produced as f64 / corpus.total_mutations as f64)
307                    < FAVORABILITY_THRESHOLD;
308                self.metrics.update_favored(is_favored, corpus.is_favored);
309                corpus.is_favored = is_favored;
310
311                trace!(
312                    target: "corpus",
313                    "updated corpus {}, total mutations: {}, new finds: {}",
314                    corpus.uuid, corpus.total_mutations, corpus.new_finds_produced
315                );
316            }
317
318            self.current_mutated = None;
319        }
320
321        // Collect inputs only if current run produced new coverage.
322        if !new_coverage {
323            return;
324        }
325
326        let corpus = CorpusEntry::from_tx_seq(inputs);
327        let corpus_uuid = corpus.uuid;
328
329        // Persist to disk if corpus dir is configured.
330        let write_result = if self.config.corpus_gzip {
331            foundry_common::fs::write_json_gzip_file(
332                corpus_dir.join(format!("{corpus_uuid}{JSON_EXTENSION}.gz")).as_path(),
333                &corpus.tx_seq,
334            )
335        } else {
336            foundry_common::fs::write_json_file(
337                corpus_dir.join(format!("{corpus_uuid}{JSON_EXTENSION}")).as_path(),
338                &corpus.tx_seq,
339            )
340        };
341
342        if let Err(err) = write_result {
343            debug!(target: "corpus", %err, "Failed to record call sequence {:?}", &corpus.tx_seq);
344        } else {
345            trace!(
346                target: "corpus",
347                "persisted {} inputs for new coverage in {corpus_uuid} corpus",
348                &corpus.tx_seq.len()
349            );
350        }
351
352        // This includes reverting txs in the corpus and `can_continue` removes
353        // them. We want this as it is new coverage and may help reach the other branch.
354        self.metrics.corpus_count += 1;
355        self.in_memory_corpus.push(corpus);
356    }

358    /// Generates new call sequence from in memory corpus. Evicts oldest corpus mutated more than
359    /// configured max mutations value. Used by invariant test campaigns.
360    pub fn new_inputs(
361        &mut self,
362        test_runner: &mut TestRunner,
363        fuzz_state: &EvmFuzzState,
364        targeted_contracts: &FuzzRunIdentifiedContracts,
365    ) -> eyre::Result<Vec<BasicTxDetails>> {
366        let mut new_seq = vec![];
367
368        // Early return with first_input only if corpus dir / coverage guided fuzzing not
369        // configured.
370        if !self.config.is_coverage_guided() {
371            new_seq.push(self.new_tx(test_runner)?);
372            return Ok(new_seq);
373        };
374
375        if !self.in_memory_corpus.is_empty() {
376            self.evict_oldest_corpus()?;
377
378            let mutation_type = self
379                .mutation_generator
380                .new_tree(test_runner)
381                .map_err(|err| eyre!("Could not generate mutation type {err}"))?
382                .current();
383            let rng = test_runner.rng();
384            let corpus_len = self.in_memory_corpus.len();
385            let primary = &self.in_memory_corpus[rng.random_range(0..corpus_len)];
386            let secondary = &self.in_memory_corpus[rng.random_range(0..corpus_len)];
387
388            match mutation_type {
389                MutationType::Splice => {
390                    trace!(target: "corpus", "splice {} and {}", primary.uuid, secondary.uuid);
391
392                    self.current_mutated = Some(primary.uuid);
393
394                    let start1 = rng.random_range(0..primary.tx_seq.len());
395                    let end1 = rng.random_range(start1..primary.tx_seq.len());
396
397                    let start2 = rng.random_range(0..secondary.tx_seq.len());
398                    let end2 = rng.random_range(start2..secondary.tx_seq.len());
399
400                    for tx in primary.tx_seq.iter().take(end1).skip(start1) {
401                        new_seq.push(tx.clone());
402                    }
403                    for tx in secondary.tx_seq.iter().take(end2).skip(start2) {
404                        new_seq.push(tx.clone());
405                    }
406                }
407                MutationType::Repeat => {
408                    let corpus = if rng.random::<bool>() { primary } else { secondary };
409                    trace!(target: "corpus", "repeat {}", corpus.uuid);
410
411                    self.current_mutated = Some(corpus.uuid);
412
413                    new_seq = corpus.tx_seq.clone();
414                    let start = rng.random_range(0..corpus.tx_seq.len());
415                    let end = rng.random_range(start..corpus.tx_seq.len());
416                    let item_idx = rng.random_range(0..corpus.tx_seq.len());
417                    let repeated = vec![new_seq[item_idx].clone(); end - start];
418                    new_seq.splice(start..end, repeated);
419                }
420                MutationType::Interleave => {
421                    trace!(target: "corpus", "interleave {} with {}", primary.uuid, secondary.uuid);
422
423                    self.current_mutated = Some(primary.uuid);
424
425                    for (tx1, tx2) in primary.tx_seq.iter().zip(secondary.tx_seq.iter()) {
426                        // chunks?
427                        let tx = if rng.random::<bool>() { tx1.clone() } else { tx2.clone() };
428                        new_seq.push(tx);
429                    }
430                }
431                MutationType::Prefix => {
432                    let corpus = if rng.random::<bool>() { primary } else { secondary };
433                    trace!(target: "corpus", "overwrite prefix of {}", corpus.uuid);
434
435                    self.current_mutated = Some(corpus.uuid);
436
437                    new_seq = corpus.tx_seq.clone();
438                    for i in 0..rng.random_range(0..=new_seq.len()) {
439                        new_seq[i] = self.new_tx(test_runner)?;
440                    }
441                }
442                MutationType::Suffix => {
443                    let corpus = if rng.random::<bool>() { primary } else { secondary };
444                    trace!(target: "corpus", "overwrite suffix of {}", corpus.uuid);
445
446                    self.current_mutated = Some(corpus.uuid);
447
448                    new_seq = corpus.tx_seq.clone();
449                    for i in new_seq.len() - rng.random_range(0..new_seq.len())..corpus.tx_seq.len()
450                    {
451                        new_seq[i] = self.new_tx(test_runner)?;
452                    }
453                }
454                MutationType::Abi => {
455                    let targets = targeted_contracts.targets.lock();
456                    let corpus = if rng.random::<bool>() { primary } else { secondary };
457                    trace!(target: "corpus", "ABI mutate args of {}", corpus.uuid);
458
459                    self.current_mutated = Some(corpus.uuid);
460
461                    new_seq = corpus.tx_seq.clone();
462
463                    let idx = rng.random_range(0..new_seq.len());
464                    let tx = new_seq.get_mut(idx).unwrap();
465                    if let (_, Some(function)) = targets.fuzzed_artifacts(tx) {
466                        // TODO add call_value to call details and mutate it as well as sender some
467                        // of the time
468                        if !function.inputs.is_empty() {
469                            self.abi_mutate(tx, function, test_runner, fuzz_state)?;
470                        }
471                    }
472                }
473            }
474        }
475
476        // Make sure the new sequence contains at least one tx to start fuzzing from.
477        if new_seq.is_empty() {
478            new_seq.push(self.new_tx(test_runner)?);
479        }
480        trace!(target: "corpus", "new sequence of {} calls generated", new_seq.len());
481
482        Ok(new_seq)
483    }

485    /// Generates new input from in memory corpus. Evicts oldest corpus mutated more than
486    /// configured max mutations value. Used by fuzz test campaigns.
487    pub fn new_input(
488        &mut self,
489        test_runner: &mut TestRunner,
490        fuzz_state: &EvmFuzzState,
491        function: &Function,
492    ) -> eyre::Result<Bytes> {
493        // Early return if not running with coverage guided fuzzing.
494        if !self.config.is_coverage_guided() {
495            return Ok(self.new_tx(test_runner)?.call_details.calldata);
496        }
497
498        let tx = if !self.in_memory_corpus.is_empty() {
499            self.evict_oldest_corpus()?;
500
501            let corpus = &self.in_memory_corpus
502                [test_runner.rng().random_range(0..self.in_memory_corpus.len())];
503            self.current_mutated = Some(corpus.uuid);
504            let new_seq = corpus.tx_seq.clone();
505            let mut tx = new_seq.first().unwrap().clone();
506            self.abi_mutate(&mut tx, function, test_runner, fuzz_state)?;
507            tx
508        } else {
509            self.new_tx(test_runner)?
510        };
511
512        Ok(tx.call_details.calldata)
513    }

515    /// Returns the next call to be used in call sequence.
516    /// If coverage guided fuzzing is not configured or if previous input was discarded then this is
517    /// a new tx from strategy.
518    /// If running with coverage guided fuzzing it returns a new call only when sequence
519    /// does not have enough entries, or randomly. Otherwise, returns the next call from initial
520    /// sequence.
521    pub fn generate_next_input(
522        &mut self,
523        test_runner: &mut TestRunner,
524        sequence: &[BasicTxDetails],
525        discarded: bool,
526        depth: usize,
527    ) -> eyre::Result<BasicTxDetails> {
528        // Early return with new input if corpus dir / coverage guided fuzzing not configured or if
529        // call was discarded.
530        if self.config.corpus_dir.is_none() || discarded {
531            return self.new_tx(test_runner);
532        }
533
534        // When running with coverage guided fuzzing enabled then generate new sequence if initial
535        // sequence's length is less than depth or randomly, to occasionally intermix new txs.
536        if depth > sequence.len().saturating_sub(1) || test_runner.rng().random_ratio(1, 10) {
537            return self.new_tx(test_runner);
538        }
539
540        // Continue with the next call initial sequence
541        Ok(sequence[depth].clone())
542    }

558    /// Collects coverage from call result and updates metrics.
559    pub fn merge_edge_coverage(&mut self, call_result: &mut RawCallResult) -> bool {
560        if !self.config.collect_edge_coverage() {
561            return false;
562        }
563
564        let (new_coverage, is_edge) = call_result.merge_edge_coverage(&mut self.history_map);
565        if new_coverage {
566            self.metrics.update_seen(is_edge);
567        }
568        new_coverage
569    }

571    /// Flush the oldest corpus mutated more than configured max mutations unless they are
572    /// favored.
573    fn evict_oldest_corpus(&mut self) -> eyre::Result<()> {
574        if self.in_memory_corpus.len() > self.config.corpus_min_size.max(1)
575            && let Some(index) = self.in_memory_corpus.iter().position(|corpus| {
576                corpus.total_mutations > self.config.corpus_min_mutations && !corpus.is_favored
577            })
578        {
579            let corpus = self.in_memory_corpus.get(index).unwrap();
580
581            let uuid = corpus.uuid;
582            debug!(target: "corpus", "evict corpus {uuid}");
583
584            // Flush to disk the seed metadata at the time of eviction.
585            let eviction_time = SystemTime::now().duration_since(UNIX_EPOCH)?.as_secs();
586            foundry_common::fs::write_json_file(
587                self.config
588                    .corpus_dir
589                    .clone()
590                    .unwrap()
591                    .join(format!("{uuid}-{eviction_time}-{METADATA_SUFFIX}"))
592                    .as_path(),
593                &corpus,
594            )?;
595
596            // Remove corpus from memory.
597            self.in_memory_corpus.remove(index);
598        }
599        Ok(())
600    }

602    /// Mutates calldata of provided tx by abi decoding current values and randomly selecting the
603    /// inputs to change.
604    fn abi_mutate(
605        &self,
606        tx: &mut BasicTxDetails,
607        function: &Function,
608        test_runner: &mut TestRunner,
609        fuzz_state: &EvmFuzzState,
610    ) -> eyre::Result<()> {
611        // let rng = test_runner.rng();
612        let mut arg_mutation_rounds =
613            test_runner.rng().random_range(0..=function.inputs.len()).max(1);
614        let round_arg_idx: Vec<usize> = if function.inputs.len() <= 1 {
615            vec![0]
616        } else {
617            (0..arg_mutation_rounds)
618                .map(|_| test_runner.rng().random_range(0..function.inputs.len()))
619                .collect()
620        };
621        let mut prev_inputs = function
622            .abi_decode_input(&tx.call_details.calldata[4..])
623            .map_err(|err| eyre!("failed to load previous inputs: {err}"))?;
624
625        while arg_mutation_rounds > 0 {
626            let idx = round_arg_idx[arg_mutation_rounds - 1];
627            prev_inputs[idx] = mutate_param_value(
628                &function
629                    .inputs
630                    .get(idx)
631                    .expect("Could not get input to mutate")
632                    .selector_type()
633                    .parse()?,
634                prev_inputs[idx].clone(),
635                test_runner,
636                fuzz_state,
637            );
638            arg_mutation_rounds -= 1;
639        }
640
641        tx.call_details.calldata =
