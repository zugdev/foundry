31/// Contains data collected during fuzz test runs.
32#[derive(Default)]
33struct FuzzTestData {
34    // Stores the first fuzz case.
35    first_case: Option<FuzzCase>,
36    // Stored gas usage per fuzz case.
37    gas_by_case: Vec<(u64, u64)>,
38    // Stores the result and calldata of the last failed call, if any.
39    counterexample: (Bytes, RawCallResult),
40    // Stores up to `max_traces_to_collect` traces.
41    traces: Vec<SparsedTraceArena>,
42    // Stores breakpoints for the last fuzz case.
43    breakpoints: Option<Breakpoints>,
44    // Stores coverage information for all fuzz cases.
45    coverage: Option<HitMaps>,
46    // Stores logs for all fuzz cases
47    logs: Vec<Log>,
48    // Deprecated cheatcodes mapped to their replacements.
49    deprecated_cheatcodes: HashMap<&'static str, Option<&'static str>>,
50    // Runs performed in fuzz test.
51    runs: u32,
52    // Current assume rejects of the fuzz run.
53    rejects: u32,
54    // Test failure.
55    failure: Option<TestCaseError>,
56}

58/// Wrapper around an [`Executor`] which provides fuzzing support using [`proptest`].
59///
60/// After instantiation, calling `fuzz` will proceed to hammer the deployed smart contract with
61/// inputs, until it finds a counterexample. The provided [`TestRunner`] contains all the
62/// configuration which can be overridden via [environment variables](proptest::test_runner::Config)
63pub struct FuzzedExecutor {
64    /// The EVM executor.
65    executor: Executor,
66    /// The fuzzer
67    runner: TestRunner,
68    /// The account that calls tests.
69    sender: Address,
70    /// The fuzz configuration.
71    config: FuzzConfig,
72    /// The persisted counterexample to be replayed, if any.
73    persisted_failure: Option<BaseCounterExample>,
74}
75
76impl FuzzedExecutor {
77    /// Instantiates a fuzzed executor given a testrunner
78    pub fn new(
79        executor: Executor,
80        runner: TestRunner,
81        sender: Address,
82        config: FuzzConfig,
83        persisted_failure: Option<BaseCounterExample>,
84    ) -> Self {
85        Self { executor, runner, sender, config, persisted_failure }
86    }

88    /// Fuzzes the provided function, assuming it is available at the contract at `address`
89    /// If `should_fail` is set to `true`, then it will stop only when there's a success
90    /// test case.
91    ///
92    /// Returns a list of all the consumed gas and calldata of every fuzz case.
93    pub fn fuzz(
94        &mut self,
95        func: &Function,
96        fuzz_fixtures: &FuzzFixtures,
97        deployed_libs: &[Address],
98        address: Address,
99        rd: &RevertDecoder,
100        progress: Option<&ProgressBar>,
101    ) -> Result<FuzzTestResult> {
102        // Stores the fuzz test execution data.
103        let mut test_data = FuzzTestData::default();
104        let state = self.build_fuzz_state(deployed_libs);
105        let dictionary_weight = self.config.dictionary.dictionary_weight.min(100);
106        let strategy = proptest::prop_oneof![
107            100 - dictionary_weight => fuzz_calldata(func.clone(), fuzz_fixtures),
108            dictionary_weight => fuzz_calldata_from_state(func.clone(), &state),
109        ]
110        .prop_map(move |calldata| BasicTxDetails {
111            sender: Default::default(),
112            call_details: CallDetails { target: Default::default(), calldata },
113        });
114        // We want to collect at least one trace which will be displayed to user.
115        let max_traces_to_collect = std::cmp::max(1, self.config.gas_report_samples) as usize;
116
117        let mut corpus_manager = CorpusManager::new(
118            &self.config.corpus,
119            &func.name,
120            strategy.boxed(),
121            &self.executor,
122            Some(func),
123            None,
124        )?;
125
126        // Start timer for this fuzz test.
127        let timer = FuzzTestTimer::new(self.config.timeout);
128        let mut last_metrics_report = Instant::now();
129        let max_runs = self.config.runs;
130        let continue_campaign = |runs: u32| {
131            if timer.is_enabled() { !timer.is_timed_out() } else { runs < max_runs }
132        };
133
134        'stop: while continue_campaign(test_data.runs) {
135            // If counterexample recorded, replay it first, without incrementing runs.
136            let input = if let Some(failure) = self.persisted_failure.take() {
137                failure.calldata
138            } else {
139                // If running with progress, then increment current run.
140                if let Some(progress) = progress {
141                    progress.inc(1);
142                    // Display metrics in progress bar.
143                    if self.config.corpus.collect_edge_coverage() {
144                        progress.set_message(format!("{}", &corpus_manager.metrics));
145                    }
146                } else if self.config.corpus.collect_edge_coverage()
147                    && last_metrics_report.elapsed() > DURATION_BETWEEN_METRICS_REPORT
148                {
149                    // Display metrics inline.
150                    let metrics = json!({
151                        "timestamp": SystemTime::now()
152                            .duration_since(UNIX_EPOCH)?
153                            .as_secs(),
154                        "test": func.name,
155                        "metrics": &corpus_manager.metrics,
156                    });
157                    let _ = sh_println!("{}", serde_json::to_string(&metrics)?);
158                    last_metrics_report = Instant::now();
159                };
160
161                test_data.runs += 1;
162
163                match corpus_manager.new_input(&mut self.runner, &state, func) {
164                    Ok(input) => input,
165                    Err(err) => {
166                        test_data.failure = Some(TestCaseError::fail(format!(
167                            "failed to generate fuzzed input: {err}"
168                        )));
169                        break 'stop;
170                    }
171                }
172            };
173
174            match self.single_fuzz(address, input, &mut corpus_manager) {
175                Ok(fuzz_outcome) => match fuzz_outcome {
176                    FuzzOutcome::Case(case) => {
177                        test_data.gas_by_case.push((case.case.gas, case.case.stipend));
178
179                        if test_data.first_case.is_none() {
180                            test_data.first_case.replace(case.case);
181                        }
182
183                        if let Some(call_traces) = case.traces {
184                            if test_data.traces.len() == max_traces_to_collect {
185                                test_data.traces.pop();
186                            }
187                            test_data.traces.push(call_traces);
188                            test_data.breakpoints.replace(case.breakpoints);
189                        }
190
191                        if self.config.show_logs {
192                            test_data.logs.extend(case.logs);
193                        }
194
195                        HitMaps::merge_opt(&mut test_data.coverage, case.coverage);
196                        test_data.deprecated_cheatcodes = case.deprecated_cheatcodes;
197                    }
198                    FuzzOutcome::CounterExample(CounterExampleOutcome {
199                        exit_reason: status,
200                        counterexample: outcome,
201                        ..
202                    }) => {
203                        let reason = rd.maybe_decode(&outcome.1.result, status);
204                        test_data.logs.extend(outcome.1.logs.clone());
205                        test_data.counterexample = outcome;
206                        test_data.failure = Some(TestCaseError::fail(reason.unwrap_or_default()));
207                        break 'stop;
208                    }
209                },
210                Err(err) => {
211                    match err {
212                        TestCaseError::Fail(_) => {
213                            test_data.failure = Some(err);
214                            break 'stop;
215                        }
216                        TestCaseError::Reject(_) => {
217                            // Apply max rejects only if configured, otherwise silently discard run.
218                            if self.config.max_test_rejects > 0 {
219                                test_data.rejects += 1;
220                                if test_data.rejects >= self.config.max_test_rejects {
221                                    test_data.failure = Some(err);
222                                    break 'stop;
223                                }
224                            }
225                        }
226                    }
227                }
228            }
229        }
230
231        let (calldata, call) = test_data.counterexample;
232        let mut traces = test_data.traces;
233        let (last_run_traces, last_run_breakpoints) = if test_data.failure.is_none() {
234            (traces.pop(), test_data.breakpoints)
235        } else {
236            (call.traces.clone(), call.cheatcodes.map(|c| c.breakpoints))
237        };
238
239        let mut result = FuzzTestResult {
240            first_case: test_data.first_case.unwrap_or_default(),
241            gas_by_case: test_data.gas_by_case,
242            success: test_data.failure.is_none(),
243            skipped: false,
244            reason: None,
245            counterexample: None,
246            logs: test_data.logs,
247            labels: call.labels,
248            traces: last_run_traces,
249            breakpoints: last_run_breakpoints,
250            gas_report_traces: traces.into_iter().map(|a| a.arena).collect(),
251            line_coverage: test_data.coverage,
252            deprecated_cheatcodes: test_data.deprecated_cheatcodes,
253            failed_corpus_replays: corpus_manager.failed_replays(),
254        };
255
256        match test_data.failure {
257            Some(TestCaseError::Fail(reason)) => {
258                let reason = reason.to_string();
259                result.reason = (!reason.is_empty()).then_some(reason);
260                let args = if let Some(data) = calldata.get(4..) {
261                    func.abi_decode_input(data).unwrap_or_default()
262                } else {
263                    vec![]
264                };
265                result.counterexample = Some(CounterExample::Single(
266                    BaseCounterExample::from_fuzz_call(calldata, args, call.traces),
267                ));
268            }
269            Some(TestCaseError::Reject(reason)) => {
270                let reason = reason.to_string();
271                result.reason = (!reason.is_empty()).then_some(reason);
272            }
273            None => {}
274        }
275
276        if let Some(reason) = &result.reason
277            && let Some(reason) = SkipReason::decode_self(reason)
278        {
279            result.skipped = true;
280            result.reason = reason.0;
281        }
282
283        state.log_stats();
284
285        Ok(result)
286    }

288    /// Granular and single-step function that runs only one fuzz and returns either a `CaseOutcome`
289    /// or a `CounterExampleOutcome`
290    fn single_fuzz(
291        &mut self,
292        address: Address,
293        calldata: Bytes,
294        coverage_metrics: &mut CorpusManager,
295    ) -> Result<FuzzOutcome, TestCaseError> {
296        let mut call = self
297            .executor
298            .call_raw(self.sender, address, calldata.clone(), U256::ZERO)
299            .map_err(|e| TestCaseError::fail(e.to_string()))?;
300        let new_coverage = coverage_metrics.merge_edge_coverage(&mut call);
301        coverage_metrics.process_inputs(
302            &[BasicTxDetails {
303                sender: self.sender,
304                call_details: CallDetails { target: address, calldata: calldata.clone() },
305            }],
306            new_coverage,
307        );
308
309        // Handle `vm.assume`.
310        if call.result.as_ref() == MAGIC_ASSUME {
311            return Err(TestCaseError::reject(FuzzError::TooManyRejects(
312                self.config.max_test_rejects,
313            )));
314        }
315
316        let (breakpoints, deprecated_cheatcodes) =
317            call.cheatcodes.as_ref().map_or_else(Default::default, |cheats| {
318                (cheats.breakpoints.clone(), cheats.deprecated.clone())
319            });
320
321        // Consider call success if test should not fail on reverts and reverter is not the
322        // cheatcode or test address.
323        let success = if !self.config.fail_on_revert
324            && call
325                .reverter
326                .is_some_and(|reverter| reverter != address && reverter != CHEATCODE_ADDRESS)
327        {
328            true
329        } else {
330            self.executor.is_raw_call_mut_success(address, &mut call, false)
331        };
332
333        if success {
334            Ok(FuzzOutcome::Case(CaseOutcome {
335                case: FuzzCase { calldata, gas: call.gas_used, stipend: call.stipend },
336                traces: call.traces,
337                coverage: call.line_coverage,
338                breakpoints,
339                logs: call.logs,
340                deprecated_cheatcodes,
341            }))
342        } else {
343            Ok(FuzzOutcome::CounterExample(CounterExampleOutcome {
344                exit_reason: call.exit_reason,
345                counterexample: (calldata, call),
346                breakpoints,
347            }))
348        }
349    }

351    /// Stores fuzz state for use with [fuzz_calldata_from_state]
352    pub fn build_fuzz_state(&self, deployed_libs: &[Address]) -> EvmFuzzState {
353        if let Some(fork_db) = self.executor.backend().active_fork_db() {
354            EvmFuzzState::new(fork_db, self.config.dictionary, deployed_libs)
355        } else {
356            EvmFuzzState::new(
357                self.executor.backend().mem_db(),
358                self.config.dictionary,
359                deployed_libs,
360            )
361        }
362    }
