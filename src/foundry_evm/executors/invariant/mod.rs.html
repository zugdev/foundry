121/// Contains data collected during invariant test runs.
122struct InvariantTestData {
123    // Consumed gas and calldata of every successful fuzz call.
124    fuzz_cases: Vec<FuzzedCases>,
125    // Data related to reverts or failed assertions of the test.
126    failures: InvariantFailures,
127    // Calldata in the last invariant run.
128    last_run_inputs: Vec<BasicTxDetails>,
129    // Additional traces for gas report.
130    gas_report_traces: Vec<Vec<CallTraceArena>>,
131    // Last call results of the invariant test.
132    last_call_results: Option<RawCallResult>,
133    // Line coverage information collected from all fuzzed calls.
134    line_coverage: Option<HitMaps>,
135    // Metrics for each fuzzed selector.
136    metrics: Map<String, InvariantMetrics>,
137
138    // Proptest runner to query for random values.
139    // The strategy only comes with the first `input`. We fill the rest of the `inputs`
140    // until the desired `depth` so we can use the evolving fuzz dictionary
141    // during the run.
142    branch_runner: TestRunner,
143}

145/// Contains invariant test data.
146struct InvariantTest {
147    // Fuzz state of invariant test.
148    fuzz_state: EvmFuzzState,
149    // Contracts fuzzed by the invariant test.
150    targeted_contracts: FuzzRunIdentifiedContracts,
151    // Data collected during invariant runs.
152    test_data: InvariantTestData,
153}
154
155impl InvariantTest {
156    /// Instantiates an invariant test.
157    fn new(
158        fuzz_state: EvmFuzzState,
159        targeted_contracts: FuzzRunIdentifiedContracts,
160        failures: InvariantFailures,
161        last_call_results: Option<RawCallResult>,
162        branch_runner: TestRunner,
163    ) -> Self {
164        let mut fuzz_cases = vec![];
165        if last_call_results.is_none() {
166            fuzz_cases.push(FuzzedCases::new(vec![]));
167        }
168        let test_data = InvariantTestData {
169            fuzz_cases,
170            failures,
171            last_run_inputs: vec![],
172            gas_report_traces: vec![],
173            last_call_results,
174            line_coverage: None,
175            metrics: Map::default(),
176            branch_runner,
177        };
178        Self { fuzz_state, targeted_contracts, test_data }
179    }
180
181    /// Returns number of invariant test reverts.
182    fn reverts(&self) -> usize {
183        self.test_data.failures.reverts
184    }
185
186    /// Whether invariant test has errors or not.
187    fn has_errors(&self) -> bool {
188        self.test_data.failures.error.is_some()
189    }
190
191    /// Set invariant test error.
192    fn set_error(&mut self, error: InvariantFuzzError) {
193        self.test_data.failures.error = Some(error);
194    }
195
196    /// Set last invariant test call results.
197    fn set_last_call_results(&mut self, call_result: Option<RawCallResult>) {
198        self.test_data.last_call_results = call_result;
199    }
200
201    /// Set last invariant run call sequence.
202    fn set_last_run_inputs(&mut self, inputs: &Vec<BasicTxDetails>) {
203        self.test_data.last_run_inputs.clone_from(inputs);
204    }
205
206    /// Merge current collected line coverage with the new coverage from last fuzzed call.
207    fn merge_line_coverage(&mut self, new_coverage: Option<HitMaps>) {
208        HitMaps::merge_opt(&mut self.test_data.line_coverage, new_coverage);
209    }
210
211    /// Update metrics for a fuzzed selector, extracted from tx details.
212    /// Always increments number of calls; discarded runs (through assume cheatcodes) are tracked
213    /// separated from reverts.
214    fn record_metrics(&mut self, tx_details: &BasicTxDetails, reverted: bool, discarded: bool) {
215        if let Some(metric_key) =
216            self.targeted_contracts.targets.lock().fuzzed_metric_key(tx_details)
217        {
218            let test_metrics = &mut self.test_data.metrics;
219            let invariant_metrics = test_metrics.entry(metric_key).or_default();
220            invariant_metrics.calls += 1;
221            if discarded {
222                invariant_metrics.discards += 1;
223            } else if reverted {
224                invariant_metrics.reverts += 1;
225            }
226        }
227    }
228
229    /// End invariant test run by collecting results, cleaning collected artifacts and reverting
230    /// created fuzz state.
231    fn end_run(&mut self, run: InvariantTestRun, gas_samples: usize) {
232        // We clear all the targeted contracts created during this run.
233        self.targeted_contracts.clear_created_contracts(run.created_contracts);
234
235        if self.test_data.gas_report_traces.len() < gas_samples {
236            self.test_data
237                .gas_report_traces
238                .push(run.run_traces.into_iter().map(|arena| arena.arena).collect());
239        }
240        self.test_data.fuzz_cases.push(FuzzedCases::new(run.fuzz_runs));
241
242        // Revert state to not persist values between runs.
243        self.fuzz_state.revert();
244    }
245}

283/// Wrapper around any [`Executor`] implementer which provides fuzzing support using [`proptest`].
284///
285/// After instantiation, calling `invariant_fuzz` will proceed to hammer the deployed smart
286/// contracts with inputs, until it finds a counterexample sequence. The provided [`TestRunner`]
287/// contains all the configuration which can be overridden via [environment
288/// variables](proptest::test_runner::Config)
289pub struct InvariantExecutor<'a> {
290    pub executor: Executor,
291    /// Proptest runner.
292    runner: TestRunner,
293    /// The invariant configuration
294    config: InvariantConfig,
295    /// Contracts deployed with `setUp()`
296    setup_contracts: &'a ContractsByAddress,
297    /// Contracts that are part of the project but have not been deployed yet. We need the bytecode
298    /// to identify them from the stateset changes.
299    project_contracts: &'a ContractsByArtifact,
300    /// Filters contracts to be fuzzed through their artifact identifiers.
301    artifact_filters: ArtifactFilters,
302}
303
304impl<'a> InvariantExecutor<'a> {
305    /// Instantiates a fuzzed executor EVM given a testrunner
306    pub fn new(
307        executor: Executor,
308        runner: TestRunner,
309        config: InvariantConfig,
310        setup_contracts: &'a ContractsByAddress,
311        project_contracts: &'a ContractsByArtifact,
312    ) -> Self {
313        Self {
314            executor,
315            runner,
316            config,
317            setup_contracts,
318            project_contracts,
319            artifact_filters: ArtifactFilters::default(),
320        }
321    }

323    /// Fuzzes any deployed contract and checks any broken invariant at `invariant_address`.
324    pub fn invariant_fuzz(
325        &mut self,
326        invariant_contract: InvariantContract<'_>,
327        fuzz_fixtures: &FuzzFixtures,
328        deployed_libs: &[Address],
329        progress: Option<&ProgressBar>,
330    ) -> Result<InvariantFuzzTestResult> {
331        // Throw an error to abort test run if the invariant function accepts input params
332        if !invariant_contract.invariant_function.inputs.is_empty() {
333            return Err(eyre!("Invariant test function should have no inputs"));
334        }
335
336        let (mut invariant_test, mut corpus_manager) =
337            self.prepare_test(&invariant_contract, fuzz_fixtures, deployed_libs)?;
338
339        // Start timer for this invariant test.
340        let mut runs = 0;
341        let timer = FuzzTestTimer::new(self.config.timeout);
342        let mut last_metrics_report = Instant::now();
343        let continue_campaign = |runs: u32| {
344            if timer.is_enabled() { !timer.is_timed_out() } else { runs < self.config.runs }
345        };
346
347        // Invariant runs with edge coverage if corpus dir is set or showing edge coverage.
348        let edge_coverage_enabled = self.config.corpus.collect_edge_coverage();
349
350        'stop: while continue_campaign(runs) {
351            let initial_seq = corpus_manager.new_inputs(
352                &mut invariant_test.test_data.branch_runner,
353                &invariant_test.fuzz_state,
354                &invariant_test.targeted_contracts,
355            )?;
356
357            // Create current invariant run data.
358            let mut current_run = InvariantTestRun::new(
359                initial_seq[0].clone(),
360                // Before each run, we must reset the backend state.
361                self.executor.clone(),
362                self.config.depth as usize,
363            );
364
365            // We stop the run immediately if we have reverted, and `fail_on_revert` is set.
366            if self.config.fail_on_revert && invariant_test.reverts() > 0 {
367                return Err(eyre!("call reverted"));
368            }
369
370            while current_run.depth < self.config.depth {
371                // Check if the timeout has been reached.
372                if timer.is_timed_out() {
373                    // Since we never record a revert here the test is still considered
374                    // successful even though it timed out. We *want*
375                    // this behavior for now, so that's ok, but
376                    // future developers should be aware of this.
377                    break 'stop;
378                }
379
380                let tx = current_run
381                    .inputs
382                    .last()
383                    .ok_or_else(|| eyre!("no input generated to call fuzzed target."))?;
384
385                // Execute call from the randomly generated sequence without committing state.
386                // State is committed only if call is not a magic assume.
387                let mut call_result = current_run
388                    .executor
389                    .call_raw(
390                        tx.sender,
391                        tx.call_details.target,
392                        tx.call_details.calldata.clone(),
393                        U256::ZERO,
394                    )
395                    .map_err(|e| eyre!(format!("Could not make raw evm call: {e}")))?;
396
397                let discarded = call_result.result.as_ref() == MAGIC_ASSUME;
398                if self.config.show_metrics {
399                    invariant_test.record_metrics(tx, call_result.reverted, discarded);
400                }
401
402                // Collect line coverage from last fuzzed call.
403                invariant_test.merge_line_coverage(call_result.line_coverage.clone());
404                // Collect edge coverage and set the flag in the current run.
405                if corpus_manager.merge_edge_coverage(&mut call_result) {
406                    current_run.new_coverage = true;
407                }
408
409                if discarded {
410                    current_run.inputs.pop();
411                    current_run.rejects += 1;
412                    if current_run.rejects > self.config.max_assume_rejects {
413                        invariant_test.set_error(InvariantFuzzError::MaxAssumeRejects(
414                            self.config.max_assume_rejects,
415                        ));
416                        break 'stop;
417                    }
418                } else {
419                    // Commit executed call result.
420                    current_run.executor.commit(&mut call_result);
421
422                    // Collect data for fuzzing from the state changeset.
423                    // This step updates the state dictionary and therefore invalidates the
424                    // ValueTree in use by the current run. This manifestsitself in proptest
425                    // observing a different input case than what it was called with, and creates
426                    // inconsistencies whenever proptest tries to use the input case after test
427                    // execution.
428                    // See <https://github.com/foundry-rs/foundry/issues/9764>.
429                    let mut state_changeset = call_result.state_changeset.clone();
430                    if !call_result.reverted {
431                        collect_data(
432                            &invariant_test,
433                            &mut state_changeset,
434                            tx,
435                            &call_result,
436                            self.config.depth,
437                        );
438                    }
439
440                    // Collect created contracts and add to fuzz targets only if targeted contracts
441                    // are updatable.
442                    if let Err(error) =
443                        &invariant_test.targeted_contracts.collect_created_contracts(
444                            &state_changeset,
445                            self.project_contracts,
446                            self.setup_contracts,
447                            &self.artifact_filters,
448                            &mut current_run.created_contracts,
449                        )
450                    {
451                        warn!(target: "forge::test", "{error}");
452                    }
453                    current_run.fuzz_runs.push(FuzzCase {
454                        calldata: tx.call_details.calldata.clone(),
455                        gas: call_result.gas_used,
456                        stipend: call_result.stipend,
457                    });
458
459                    // Determine if test can continue or should exit.
460                    let result = can_continue(
461                        &invariant_contract,
462                        &mut invariant_test,
463                        &mut current_run,
464                        &self.config,
465                        call_result,
466                        &state_changeset,
467                    )
468                    .map_err(|e| eyre!(e.to_string()))?;
469                    if !result.can_continue || current_run.depth == self.config.depth - 1 {
470                        invariant_test.set_last_run_inputs(&current_run.inputs);
471                    }
472                    // If test cannot continue then stop current run and exit test suite.
473                    if !result.can_continue {
474                        break 'stop;
475                    }
476
477                    invariant_test.set_last_call_results(result.call_result);
478                    current_run.depth += 1;
479                }
480
481                current_run.inputs.push(corpus_manager.generate_next_input(
482                    &mut invariant_test.test_data.branch_runner,
483                    &initial_seq,
484                    discarded,
485                    current_run.depth as usize,
486                )?);
487            }
488
489            // Extend corpus with current run data.
490            corpus_manager.process_inputs(&current_run.inputs, current_run.new_coverage);
491
492            // Call `afterInvariant` only if it is declared and test didn't fail already.
493            if invariant_contract.call_after_invariant && !invariant_test.has_errors() {
494                assert_after_invariant(
495                    &invariant_contract,
496                    &mut invariant_test,
497                    &current_run,
498                    &self.config,
499                )
500                .map_err(|_| eyre!("Failed to call afterInvariant"))?;
501            }
502
503            // End current invariant test run.
504            invariant_test.end_run(current_run, self.config.gas_report_samples as usize);
505            if let Some(progress) = progress {
506                // If running with progress then increment completed runs.
507                progress.inc(1);
508                // Display metrics in progress bar.
509                if edge_coverage_enabled {
510                    progress.set_message(format!("{}", &corpus_manager.metrics));
511                }
512            } else if edge_coverage_enabled
513                && last_metrics_report.elapsed() > DURATION_BETWEEN_METRICS_REPORT
514            {
515                // Display metrics inline if corpus dir set.
516                let metrics = json!({
517                    "timestamp": SystemTime::now()
518                        .duration_since(UNIX_EPOCH)?
519                        .as_secs(),
520                    "invariant": invariant_contract.invariant_function.name,
521                    "metrics": &corpus_manager.metrics,
522                });
523                let _ = sh_println!("{}", serde_json::to_string(&metrics)?);
524                last_metrics_report = Instant::now();
525            }
526
527            runs += 1;
528        }
529
530        trace!(?fuzz_fixtures);
531        invariant_test.fuzz_state.log_stats();
532
533        let result = invariant_test.test_data;
534        Ok(InvariantFuzzTestResult {
535            error: result.failures.error,
536            cases: result.fuzz_cases,
537            reverts: result.failures.reverts,
538            last_run_inputs: result.last_run_inputs,
539            gas_report_traces: result.gas_report_traces,
540            line_coverage: result.line_coverage,
541            metrics: result.metrics,
542            failed_corpus_replays: corpus_manager.failed_replays(),
543        })
544    }

546    /// Prepares certain structures to execute the invariant tests:
547    /// * Invariant Fuzz Test.
548    /// * Invariant Corpus Manager.
549    fn prepare_test(
550        &mut self,
551        invariant_contract: &InvariantContract<'_>,
552        fuzz_fixtures: &FuzzFixtures,
553        deployed_libs: &[Address],
554    ) -> Result<(InvariantTest, CorpusManager)> {
555        // Finds out the chosen deployed contracts and/or senders.
556        self.select_contract_artifacts(invariant_contract.address)?;
557        let (targeted_senders, targeted_contracts) =
558            self.select_contracts_and_senders(invariant_contract.address)?;
559
560        // Stores fuzz state for use with [fuzz_calldata_from_state].
561        let fuzz_state = EvmFuzzState::new(
562            self.executor.backend().mem_db(),
563            self.config.dictionary,
564            deployed_libs,
565        );
566
567        // Creates the invariant strategy.
568        let strategy = invariant_strat(
569            fuzz_state.clone(),
570            targeted_senders,
571            targeted_contracts.clone(),
572            self.config.dictionary.dictionary_weight,
573            fuzz_fixtures.clone(),
574        )
575        .no_shrink();
576
577        // Allows `override_call_strat` to use the address given by the Fuzzer inspector during
578        // EVM execution.
579        let mut call_generator = None;
580        if self.config.call_override {
581            let target_contract_ref = Arc::new(RwLock::new(Address::ZERO));
582
583            call_generator = Some(RandomCallGenerator::new(
584                invariant_contract.address,
585                self.runner.clone(),
586                override_call_strat(
587                    fuzz_state.clone(),
588                    targeted_contracts.clone(),
589                    target_contract_ref.clone(),
590                    fuzz_fixtures.clone(),
591                ),
592                target_contract_ref,
593            ));
594        }
595
596        self.executor.inspector_mut().set_fuzzer(Fuzzer {
597            call_generator,
598            fuzz_state: fuzz_state.clone(),
599            collect: true,
600        });
601
602        // Let's make sure the invariant is sound before actually starting the run:
603        // We'll assert the invariant in its initial state, and if it fails, we'll
604        // already know if we can early exit the invariant run.
605        // This does not count as a fuzz run. It will just register the revert.
606        let mut failures = InvariantFailures::new();
607        let last_call_results = assert_invariants(
608            invariant_contract,
609            &self.config,
610            &targeted_contracts,
611            &self.executor,
612            &[],
613            &mut failures,
614        )?;
615        if let Some(error) = failures.error {
616            return Err(eyre!(error.revert_reason().unwrap_or_default()));
617        }
618
619        let corpus_manager = CorpusManager::new(
620            &self.config.corpus,
621            &invariant_contract.invariant_function.name,
622            strategy.boxed(),
623            &self.executor,
624            None,
625            Some(&targeted_contracts),
626        )?;
627        let invariant_test = InvariantTest::new(
628            fuzz_state,
629            targeted_contracts,
630            failures,
631            last_call_results,
632            self.runner.clone(),
633        );
634
635        Ok((invariant_test, corpus_manager))
636    }

638    /// Fills the `InvariantExecutor` with the artifact identifier filters (in `path:name` string
639    /// format). They will be used to filter contracts after the `setUp`, and more importantly,
640    /// during the runs.
641    ///
642    /// Also excludes any contract without any mutable functions.
643    ///
644    /// Priority:
645    ///
646    /// targetArtifactSelectors > excludeArtifacts > targetArtifacts
647    pub fn select_contract_artifacts(&mut self, invariant_address: Address) -> Result<()> {
648        let targeted_artifact_selectors = self
649            .executor
650            .call_sol_default(invariant_address, &IInvariantTest::targetArtifactSelectorsCall {});
651
652        // Insert them into the executor `targeted_abi`.
653        for IInvariantTest::FuzzArtifactSelector { artifact, selectors } in
654            targeted_artifact_selectors
655        {
656            let identifier = self.validate_selected_contract(artifact, &selectors)?;
657            self.artifact_filters.targeted.entry(identifier).or_default().extend(selectors);
658        }
659
660        let targeted_artifacts = self
661            .executor
662            .call_sol_default(invariant_address, &IInvariantTest::targetArtifactsCall {});
663        let excluded_artifacts = self
664            .executor
665            .call_sol_default(invariant_address, &IInvariantTest::excludeArtifactsCall {});
666
667        // Insert `excludeArtifacts` into the executor `excluded_abi`.
668        for contract in excluded_artifacts {
669            let identifier = self.validate_selected_contract(contract, &[])?;
670
671            if !self.artifact_filters.excluded.contains(&identifier) {
672                self.artifact_filters.excluded.push(identifier);
673            }
674        }
675
676        // Exclude any artifact without mutable functions.
677        for (artifact, contract) in self.project_contracts.iter() {
678            if contract
679                .abi
680                .functions()
681                .filter(|func| {
682                    !matches!(
683                        func.state_mutability,
684                        alloy_json_abi::StateMutability::Pure
685                            | alloy_json_abi::StateMutability::View
686                    )
687                })
688                .count()
689                == 0
690                && !self.artifact_filters.excluded.contains(&artifact.identifier())
691            {
692                self.artifact_filters.excluded.push(artifact.identifier());
693            }
694        }
695
696        // Insert `targetArtifacts` into the executor `targeted_abi`, if they have not been seen
697        // before.
698        for contract in targeted_artifacts {
699            let identifier = self.validate_selected_contract(contract, &[])?;
700
701            if !self.artifact_filters.targeted.contains_key(&identifier)
702                && !self.artifact_filters.excluded.contains(&identifier)
703            {
704                self.artifact_filters.targeted.insert(identifier, vec![]);
705            }
706        }
707        Ok(())
708    }

710    /// Makes sure that the contract exists in the project. If so, it returns its artifact
711    /// identifier.
712    fn validate_selected_contract(
713        &mut self,
714        contract: String,
715        selectors: &[FixedBytes<4>],
716    ) -> Result<String> {
717        if let Some((artifact, contract_data)) =
718            self.project_contracts.find_by_name_or_identifier(&contract)?
719        {
720            // Check that the selectors really exist for this contract.
721            for selector in selectors {
722                contract_data
723                    .abi
724                    .functions()
725                    .find(|func| func.selector().as_slice() == selector.as_slice())
726                    .wrap_err(format!("{contract} does not have the selector {selector:?}"))?;
727            }
728
729            return Ok(artifact.identifier());
730        }
731        eyre::bail!(
732            "{contract} not found in the project. Allowed format: `contract_name` or `contract_path:contract_name`."
733        );
734    }

736    /// Selects senders and contracts based on the contract methods `targetSenders() -> address[]`,
737    /// `targetContracts() -> address[]` and `excludeContracts() -> address[]`.
738    pub fn select_contracts_and_senders(
739        &self,
740        to: Address,
741    ) -> Result<(SenderFilters, FuzzRunIdentifiedContracts)> {
742        let targeted_senders =
743            self.executor.call_sol_default(to, &IInvariantTest::targetSendersCall {});
744        let mut excluded_senders =
745            self.executor.call_sol_default(to, &IInvariantTest::excludeSendersCall {});
746        // Extend with default excluded addresses - https://github.com/foundry-rs/foundry/issues/4163
747        excluded_senders.extend([
748            CHEATCODE_ADDRESS,
749            HARDHAT_CONSOLE_ADDRESS,
750            DEFAULT_CREATE2_DEPLOYER,
751        ]);
752        // Extend with precompiles - https://github.com/foundry-rs/foundry/issues/4287
753        excluded_senders.extend(PRECOMPILES);
754        let sender_filters = SenderFilters::new(targeted_senders, excluded_senders);
755
756        let selected = self.executor.call_sol_default(to, &IInvariantTest::targetContractsCall {});
757        let excluded = self.executor.call_sol_default(to, &IInvariantTest::excludeContractsCall {});
758
759        let contracts = self
760            .setup_contracts
761            .iter()
762            .filter(|&(addr, (identifier, _))| {
763                // Include to address if explicitly set as target.
764                if *addr == to && selected.contains(&to) {
765                    return true;
766                }
767
768                *addr != to
769                    && *addr != CHEATCODE_ADDRESS
770                    && *addr != HARDHAT_CONSOLE_ADDRESS
771                    && (selected.is_empty() || selected.contains(addr))
772                    && (excluded.is_empty() || !excluded.contains(addr))
773                    && self.artifact_filters.matches(identifier)
774            })
775            .map(|(addr, (identifier, abi))| {
776                (
777                    *addr,
778                    TargetedContract::new(identifier.clone(), abi.clone())
779                        .with_project_contracts(self.project_contracts),
780                )
781            })
782            .collect();
783        let mut contracts = TargetedContracts { inner: contracts };
784
785        self.target_interfaces(to, &mut contracts)?;
786
787        self.select_selectors(to, &mut contracts)?;
788
789        // There should be at least one contract identified as target for fuzz runs.
790        if contracts.is_empty() {
791            eyre::bail!("No contracts to fuzz.");
792        }
793
794        Ok((sender_filters, FuzzRunIdentifiedContracts::new(contracts, selected.is_empty())))
795    }

797    /// Extends the contracts and selectors to fuzz with the addresses and ABIs specified in
798    /// `targetInterfaces() -> (address, string[])[]`. Enables targeting of addresses that are
799    /// not deployed during `setUp` such as when fuzzing in a forked environment. Also enables
800    /// targeting of delegate proxies and contracts deployed with `create` or `create2`.
801    pub fn target_interfaces(
802        &self,
803        invariant_address: Address,
804        targeted_contracts: &mut TargetedContracts,
805    ) -> Result<()> {
806        let interfaces = self
807            .executor
808            .call_sol_default(invariant_address, &IInvariantTest::targetInterfacesCall {});
809
810        // Since `targetInterfaces` returns a tuple array there is no guarantee
811        // that the addresses are unique this map is used to merge functions of
812        // the specified interfaces for the same address. For example:
813        // `[(addr1, ["IERC20", "IOwnable"])]` and `[(addr1, ["IERC20"]), (addr1, ("IOwnable"))]`
814        // should be equivalent.
815        let mut combined = TargetedContracts::new();
816
817        // Loop through each address and its associated artifact identifiers.
818        // We're borrowing here to avoid taking full ownership.
819        for IInvariantTest::FuzzInterface { addr, artifacts } in &interfaces {
820            // Identifiers are specified as an array, so we loop through them.
821            for identifier in artifacts {
822                // Try to find the contract by name or identifier in the project's contracts.
823                if let Some((_, contract_data)) =
824                    self.project_contracts.iter().find(|(artifact, _)| {
825                        &artifact.name == identifier || &artifact.identifier() == identifier
826                    })
827                {
828                    let abi = &contract_data.abi;
829                    combined
830                        // Check if there's an entry for the given key in the 'combined' map.
831                        .entry(*addr)
832                        // If the entry exists, extends its ABI with the function list.
833                        .and_modify(|entry| {
834                            // Extend the ABI's function list with the new functions.
835                            entry.abi.functions.extend(abi.functions.clone());
836                        })
837                        // Otherwise insert it into the map.
838                        .or_insert_with(|| {
839                            let mut contract =
840                                TargetedContract::new(identifier.to_string(), abi.clone());
841                            contract.storage_layout =
842                                contract_data.storage_layout.as_ref().map(Arc::clone);
843                            contract
844                        });
845                }
846            }
847        }
848
849        targeted_contracts.extend(combined.inner);
850
851        Ok(())
852    }

854    /// Selects the functions to fuzz based on the contract method `targetSelectors()` and
855    /// `targetArtifactSelectors()`.
856    pub fn select_selectors(
857        &self,
858        address: Address,
859        targeted_contracts: &mut TargetedContracts,
860    ) -> Result<()> {
861        for (address, (identifier, _)) in self.setup_contracts {
862            if let Some(selectors) = self.artifact_filters.targeted.get(identifier) {
863                self.add_address_with_functions(*address, selectors, false, targeted_contracts)?;
864            }
865        }
866
867        let mut target_test_selectors = vec![];
868        let mut excluded_test_selectors = vec![];
869
870        // Collect contract functions marked as target for fuzzing campaign.
871        let selectors =
872            self.executor.call_sol_default(address, &IInvariantTest::targetSelectorsCall {});
873        for IInvariantTest::FuzzSelector { addr, selectors } in selectors {
874            if addr == address {
875                target_test_selectors = selectors.clone();
876            }
877            self.add_address_with_functions(addr, &selectors, false, targeted_contracts)?;
878        }
879
880        // Collect contract functions excluded from fuzzing campaign.
881        let excluded_selectors =
882            self.executor.call_sol_default(address, &IInvariantTest::excludeSelectorsCall {});
883        for IInvariantTest::FuzzSelector { addr, selectors } in excluded_selectors {
884            if addr == address {
885                // If fuzz selector address is the test contract, then record selectors to be
886                // later excluded if needed.
887                excluded_test_selectors = selectors.clone();
888            }
889            self.add_address_with_functions(addr, &selectors, true, targeted_contracts)?;
890        }
891
892        if target_test_selectors.is_empty()
893            && let Some(target) = targeted_contracts.get(&address)
894        {
895            // If test contract is marked as a target and no target selector explicitly set, then
896            // include only state-changing functions that are not reserved and selectors that are
897            // not explicitly excluded.
898            let selectors: Vec<_> = target
899                .abi
900                .functions()
901                .filter_map(|func| {
902                    if matches!(
903                        func.state_mutability,
904                        alloy_json_abi::StateMutability::Pure
905                            | alloy_json_abi::StateMutability::View
906                    ) || func.is_reserved()
907                        || excluded_test_selectors.contains(&func.selector())
908                    {
909                        None
910                    } else {
911                        Some(func.selector())
912                    }
913                })
914                .collect();
915            self.add_address_with_functions(address, &selectors, false, targeted_contracts)?;
916        }
917
918        Ok(())
919    }
920
921    /// Adds the address and fuzzed or excluded functions to `TargetedContracts`.
922    fn add_address_with_functions(
923        &self,
924        address: Address,
925        selectors: &[Selector],
926        should_exclude: bool,
927        targeted_contracts: &mut TargetedContracts,
928    ) -> eyre::Result<()> {
929        // Do not add address in target contracts if no function selected.
930        if selectors.is_empty() {
931            return Ok(());
932        }
933
934        let contract = match targeted_contracts.entry(address) {
935            Entry::Occupied(entry) => entry.into_mut(),
936            Entry::Vacant(entry) => {
937                let (identifier, abi) = self.setup_contracts.get(&address).ok_or_else(|| {
938                    eyre::eyre!(
939                        "[{}] address does not have an associated contract: {}",
940                        if should_exclude { "excludeSelectors" } else { "targetSelectors" },
941                        address
942                    )
943                })?;
944                entry.insert(
945                    TargetedContract::new(identifier.clone(), abi.clone())
946                        .with_project_contracts(self.project_contracts),
947                )
948            }
949        };
950        contract.add_selectors(selectors.iter().copied(), should_exclude)?;
951        Ok(())
952    }
953}

955/// Collects data from call for fuzzing. However, it first verifies that the sender is not an EOA
956/// before inserting it into the dictionary. Otherwise, we flood the dictionary with
957/// randomly generated addresses.
958fn collect_data(
959    invariant_test: &InvariantTest,
960    state_changeset: &mut HashMap<Address, Account>,
961    tx: &BasicTxDetails,
962    call_result: &RawCallResult,
963    run_depth: u32,
964) {
965    // Verify it has no code.
966    let mut has_code = false;
967    if let Some(Some(code)) =
968        state_changeset.get(&tx.sender).map(|account| account.info.code.as_ref())
969    {
970        has_code = !code.is_empty();
971    }
972
973    // We keep the nonce changes to apply later.
974    let mut sender_changeset = None;
975    if !has_code {
976        sender_changeset = state_changeset.remove(&tx.sender);
977    }
978
979    // Collect values from fuzzed call result and add them to fuzz dictionary.
980    invariant_test.fuzz_state.collect_values_from_call(
981        &invariant_test.targeted_contracts,
982        tx,
983        &call_result.result,
984        &call_result.logs,
985        &*state_changeset,
986        run_depth,
987    );
988
989    // Re-add changes
990    if let Some(changed) = sender_changeset {
991        state_changeset.insert(tx.sender, changed);
992    }
993}

995/// Calls the `afterInvariant()` function on a contract.
996/// Returns call result and if call succeeded.
997/// The state after the call is not persisted.
998pub(crate) fn call_after_invariant_function(
999    executor: &Executor,
1000    to: Address,
1001) -> Result<(RawCallResult, bool), EvmError> {
1002    let calldata = Bytes::from_static(&IInvariantTest::afterInvariantCall::SELECTOR);
1003    let mut call_result = executor.call_raw(CALLER, to, calldata, U256::ZERO)?;
1004    let success = executor.is_raw_call_mut_success(to, &mut call_result, false);
1005    Ok((call_result, success))
1006}
