76/// EVM executor.
77///
78/// The executor can be configured with various `revm::Inspector`s, like `Cheatcodes`.
79///
80/// There are multiple ways of interacting the EVM:
81/// - `call`: executes a transaction, but does not persist any state changes; similar to `eth_call`,
82///   where the EVM state is unchanged after the call.
83/// - `transact`: executes a transaction and persists the state changes
84/// - `deploy`: a special case of `transact`, specialized for persisting the state of a contract
85///   deployment
86/// - `setup`: a special case of `transact`, used to set up the environment for a test
87#[derive(Clone, Debug)]
88pub struct Executor {
89    /// The underlying `revm::Database` that contains the EVM storage.
90    // Note: We do not store an EVM here, since we are really
91    // only interested in the database. REVM's `EVM` is a thin
92    // wrapper around spawning a new EVM on every call anyway,
93    // so the performance difference should be negligible.
94    backend: Backend,
95    /// The EVM environment.
96    env: Env,
97    /// The Revm inspector stack.
98    inspector: InspectorStack,
99    /// The gas limit for calls and deployments.
100    gas_limit: u64,
101    /// Whether `failed()` should be called on the test contract to determine if the test failed.
102    legacy_assertions: bool,
103}

212    /// Creates the default CREATE2 Contract Deployer for local tests and scripts.
213    pub fn deploy_create2_deployer(&mut self) -> eyre::Result<()> {
214        trace!("deploying local create2 deployer");
215        let create2_deployer_account = self
216            .backend()
217            .basic_ref(DEFAULT_CREATE2_DEPLOYER)?
218            .ok_or_else(|| BackendError::MissingAccount(DEFAULT_CREATE2_DEPLOYER))?;
219
220        // If the deployer is not currently deployed, deploy the default one.
221        if create2_deployer_account.code.is_none_or(|code| code.is_empty()) {
222            let creator = DEFAULT_CREATE2_DEPLOYER_DEPLOYER;
223
224            // Probably 0, but just in case.
225            let initial_balance = self.get_balance(creator)?;
226            self.set_balance(creator, U256::MAX)?;
227
228            let res =
229                self.deploy(creator, DEFAULT_CREATE2_DEPLOYER_CODE.into(), U256::ZERO, None)?;
230            trace!(create2=?res.address, "deployed local create2 deployer");
231
232            self.set_balance(creator, initial_balance)?;
233        }
234        Ok(())
235    }

337    /// Deploys a contract using the given `env` and commits the new state to the underlying
338    /// database.
339    ///
340    /// # Panics
341    ///
342    /// Panics if `env.tx.kind` is not `TxKind::Create(_)`.
343    #[instrument(name = "deploy", level = "debug", skip_all)]
344    pub fn deploy_with_env(
345        &mut self,
346        env: Env,
347        rd: Option<&RevertDecoder>,
348    ) -> Result<DeployResult, EvmError> {
349        assert!(
350            matches!(env.tx.kind, TxKind::Create),
351            "Expected create transaction, got {:?}",
352            env.tx.kind
353        );
354        trace!(sender=%env.tx.caller, "deploying contract");
355
356        let mut result = self.transact_with_env(env)?;
357        result = result.into_result(rd)?;
358        let Some(Output::Create(_, Some(address))) = result.out else {
359            panic!("Deployment succeeded, but no address was returned: {result:#?}");
360        };
361
362        // also mark this library as persistent, this will ensure that the state of the library is
363        // persistent across fork swaps in forking mode
364        self.backend_mut().add_persistent_account(address);
365
366        debug!(%address, "deployed contract");
367
368        Ok(DeployResult { raw: result, address })
369    }

371    /// Calls the `setUp()` function on a contract.
372    ///
373    /// This will commit any state changes to the underlying database.
374    ///
375    /// Ayn changes made during the setup call to env's block environment are persistent, for
376    /// example `vm.chainId()` will change the `block.chainId` for all subsequent test calls.
377    #[instrument(name = "setup", level = "debug", skip_all)]
378    pub fn setup(
379        &mut self,
380        from: Option<Address>,
381        to: Address,
382        rd: Option<&RevertDecoder>,
383    ) -> Result<RawCallResult, EvmError> {
384        trace!(?from, ?to, "setting up contract");
385
386        let from = from.unwrap_or(CALLER);
387        self.backend_mut().set_test_contract(to).set_caller(from);
388        let calldata = Bytes::from_static(&ITest::setUpCall::SELECTOR);
389        let mut res = self.transact_raw(from, to, calldata, U256::ZERO)?;
390        res = res.into_result(rd)?;
391
392        // record any changes made to the block's environment during setup
393        self.env_mut().evm_env.block_env = res.env.evm_env.block_env.clone();
394        // and also the chainid, which can be set manually
395        self.env_mut().evm_env.cfg_env.chain_id = res.env.evm_env.cfg_env.chain_id;
396
397        let success =
398            self.is_raw_call_success(to, Cow::Borrowed(&res.state_changeset), &res, false);
399        if !success {
400            return Err(res.into_execution_error("execution error".to_string()).into());
401        }
402
403        Ok(res)
404    }

502    /// Execute the transaction configured in `env.tx`.
503    #[instrument(name = "transact", level = "debug", skip_all)]
504    pub fn transact_with_env(&mut self, mut env: Env) -> eyre::Result<RawCallResult> {
505        let mut stack = self.inspector().clone();
506        let backend = self.backend_mut();
507        let result = backend.inspect(&mut env, stack.as_inspector())?;
508        let mut result =
509            convert_executed_result(env, stack, result, backend.has_state_snapshot_failure())?;
510        self.commit(&mut result);
511        Ok(result)
512    }

514    /// Commit the changeset to the database and adjust `self.inspector_config` values according to
515    /// the executed call result.
516    ///
517    /// This should not be exposed to the user, as it should be called only by `transact*`.
518    #[instrument(name = "commit", level = "debug", skip_all)]
519    fn commit(&mut self, result: &mut RawCallResult) {
520        // Persist changes to db.
521        self.backend_mut().commit(result.state_changeset.clone());
522
523        // Persist cheatcode state.
524        self.inspector_mut().cheatcodes = result.cheatcodes.take();
525        if let Some(cheats) = self.inspector_mut().cheatcodes.as_mut() {
526            // Clear broadcastable transactions
527            cheats.broadcastable_transactions.clear();
528            cheats.ignored_traces.ignored.clear();
529
530            // if tracing was paused but never unpaused, we should begin next frame with tracing
531            // still paused
532            if let Some(last_pause_call) = cheats.ignored_traces.last_pause_call.as_mut() {
533                *last_pause_call = (0, 0);
534            }
535        }
536
537        // Persist the changed environment.
538        self.inspector_mut().set_env(&result.env);
539    }

559    /// Returns `true` if a test can be considered successful.
560    ///
561    /// This is the same as [`Self::is_success`], but intended for outcomes of [`Self::call_raw`].
562    pub fn is_raw_call_success(
563        &self,
564        address: Address,
565        state_changeset: Cow<'_, StateChangeset>,
566        call_result: &RawCallResult,
567        should_fail: bool,
568    ) -> bool {
569        if call_result.has_state_snapshot_failure {
570            // a failure occurred in a reverted snapshot, which is considered a failed test
571            return should_fail;
572        }
573        self.is_success(address, call_result.reverted, state_changeset, should_fail)
574    }

576    /// Returns `true` if a test can be considered successful.
577    ///
578    /// If the call succeeded, we also have to check the global and local failure flags.
579    ///
580    /// These are set by the test contract itself when an assertion fails, using the internal `fail`
581    /// function. The global flag is located in [`CHEATCODE_ADDRESS`] at slot [`GLOBAL_FAIL_SLOT`],
582    /// and the local flag is located in the test contract at an unspecified slot.
583    ///
584    /// This behavior is inherited from Dapptools, where initially only a public
585    /// `failed` variable was used to track test failures, and later, a global failure flag was
586    /// introduced to track failures across multiple contracts in
587    /// [ds-test#30](https://github.com/dapphub/ds-test/pull/30).
588    ///
589    /// The assumption is that the test runner calls `failed` on the test contract to determine if
590    /// it failed. However, we want to avoid this as much as possible, as it is relatively
591    /// expensive to set up an EVM call just for checking a single boolean flag.
592    ///
593    /// See:
594    /// - Newer DSTest: <https://github.com/dapphub/ds-test/blob/e282159d5170298eb2455a6c05280ab5a73a4ef0/src/test.sol#L47-L63>
595    /// - Older DSTest: <https://github.com/dapphub/ds-test/blob/9ca4ecd48862b40d7b0197b600713f64d337af12/src/test.sol#L38-L49>
596    /// - forge-std: <https://github.com/foundry-rs/forge-std/blob/19891e6a0b5474b9ea6827ddb90bb9388f7acfc0/src/StdAssertions.sol#L38-L44>
597    pub fn is_success(
598        &self,
599        address: Address,
600        reverted: bool,
601        state_changeset: Cow<'_, StateChangeset>,
602        should_fail: bool,
603    ) -> bool {
604        let success = self.is_success_raw(address, reverted, state_changeset);
605        should_fail ^ success
606    }

608    #[instrument(name = "is_success", level = "debug", skip_all)]
609    fn is_success_raw(
610        &self,
611        address: Address,
612        reverted: bool,
613        state_changeset: Cow<'_, StateChangeset>,
614    ) -> bool {
615        // The call reverted.
616        if reverted {
617            return false;
618        }
619
620        // A failure occurred in a reverted snapshot, which is considered a failed test.
621        if self.backend().has_state_snapshot_failure() {
622            return false;
623        }
624
625        // Check the global failure slot.
626        if let Some(acc) = state_changeset.get(&CHEATCODE_ADDRESS)
627            && let Some(failed_slot) = acc.storage.get(&GLOBAL_FAIL_SLOT)
628            && !failed_slot.present_value().is_zero()
629        {
630            return false;
631        }
632        if let Ok(failed_slot) = self.backend().storage_ref(CHEATCODE_ADDRESS, GLOBAL_FAIL_SLOT)
633            && !failed_slot.is_zero()
634        {
635            return false;
636        }
637
638        if !self.legacy_assertions {
639            return true;
640        }
641
642        // Finally, resort to calling `DSTest::failed`.
643        {
644            // Construct a new bare-bones backend to evaluate success.
645            let mut backend = self.backend().clone_empty();
646
647            // We only clone the test contract and cheatcode accounts,
648            // that's all we need to evaluate success.
649            for address in [address, CHEATCODE_ADDRESS] {
650                let Ok(acc) = self.backend().basic_ref(address) else { return false };
651                backend.insert_account_info(address, acc.unwrap_or_default());
652            }
653
654            // If this test failed any asserts, then this changeset will contain changes
655            // `false -> true` for the contract's `failed` variable and the `globalFailure` flag
656            // in the state of the cheatcode address,
657            // which are both read when we call `"failed()(bool)"` in the next step.
658            backend.commit(state_changeset.into_owned());
659
660            // Check if a DSTest assertion failed
661            let executor = self.clone_with_backend(backend);
662            let call = executor.call_sol(CALLER, address, &ITest::failedCall {}, U256::ZERO, None);
663            match call {
664                Ok(CallResult { raw: _, decoded_result: failed }) => {
665                    trace!(failed, "DSTest::failed()");
666                    !failed
667                }
668                Err(err) => {
669                    trace!(%err, "failed to call DSTest::failed()");
670                    true
671                }
672            }
673        }
674    }
675
676    /// Creates the environment to use when executing a transaction in a test context
677    ///
678    /// If using a backend with cheatcodes, `tx.gas_price` and `block.number` will be overwritten by
679    /// the cheatcode state in between calls.
680    fn build_test_env(&self, caller: Address, kind: TxKind, data: Bytes, value: U256) -> Env {
681        Env {
682            evm_env: EvmEnv {
683                cfg_env: {
684                    let mut cfg = self.env().evm_env.cfg_env.clone();
685                    cfg.spec = self.spec_id();
686                    cfg
687                },
688                // We always set the gas price to 0 so we can execute the transaction regardless of
689                // network conditions - the actual gas price is kept in `self.block` and is applied
690                // by the cheatcode handler if it is enabled
691                block_env: BlockEnv {
692                    basefee: 0,
693                    gas_limit: self.gas_limit,
694                    ..self.env().evm_env.block_env.clone()
695                },
696            },
697            tx: TxEnv {
698                caller,
699                kind,
700                data,
701                value,
702                // As above, we set the gas price to 0.
703                gas_price: 0,
704                gas_priority_fee: None,
705                gas_limit: self.gas_limit,
706                chain_id: Some(self.env().evm_env.cfg_env.chain_id),
707                ..self.env().tx.clone()
708            },
709        }
710    }
711
712    pub fn call_sol_default<C: SolCall>(&self, to: Address, args: &C) -> C::Return
713    where
714        C::Return: Default,
715    {
716        self.call_sol(CALLER, to, args, U256::ZERO, None)
717            .map(|c| c.decoded_result)
718            .inspect_err(|e| warn!(target: "forge::test", "failed calling {:?}: {e}", C::SIGNATURE))
719            .unwrap_or_default()
720    }
721}

883impl RawCallResult {
884    /// Unpacks an EVM result.
885    pub fn from_evm_result(r: Result<Self, EvmError>) -> eyre::Result<(Self, Option<String>)> {
886        match r {
887            Ok(r) => Ok((r, None)),
888            Err(EvmError::Execution(e)) => Ok((e.raw, Some(e.reason))),
889            Err(e) => Err(e.into()),
890        }
891    }
892
893    /// Unpacks an execution result.
894    pub fn from_execution_result(r: Result<Self, ExecutionErr>) -> (Self, Option<String>) {
895        match r {
896            Ok(r) => (r, None),
897            Err(e) => (e.raw, Some(e.reason)),
898        }
899    }
900
901    /// Converts the result of the call into an `EvmError`.
902    pub fn into_evm_error(self, rd: Option<&RevertDecoder>) -> EvmError {
903        if let Some(reason) = SkipReason::decode(&self.result) {
904            return EvmError::Skip(reason);
905        }
906        let reason = rd.unwrap_or_default().decode(&self.result, self.exit_reason);
907        EvmError::Execution(Box::new(self.into_execution_error(reason)))
908    }
909
910    /// Converts the result of the call into an `ExecutionErr`.
911    pub fn into_execution_error(self, reason: String) -> ExecutionErr {
912        ExecutionErr { raw: self, reason }
913    }
914
915    /// Returns an `EvmError` if the call failed, otherwise returns `self`.
916    pub fn into_result(self, rd: Option<&RevertDecoder>) -> Result<Self, EvmError> {
917        if let Some(reason) = self.exit_reason
918            && reason.is_ok()
919        {
920            Ok(self)
921        } else {
922            Err(self.into_evm_error(rd))
923        }
924    }
925
926    /// Decodes the result of the call with the given function.
927    pub fn into_decoded_result(
928        mut self,
929        func: &Function,
930        rd: Option<&RevertDecoder>,
931    ) -> Result<CallResult, EvmError> {
932        self = self.into_result(rd)?;
933        let mut result = func.abi_decode_output(&self.result)?;
934        let decoded_result = if result.len() == 1 {
935            result.pop().unwrap()
936        } else {
937            // combine results into a tuple
938            DynSolValue::Tuple(result)
939        };
940        Ok(CallResult { raw: self, decoded_result })
941    }
942
943    /// Returns the transactions generated from this call.
944    pub fn transactions(&self) -> Option<&BroadcastableTransactions> {
945        self.cheatcodes.as_ref().map(|c| &c.broadcastable_transactions)
946    }
947
948    /// Update provided history map with edge coverage info collected during this call.
949    /// Uses AFL binning algo <https://github.com/h0mbre/Lucid/blob/3026e7323c52b30b3cf12563954ac1eaa9c6981e/src/coverage.rs#L57-L85>
950    pub fn merge_edge_coverage(&mut self, history_map: &mut [u8]) -> (bool, bool) {
951        let mut new_coverage = false;
952        let mut is_edge = false;
953        if let Some(x) = &mut self.edge_coverage {
954            // Iterate over the current map and the history map together and update
955            // the history map, if we discover some new coverage, report true
956            for (curr, hist) in std::iter::zip(x, history_map) {
957                // If we got a hitcount of at least 1
958                if *curr > 0 {
959                    // Convert hitcount into bucket count
960                    let bucket = match *curr {
961                        0 => 0,
962                        1 => 1,
963                        2 => 2,
964                        3 => 4,
965                        4..=7 => 8,
966                        8..=15 => 16,
967                        16..=31 => 32,
968                        32..=127 => 64,
969                        128..=255 => 128,
970                    };
971
972                    // If the old record for this edge pair is lower, update
973                    if *hist < bucket {
974                        if *hist == 0 {
975                            // Counts as an edge the first time we see it, otherwise it's a feature.
976                            is_edge = true;
977                        }
978                        *hist = bucket;
979                        new_coverage = true;
980                    }
981
982                    // Zero out the current map for next iteration.
983                    *curr = 0;
984                }
985            }
986        }
987        (new_coverage, is_edge)
988    }
989}

1015/// Converts the data aggregated in the `inspector` and `call` to a `RawCallResult`
1016fn convert_executed_result(
1017    env: Env,
1018    inspector: InspectorStack,
1019    ResultAndState { result, state: state_changeset }: ResultAndState,
1020    has_state_snapshot_failure: bool,
1021) -> eyre::Result<RawCallResult> {
1022    let (exit_reason, gas_refunded, gas_used, out, exec_logs) = match result {
1023        ExecutionResult::Success { reason, gas_used, gas_refunded, output, logs, .. } => {
1024            (reason.into(), gas_refunded, gas_used, Some(output), logs)
1025        }
1026        ExecutionResult::Revert { gas_used, output } => {
1027            // Need to fetch the unused gas
1028            (InstructionResult::Revert, 0_u64, gas_used, Some(Output::Call(output)), vec![])
1029        }
1030        ExecutionResult::Halt { reason, gas_used } => {
1031            (reason.into(), 0_u64, gas_used, None, vec![])
1032        }
1033    };
1034    let gas = revm::interpreter::gas::calculate_initial_tx_gas(
1035        env.evm_env.cfg_env.spec,
1036        &env.tx.data,
1037        env.tx.kind.is_create(),
1038        env.tx.access_list.len().try_into()?,
1039        0,
1040        0,
1041    );
1042
1043    let result = match &out {
1044        Some(Output::Call(data)) => data.clone(),
1045        _ => Bytes::new(),
1046    };
1047
1048    let InspectorData {
1049        mut logs,
1050        labels,
1051        traces,
1052        line_coverage,
1053        edge_coverage,
1054        cheatcodes,
1055        chisel_state,
1056        reverter,
1057    } = inspector.collect();
1058
1059    if logs.is_empty() {
1060        logs = exec_logs;
1061    }
1062
1063    let transactions = cheatcodes
1064        .as_ref()
1065        .map(|c| c.broadcastable_transactions.clone())
1066        .filter(|txs| !txs.is_empty());
1067
1068    Ok(RawCallResult {
1069        exit_reason: Some(exit_reason),
1070        reverted: !matches!(exit_reason, return_ok!()),
1071        has_state_snapshot_failure,
1072        result,
1073        gas_used,
1074        gas_refunded,
1075        stipend: gas.initial_gas,
1076        logs,
1077        labels,
1078        traces,
1079        line_coverage,
1080        edge_coverage,
1081        transactions,
1082        state_changeset,
1083        env,
1084        cheatcodes,
1085        out,
1086        chisel_state,
1087        reverter,
1088    })
1089}
