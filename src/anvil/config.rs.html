206    fn as_string(&self, fork: Option<&ClientFork>) -> String {
207        let mut s: String = String::new();
208        let _ = write!(s, "\n{}", BANNER.green());
209        let _ = write!(s, "\n    {VERSION_MESSAGE}");
210        let _ = write!(s, "\n    {}", "https://github.com/foundry-rs/foundry".green());
211
212        let _ = write!(
213            s,
214            r#"
215
216Available Accounts
217==================
218"#
219        );
220        let balance = alloy_primitives::utils::format_ether(self.genesis_balance);
221        for (idx, wallet) in self.genesis_accounts.iter().enumerate() {
222            write!(s, "\n({idx}) {} ({balance} ETH)", wallet.address()).unwrap();
223        }
224
225        let _ = write!(
226            s,
227            r#"
228
229Private Keys
230==================
231"#
232        );
233
234        for (idx, wallet) in self.genesis_accounts.iter().enumerate() {
235            let hex = hex::encode(wallet.credential().to_bytes());
236            let _ = write!(s, "\n({idx}) 0x{hex}");
237        }
238
239        if let Some(generator) = &self.account_generator {
240            let _ = write!(
241                s,
242                r#"
243
244Wallet
245==================
246Mnemonic:          {}
247Derivation path:   {}
248"#,
249                generator.phrase,
250                generator.get_derivation_path()
251            );
252        }
253
254        if let Some(fork) = fork {
255            let _ = write!(
256                s,
257                r#"
258
259Fork
260==================
261Endpoint:       {}
262Block number:   {}
263Block hash:     {:?}
264Chain ID:       {}
265"#,
266                fork.eth_rpc_url(),
267                fork.block_number(),
268                fork.block_hash(),
269                fork.chain_id()
270            );
271
272            if let Some(tx_hash) = fork.transaction_hash() {
273                let _ = writeln!(s, "Transaction hash: {tx_hash}");
274            }
275        } else {
276            let _ = write!(
277                s,
278                r#"
279
280Chain ID
281==================
282
283{}
284"#,
285                self.get_chain_id().green()
286            );
287        }
288
289        if (SpecId::from(self.get_hardfork()) as u8) < (SpecId::LONDON as u8) {
290            let _ = write!(
291                s,
292                r#"
293Gas Price
294==================
295
296{}
297"#,
298                self.get_gas_price().green()
299            );
300        } else {
301            let _ = write!(
302                s,
303                r#"
304Base Fee
305==================
306
307{}
308"#,
309                self.get_base_fee().green()
310            );
311        }
312
313        let _ = write!(
314            s,
315            r#"
316Gas Limit
317==================
318
319{}
320"#,
321            {
322                if self.disable_block_gas_limit {
323                    "Disabled".to_string()
324                } else {
325                    self.gas_limit.map(|l| l.to_string()).unwrap_or_else(|| {
326                        if self.fork_choice.is_some() {
327                            "Forked".to_string()
328                        } else {
329                            DEFAULT_GAS_LIMIT.to_string()
330                        }
331                    })
332                }
333            }
334            .green()
335        );
336
337        let _ = write!(
338            s,
339            r#"
340Genesis Timestamp
341==================
342
343{}
344"#,
345            self.get_genesis_timestamp().green()
346        );
347
348        let _ = write!(
349            s,
350            r#"
351Genesis Number
352==================
353
354{}
355"#,
356            self.get_genesis_number().green()
357        );
358
359        s
360    }

362    fn as_json(&self, fork: Option<&ClientFork>) -> Value {
363        let mut wallet_description = HashMap::new();
364        let mut available_accounts = Vec::with_capacity(self.genesis_accounts.len());
365        let mut private_keys = Vec::with_capacity(self.genesis_accounts.len());
366
367        for wallet in &self.genesis_accounts {
368            available_accounts.push(format!("{:?}", wallet.address()));
369            private_keys.push(format!("0x{}", hex::encode(wallet.credential().to_bytes())));
370        }
371
372        if let Some(generator) = &self.account_generator {
373            let phrase = generator.get_phrase().to_string();
374            let derivation_path = generator.get_derivation_path().to_string();
375
376            wallet_description.insert("derivation_path".to_string(), derivation_path);
377            wallet_description.insert("mnemonic".to_string(), phrase);
378        };
379
380        let gas_limit = match self.gas_limit {
381            // if we have a disabled flag we should max out the limit
382            Some(_) | None if self.disable_block_gas_limit => Some(u64::MAX.to_string()),
383            Some(limit) => Some(limit.to_string()),
384            _ => None,
385        };
386
387        if let Some(fork) = fork {
388            json!({
389              "available_accounts": available_accounts,
390              "private_keys": private_keys,
391              "endpoint": fork.eth_rpc_url(),
392              "block_number": fork.block_number(),
393              "block_hash": fork.block_hash(),
394              "chain_id": fork.chain_id(),
395              "wallet": wallet_description,
396              "base_fee": format!("{}", self.get_base_fee()),
397              "gas_price": format!("{}", self.get_gas_price()),
398              "gas_limit": gas_limit,
399            })
400        } else {
401            json!({
402              "available_accounts": available_accounts,
403              "private_keys": private_keys,
404              "wallet": wallet_description,
405              "base_fee": format!("{}", self.get_base_fee()),
406              "gas_price": format!("{}", self.get_gas_price()),
407              "gas_limit": gas_limit,
408              "genesis_timestamp": format!("{}", self.get_genesis_timestamp()),
409            })
410        }
411    }

1039    /// Configures everything related to env, backend and database and returns the
1040    /// [Backend](mem::Backend)
1041    ///
1042    /// *Note*: only memory based backend for now
1043    pub(crate) async fn setup(&mut self) -> Result<mem::Backend> {
1044        // configure the revm environment
1045
1046        let mut cfg = CfgEnv::default();
1047        cfg.spec = self.get_hardfork().into();
1048
1049        cfg.chain_id = self.get_chain_id();
1050        cfg.limit_contract_code_size = self.code_size_limit;
1051        // EIP-3607 rejects transactions from senders with deployed code.
1052        // If EIP-3607 is enabled it can cause issues during fuzz/invariant tests if the
1053        // caller is a contract. So we disable the check by default.
1054        cfg.disable_eip3607 = true;
1055        cfg.disable_block_gas_limit = self.disable_block_gas_limit;
1056
1057        if let Some(value) = self.memory_limit {
1058            cfg.memory_limit = value;
1059        }
1060
1061        let spec_id = cfg.spec;
1062        let mut env = Env::new(
1063            cfg,
1064            BlockEnv {
1065                gas_limit: self.gas_limit(),
1066                basefee: self.get_base_fee(),
1067                ..Default::default()
1068            },
1069            OpTransaction {
1070                base: TxEnv { chain_id: Some(self.get_chain_id()), ..Default::default() },
1071                ..Default::default()
1072            },
1073            self.enable_optimism,
1074        );
1075
1076        let fees = FeeManager::new(
1077            spec_id,
1078            self.get_base_fee(),
1079            !self.disable_min_priority_fee,
1080            self.get_gas_price(),
1081            self.get_blob_excess_gas_and_price(),
1082        );
1083
1084        let (db, fork): (Arc<TokioRwLock<Box<dyn Db>>>, Option<ClientFork>) =
1085            if let Some(eth_rpc_url) = self.eth_rpc_url.clone() {
1086                self.setup_fork_db(eth_rpc_url, &mut env, &fees).await?
1087            } else {
1088                (Arc::new(TokioRwLock::new(Box::<MemDb>::default())), None)
1089            };
1090
1091        // if provided use all settings of `genesis.json`
1092        if let Some(ref genesis) = self.genesis {
1093            // --chain-id flag gets precedence over the genesis.json chain id
1094            // <https://github.com/foundry-rs/foundry/issues/10059>
1095            if self.chain_id.is_none() {
1096                env.evm_env.cfg_env.chain_id = genesis.config.chain_id;
1097            }
1098            env.evm_env.block_env.timestamp = U256::from(genesis.timestamp);
1099            if let Some(base_fee) = genesis.base_fee_per_gas {
1100                env.evm_env.block_env.basefee = base_fee.try_into()?;
1101            }
1102            if let Some(number) = genesis.number {
1103                env.evm_env.block_env.number = U256::from(number);
1104            }
1105            env.evm_env.block_env.beneficiary = genesis.coinbase;
1106        }
1107
1108        let genesis = GenesisConfig {
1109            number: self.get_genesis_number(),
1110            timestamp: self.get_genesis_timestamp(),
1111            balance: self.genesis_balance,
1112            accounts: self.genesis_accounts.iter().map(|acc| acc.address()).collect(),
1113            genesis_init: self.genesis.clone(),
1114        };
1115
1116        let mut decoder_builder = CallTraceDecoderBuilder::new();
1117        if self.print_traces {
1118            // if traces should get printed we configure the decoder with the signatures cache
1119            if let Ok(identifier) = SignaturesIdentifier::new(false) {
1120                debug!(target: "node", "using signature identifier");
1121                decoder_builder = decoder_builder.with_signature_identifier(identifier);
1122            }
1123        }
1124
1125        // only memory based backend for now
1126        let backend = mem::Backend::with_genesis(
1127            db,
1128            Arc::new(RwLock::new(env)),
1129            genesis,
1130            fees,
1131            Arc::new(RwLock::new(fork)),
1132            self.enable_steps_tracing,
1133            self.print_logs,
1134            self.print_traces,
1135            Arc::new(decoder_builder.build()),
1136            self.odyssey,
1137            self.prune_history,
1138            self.max_persisted_states,
1139            self.transaction_block_keeper,
1140            self.block_time,
1141            self.cache_path.clone(),
1142            Arc::new(TokioRwLock::new(self.clone())),
1143        )
1144        .await?;
1145
1146        // Writes the default create2 deployer to the backend,
1147        // if the option is not disabled and we are not forking.
1148        if !self.disable_default_create2_deployer && self.eth_rpc_url.is_none() {
1149            backend
1150                .set_create2_deployer(DEFAULT_CREATE2_DEPLOYER)
1151                .await
1152                .wrap_err("failed to create default create2 deployer")?;
1153        }
1154
1155        if let Some(state) = self.init_state.clone() {
1156            backend.load_state(state).await.wrap_err("failed to load init state")?;
1157        }
1158
1159        Ok(backend)
1160    }

1162    /// Configures everything related to forking based on the passed `eth_rpc_url`:
1163    ///  - returning a tuple of a [ForkedDatabase] wrapped in an [Arc] [RwLock](TokioRwLock) and
1164    ///    [ClientFork] wrapped in an [Option] which can be used in a [Backend](mem::Backend) to
1165    ///    fork from.
1166    ///  - modifying some parameters of the passed `env`
1167    ///  - mutating some members of `self`
1168    pub async fn setup_fork_db(
1169        &mut self,
1170        eth_rpc_url: String,
1171        env: &mut Env,
1172        fees: &FeeManager,
1173    ) -> Result<(Arc<TokioRwLock<Box<dyn Db>>>, Option<ClientFork>)> {
1174        let (db, config) = self.setup_fork_db_config(eth_rpc_url, env, fees).await?;
1175        let db: Arc<TokioRwLock<Box<dyn Db>>> = Arc::new(TokioRwLock::new(Box::new(db)));
1176        let fork = ClientFork::new(config, Arc::clone(&db));
1177        Ok((db, Some(fork)))
1178    }
1179
1180    /// Configures everything related to forking based on the passed `eth_rpc_url`:
1181    ///  - returning a tuple of a [ForkedDatabase] and [ClientForkConfig] which can be used to build
1182    ///    a [ClientFork] to fork from.
1183    ///  - modifying some parameters of the passed `env`
1184    ///  - mutating some members of `self`
1185    pub async fn setup_fork_db_config(
1186        &mut self,
1187        eth_rpc_url: String,
1188        env: &mut Env,
1189        fees: &FeeManager,
1190    ) -> Result<(ForkedDatabase, ClientForkConfig)> {
1191        debug!(target: "node", ?eth_rpc_url, "setting up fork db");
1192        let provider = Arc::new(
1193            ProviderBuilder::new(&eth_rpc_url)
1194                .timeout(self.fork_request_timeout)
1195                .initial_backoff(self.fork_retry_backoff.as_millis() as u64)
1196                .compute_units_per_second(self.compute_units_per_second)
1197                .max_retry(self.fork_request_retries)
1198                .initial_backoff(1000)
1199                .headers(self.fork_headers.clone())
1200                .build()
1201                .wrap_err("failed to establish provider to fork url")?,
1202        );
1203
1204        let (fork_block_number, fork_chain_id, force_transactions) = if let Some(fork_choice) =
1205            &self.fork_choice
1206        {
1207            let (fork_block_number, force_transactions) =
1208                derive_block_and_transactions(fork_choice, &provider).await.wrap_err(
1209                    "failed to derive fork block number and force transactions from fork choice",
1210                )?;
1211            let chain_id = if let Some(chain_id) = self.fork_chain_id {
1212                Some(chain_id)
1213            } else if self.hardfork.is_none() {
1214                // Auto-adjust hardfork if not specified, but only if we're forking mainnet.
1215                let chain_id =
1216                    provider.get_chain_id().await.wrap_err("failed to fetch network chain ID")?;
1217                if alloy_chains::NamedChain::Mainnet == chain_id {
1218                    let hardfork: EthereumHardfork =
1219                        ethereum_hardfork_from_block_tag(fork_block_number);
1220
1221                    env.evm_env.cfg_env.spec = spec_id_from_ethereum_hardfork(hardfork);
1222                    self.hardfork = Some(ChainHardfork::Ethereum(hardfork));
1223                }
1224                Some(U256::from(chain_id))
1225            } else {
1226                None
1227            };
1228
1229            (fork_block_number, chain_id, force_transactions)
1230        } else {
1231            // pick the last block number but also ensure it's not pending anymore
1232            let bn = find_latest_fork_block(&provider)
1233                .await
1234                .wrap_err("failed to get fork block number")?;
1235            (bn, None, None)
1236        };
1237
1238        let block = provider
1239            .get_block(BlockNumberOrTag::Number(fork_block_number).into())
1240            .await
1241            .wrap_err("failed to get fork block")?;
1242
1243        let block = if let Some(block) = block {
1244            block
1245        } else {
1246            if let Ok(latest_block) = provider.get_block_number().await {
1247                let mut message = format!(
1248                    "Failed to get block for block number: {fork_block_number}\n\
1249latest block number: {latest_block}"
1250                );
1251                // If the `eth_getBlockByNumber` call succeeds, but returns null instead of
1252                // the block, and the block number is less than equal the latest block, then
1253                // the user is forking from a non-archive node with an older block number.
1254                if fork_block_number <= latest_block {
1255                    message.push_str(&format!("\n{NON_ARCHIVE_NODE_WARNING}"));
1256                }
1257                eyre::bail!("{message}");
1258            }
1259            eyre::bail!("failed to get block for block number: {fork_block_number}")
1260        };
1261
1262        let gas_limit = self.fork_gas_limit(&block);
1263        self.gas_limit = Some(gas_limit);
1264
1265        env.evm_env.block_env = BlockEnv {
1266            number: U256::from(fork_block_number),
1267            timestamp: U256::from(block.header.timestamp),
1268            difficulty: block.header.difficulty,
1269            // ensures prevrandao is set
1270            prevrandao: Some(block.header.mix_hash.unwrap_or_default()),
1271            gas_limit,
1272            // Keep previous `coinbase` and `basefee` value
1273            beneficiary: env.evm_env.block_env.beneficiary,
1274            basefee: env.evm_env.block_env.basefee,
1275            ..Default::default()
1276        };
1277
1278        // if not set explicitly we use the base fee of the latest block
1279        if self.base_fee.is_none() {
1280            if let Some(base_fee) = block.header.base_fee_per_gas {
1281                self.base_fee = Some(base_fee);
1282                env.evm_env.block_env.basefee = base_fee;
1283                // this is the base fee of the current block, but we need the base fee of
1284                // the next block
1285                let next_block_base_fee = fees.get_next_block_base_fee_per_gas(
1286                    block.header.gas_used,
1287                    gas_limit,
1288                    block.header.base_fee_per_gas.unwrap_or_default(),
1289                );
1290
1291                // update next base fee
1292                fees.set_base_fee(next_block_base_fee);
1293            }
1294            if let (Some(blob_excess_gas), Some(blob_gas_used)) =
1295                (block.header.excess_blob_gas, block.header.blob_gas_used)
1296            {
1297                let blob_base_fee_update_fraction = get_blob_base_fee_update_fraction(
1298                    fork_chain_id
1299                        .unwrap_or_else(|| U256::from(Chain::mainnet().id()))
1300                        .saturating_to(),
1301                    block.header.timestamp,
1302                );
1303
1304                env.evm_env.block_env.blob_excess_gas_and_price = Some(BlobExcessGasAndPrice::new(
1305                    blob_excess_gas,
1306                    blob_base_fee_update_fraction,
1307                ));
1308
1309                let next_block_blob_excess_gas =
1310                    fees.get_next_block_blob_excess_gas(blob_excess_gas, blob_gas_used);
1311
1312                fees.set_blob_excess_gas_and_price(BlobExcessGasAndPrice::new(
1313                    next_block_blob_excess_gas,
1314                    blob_base_fee_update_fraction,
1315                ));
1316            }
1317        }
1318
1319        // use remote gas price
1320        if self.gas_price.is_none()
1321            && let Ok(gas_price) = provider.get_gas_price().await
1322        {
1323            self.gas_price = Some(gas_price);
1324            fees.set_gas_price(gas_price);
1325        }
1326
1327        let block_hash = block.header.hash;
1328
1329        let chain_id = if let Some(chain_id) = self.chain_id {
1330            chain_id
1331        } else {
1332            let chain_id = if let Some(fork_chain_id) = fork_chain_id {
1333                fork_chain_id.to()
1334            } else {
1335                provider.get_chain_id().await.wrap_err("failed to fetch network chain ID")?
1336            };
1337
1338            // need to update the dev signers and env with the chain id
1339            self.set_chain_id(Some(chain_id));
1340            env.evm_env.cfg_env.chain_id = chain_id;
1341            env.tx.base.chain_id = chain_id.into();
1342            chain_id
1343        };
1344        let override_chain_id = self.chain_id;
1345        // apply changes such as difficulty -> prevrandao and chain specifics for current chain id
1346        apply_chain_and_block_specific_env_changes::<AnyNetwork>(env.as_env_mut(), &block);
1347
1348        let meta = BlockchainDbMeta::new(env.evm_env.block_env.clone(), eth_rpc_url.clone());
1349        let block_chain_db = if self.fork_chain_id.is_some() {
1350            BlockchainDb::new_skip_check(meta, self.block_cache_path(fork_block_number))
1351        } else {
1352            BlockchainDb::new(meta, self.block_cache_path(fork_block_number))
1353        };
1354
1355        // This will spawn the background thread that will use the provider to fetch
1356        // blockchain data from the other client
1357        let backend = SharedBackend::spawn_backend(
1358            Arc::clone(&provider),
1359            block_chain_db.clone(),
1360            Some(fork_block_number.into()),
1361        )
1362        .await;
1363
1364        let config = ClientForkConfig {
1365            eth_rpc_url,
1366            block_number: fork_block_number,
1367            block_hash,
1368            transaction_hash: self.fork_choice.and_then(|fc| fc.transaction_hash()),
1369            provider,
1370            chain_id,
1371            override_chain_id,
1372            timestamp: block.header.timestamp,
1373            base_fee: block.header.base_fee_per_gas.map(|g| g as u128),
1374            timeout: self.fork_request_timeout,
1375            retries: self.fork_request_retries,
1376            backoff: self.fork_retry_backoff,
1377            compute_units_per_second: self.compute_units_per_second,
1378            total_difficulty: block.header.total_difficulty.unwrap_or_default(),
1379            blob_gas_used: block.header.blob_gas_used.map(|g| g as u128),
1380            blob_excess_gas_and_price: env.evm_env.block_env.blob_excess_gas_and_price,
1381            force_transactions,
1382        };
1383
1384        debug!(target: "node", fork_number=config.block_number, fork_hash=%config.block_hash, "set up fork db");
1385
1386        let mut db = ForkedDatabase::new(backend, block_chain_db);
1387
1388        // need to insert the forked block's hash
1389        db.insert_block_hash(U256::from(config.block_number), config.block_hash);
1390
1391        Ok((db, config))
1392    }
1393
1394    /// we only use the gas limit value of the block if it is non-zero and the block gas
1395    /// limit is enabled, since there are networks where this is not used and is always
1396    /// `0x0` which would inevitably result in `OutOfGas` errors as soon as the evm is about to record gas, See also <https://github.com/foundry-rs/foundry/issues/3247>
1397    pub(crate) fn fork_gas_limit<T: TransactionResponse, H: BlockHeader>(
1398        &self,
1399        block: &Block<T, H>,
1400    ) -> u64 {
1401        if !self.disable_block_gas_limit {
1402            if let Some(gas_limit) = self.gas_limit {
1403                return gas_limit;
1404            } else if block.header.gas_limit() > 0 {
1405                return block.header.gas_limit();
1406            }
1407        }
1408
1409        u64::MAX
1410    }
1411
1412    /// Returns the gas limit for a non forked anvil instance
1413    ///
1414    /// Checks the config for the `disable_block_gas_limit` flag
1415    pub(crate) fn gas_limit(&self) -> u64 {
1416        if self.disable_block_gas_limit {
1417            return u64::MAX;
1418        }
1419
1420        self.gas_limit.unwrap_or(DEFAULT_GAS_LIMIT)
1421    }
1422}

1424/// If the fork choice is a block number, simply return it with an empty list of transactions.
1425/// If the fork choice is a transaction hash, determine the block that the transaction was mined in,
1426/// and return the block number before the fork block along with all transactions in the fork block
1427/// that are before (and including) the fork transaction.
1428async fn derive_block_and_transactions(
1429    fork_choice: &ForkChoice,
1430    provider: &Arc<RetryProvider>,
1431) -> eyre::Result<(BlockNumber, Option<Vec<PoolTransaction>>)> {
1432    match fork_choice {
1433        ForkChoice::Block(block_number) => {
1434            let block_number = *block_number;
1435            if block_number >= 0 {
1436                return Ok((block_number as u64, None));
1437            }
1438            // subtract from latest block number
1439            let latest = provider.get_block_number().await?;
1440
1441            Ok((block_number.saturating_add(latest as i128) as u64, None))
1442        }
1443        ForkChoice::Transaction(transaction_hash) => {
1444            // Determine the block that this transaction was mined in
1445            let transaction = provider
1446                .get_transaction_by_hash(transaction_hash.0.into())
1447                .await?
1448                .ok_or_else(|| eyre::eyre!("failed to get fork transaction by hash"))?;
1449            let transaction_block_number = transaction.block_number.ok_or_else(|| {
1450                eyre::eyre!("fork transaction is not mined yet (no block number)")
1451            })?;
1452
1453            // Get the block pertaining to the fork transaction
1454            let transaction_block = provider
1455                .get_block_by_number(transaction_block_number.into())
1456                .full()
1457                .await?
1458                .ok_or_else(|| eyre::eyre!("failed to get fork block by number"))?;
1459
1460            // Filter out transactions that are after the fork transaction
1461            let filtered_transactions = transaction_block
1462                .transactions
1463                .as_transactions()
1464                .ok_or_else(|| eyre::eyre!("failed to get transactions from full fork block"))?
1465                .iter()
1466                .take_while_inclusive(|&transaction| transaction.tx_hash() != transaction_hash.0)
1467                .collect::<Vec<_>>();
1468
1469            // Convert the transactions to PoolTransactions
1470            let force_transactions = filtered_transactions
1471                .iter()
1472                .map(|&transaction| PoolTransaction::try_from(transaction.clone()))
1473                .collect::<Result<Vec<_>, _>>()
1474                .map_err(|e| eyre::eyre!("Err converting to pool transactions {e}"))?;
1475            Ok((transaction_block_number.saturating_sub(1), Some(force_transactions)))
1476        }
1477    }
1478}

1480/// Fork delimiter used to specify which block or transaction to fork from.
1481#[derive(Clone, Copy, Debug, PartialEq, Eq)]
1482pub enum ForkChoice {
1483    /// Block number to fork from.
1484    ///
1485    /// If negative, the given value is subtracted from the `latest` block number.
1486    Block(i128),
1487    /// Transaction hash to fork from.
1488    Transaction(TxHash),
1489}
1490
1491impl ForkChoice {
1492    /// Returns the block number to fork from
1493    pub fn block_number(&self) -> Option<i128> {
1494        match self {
1495            Self::Block(block_number) => Some(*block_number),
1496            Self::Transaction(_) => None,
1497        }
1498    }
1499
1500    /// Returns the transaction hash to fork from
1501    pub fn transaction_hash(&self) -> Option<TxHash> {
1502        match self {
1503            Self::Block(_) => None,
1504            Self::Transaction(transaction_hash) => Some(*transaction_hash),
1505        }
1506    }
1507}
1508
1509/// Convert a transaction hash into a ForkChoice
1510impl From<TxHash> for ForkChoice {
1511    fn from(tx_hash: TxHash) -> Self {
1512        Self::Transaction(tx_hash)
1513    }
1514}
1515
1516/// Convert a decimal block number into a ForkChoice
1517impl From<u64> for ForkChoice {
1518    fn from(block: u64) -> Self {
1519        Self::Block(block as i128)
1520    }
1521}

1623/// Finds the latest appropriate block to fork
1624///
1625/// This fetches the "latest" block and checks whether the `Block` is fully populated (`hash` field
1626/// is present). This prevents edge cases where anvil forks the "latest" block but `eth_getBlockByNumber` still returns a pending block, <https://github.com/foundry-rs/foundry/issues/2036>
1627async fn find_latest_fork_block<P: Provider<AnyNetwork>>(
1628    provider: P,
1629) -> Result<u64, TransportError> {
1630    let mut num = provider.get_block_number().await?;
1631
1632    // walk back from the head of the chain, but at most 2 blocks, which should be more than enough
1633    // leeway
1634    for _ in 0..2 {
1635        if let Some(block) = provider.get_block(num.into()).await?
1636            && !block.header.hash.is_zero()
1637        {
1638            break;
1639        }
1640        // block not actually finalized, so we try the block before
1641        num = num.saturating_sub(1)
1642    }
1643
1644    Ok(num)
1645}
