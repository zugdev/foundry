37/// Helper trait get access to the full state data of the database
38pub trait MaybeFullDatabase: DatabaseRef<Error = DatabaseError> + Debug {
39    /// Returns a reference to the database as a `dyn DatabaseRef`.
40    // TODO: Required until trait upcasting is stabilized: <https://github.com/rust-lang/rust/issues/65991>
41    fn as_dyn(&self) -> &dyn DatabaseRef<Error = DatabaseError>;
42
43    fn maybe_as_full_db(&self) -> Option<&HashMap<Address, DbAccount>> {
44        None
45    }
46
47    /// Clear the state and move it into a new `StateSnapshot`.
48    fn clear_into_state_snapshot(&mut self) -> StateSnapshot;
49
50    /// Read the state snapshot.
51    ///
52    /// This clones all the states and returns a new `StateSnapshot`.
53    fn read_as_state_snapshot(&self) -> StateSnapshot;
54
55    /// Clears the entire database
56    fn clear(&mut self);
57
58    /// Reverses `clear_into_snapshot` by initializing the db's state with the state snapshot.
59    fn init_from_state_snapshot(&mut self, state_snapshot: StateSnapshot);
60}

62impl<'a, T: 'a + MaybeFullDatabase + ?Sized> MaybeFullDatabase for &'a T
63where
64    &'a T: DatabaseRef<Error = DatabaseError>,
65{
66    fn as_dyn(&self) -> &dyn DatabaseRef<Error = DatabaseError> {
67        T::as_dyn(self)
68    }
69
70    fn maybe_as_full_db(&self) -> Option<&HashMap<Address, DbAccount>> {
71        T::maybe_as_full_db(self)
72    }
73
74    fn clear_into_state_snapshot(&mut self) -> StateSnapshot {
75        unreachable!("never called for DatabaseRef")
76    }
77
78    fn read_as_state_snapshot(&self) -> StateSnapshot {
79        unreachable!("never called for DatabaseRef")
80    }
81
82    fn clear(&mut self) {}
83
84    fn init_from_state_snapshot(&mut self, _state_snapshot: StateSnapshot) {}
85}

96/// This bundles all required revm traits
97pub trait Db:
98    DatabaseRef<Error = DatabaseError>
99    + Database<Error = DatabaseError>
100    + DatabaseCommit
101    + MaybeFullDatabase
102    + MaybeForkedDatabase
103    + fmt::Debug
104    + Send
105    + Sync
106{
107    /// Inserts an account
108    fn insert_account(&mut self, address: Address, account: AccountInfo);
109
110    /// Sets the nonce of the given address
111    fn set_nonce(&mut self, address: Address, nonce: u64) -> DatabaseResult<()> {
112        let mut info = self.basic(address)?.unwrap_or_default();
113        info.nonce = nonce;
114        self.insert_account(address, info);
115        Ok(())
116    }
117
118    /// Sets the balance of the given address
119    fn set_balance(&mut self, address: Address, balance: U256) -> DatabaseResult<()> {
120        let mut info = self.basic(address)?.unwrap_or_default();
121        info.balance = balance;
122        self.insert_account(address, info);
123        Ok(())
124    }
125
126    /// Sets the code of the given address
127    fn set_code(&mut self, address: Address, code: Bytes) -> DatabaseResult<()> {
128        let mut info = self.basic(address)?.unwrap_or_default();
129        let code_hash = if code.as_ref().is_empty() {
130            KECCAK_EMPTY
131        } else {
132            B256::from_slice(&keccak256(code.as_ref())[..])
133        };
134        info.code_hash = code_hash;
135        info.code = Some(Bytecode::new_raw(alloy_primitives::Bytes(code.0)));
136        self.insert_account(address, info);
137        Ok(())
138    }
139
140    /// Sets the storage value at the given slot for the address
141    fn set_storage_at(&mut self, address: Address, slot: B256, val: B256) -> DatabaseResult<()>;
142
143    /// inserts a blockhash for the given number
144    fn insert_block_hash(&mut self, number: U256, hash: B256);
145
146    /// Write all chain data to serialized bytes buffer
147    fn dump_state(
148        &self,
149        at: BlockEnv,
150        best_number: u64,
151        blocks: Vec<SerializableBlock>,
152        transactions: Vec<SerializableTransaction>,
153        historical_states: Option<SerializableHistoricalStates>,
154    ) -> DatabaseResult<Option<SerializableState>>;
155
156    /// Deserialize and add all chain data to the backend storage
157    fn load_state(&mut self, state: SerializableState) -> DatabaseResult<bool> {
158        for (addr, account) in state.accounts.into_iter() {
159            let old_account_nonce = DatabaseRef::basic_ref(self, addr)
160                .ok()
161                .and_then(|acc| acc.map(|acc| acc.nonce))
162                .unwrap_or_default();
163            // use max nonce in case account is imported multiple times with difference
164            // nonces to prevent collisions
165            let nonce = std::cmp::max(old_account_nonce, account.nonce);
166
167            self.insert_account(
168                addr,
169                AccountInfo {
170                    balance: account.balance,
171                    code_hash: KECCAK_EMPTY, // will be set automatically
172                    code: if account.code.0.is_empty() {
173                        None
174                    } else {
175                        Some(Bytecode::new_raw(alloy_primitives::Bytes(account.code.0)))
176                    },
177                    nonce,
178                },
179            );
180
181            for (k, v) in account.storage.into_iter() {
182                self.set_storage_at(addr, k, v)?;
183            }
184        }
185        Ok(true)
186    }
187
188    /// Creates a new state snapshot.
189    fn snapshot_state(&mut self) -> U256;
190
191    /// Reverts a state snapshot.
192    ///
193    /// Returns `true` if the state snapshot was reverted.
194    fn revert_state(&mut self, state_snapshot: U256, action: RevertStateSnapshotAction) -> bool;
195
196    /// Returns the state root if possible to compute
197    fn maybe_state_root(&self) -> Option<B256> {
198        None
199    }
200
201    /// Returns the current, standalone state of the Db
202    fn current_state(&self) -> StateDb;
203}

208        self.as_dyn()
209    }

253impl<T: DatabaseRef<Error = DatabaseError> + Debug> MaybeFullDatabase for CacheDB<T> {
254    fn as_dyn(&self) -> &dyn DatabaseRef<Error = DatabaseError> {
255        self
256    }
257
258    fn maybe_as_full_db(&self) -> Option<&HashMap<Address, DbAccount>> {
259        Some(&self.cache.accounts)
260    }
261
262    fn clear_into_state_snapshot(&mut self) -> StateSnapshot {
263        let db_accounts = std::mem::take(&mut self.cache.accounts);
264        let mut accounts = HashMap::default();
265        let mut account_storage = HashMap::default();
266
267        for (addr, mut acc) in db_accounts {
268            account_storage.insert(addr, std::mem::take(&mut acc.storage));
269            let mut info = acc.info;
270            info.code = self.cache.contracts.remove(&info.code_hash);
271            accounts.insert(addr, info);
272        }
273        let block_hashes = std::mem::take(&mut self.cache.block_hashes);
274        StateSnapshot { accounts, storage: account_storage, block_hashes }
275    }
276
277    fn read_as_state_snapshot(&self) -> StateSnapshot {
278        let db_accounts = self.cache.accounts.clone();
279        let mut accounts = HashMap::default();
280        let mut account_storage = HashMap::default();
281
282        for (addr, acc) in db_accounts {
283            account_storage.insert(addr, acc.storage.clone());
284            let mut info = acc.info;
285            info.code = self.cache.contracts.get(&info.code_hash).cloned();
286            accounts.insert(addr, info);
287        }
288
289        let block_hashes = self.cache.block_hashes.clone();
290        StateSnapshot { accounts, storage: account_storage, block_hashes }
291    }
292
293    fn clear(&mut self) {
294        self.clear_into_state_snapshot();
295    }
296
297    fn init_from_state_snapshot(&mut self, state_snapshot: StateSnapshot) {
298        let StateSnapshot { accounts, mut storage, block_hashes } = state_snapshot;
299
300        for (addr, mut acc) in accounts {
301            if let Some(code) = acc.code.take() {
302                self.cache.contracts.insert(acc.code_hash, code);
303            }
304            self.cache.accounts.insert(
305                addr,
306                DbAccount {
307                    info: acc,
308                    storage: storage.remove(&addr).unwrap_or_default(),
309                    ..Default::default()
310                },
311            );
312        }
313        self.cache.block_hashes = block_hashes;
314    }
315}

414/// Legacy string or u64 type from before v1.3.
415#[derive(Debug, Deserialize)]
416#[serde(untagged)]
417pub enum StringOrU64 {
418    Hex(String),
419    Dec(u64),
420}
421
422impl StringOrU64 {
423    pub fn to_u64(&self) -> Option<u64> {
424        match self {
425            Self::Dec(n) => Some(*n),
426            Self::Hex(s) => s.strip_prefix("0x").and_then(|s| u64::from_str_radix(s, 16).ok()),
427        }
428    }
429
430    pub fn to_u256(&self) -> Option<U256> {
431        match self {
432            Self::Dec(n) => Some(U256::from(*n)),
433            Self::Hex(s) => s.strip_prefix("0x").and_then(|s| U256::from_str_radix(s, 16).ok()),
434        }
435    }
436}

438/// Converts a `LegacyBlockEnv` to a `BlockEnv`, handling the conversion of legacy fields.
439impl TryFrom<LegacyBlockEnv> for BlockEnv {
440    type Error = &'static str;
441
442    fn try_from(legacy: LegacyBlockEnv) -> Result<Self, Self::Error> {
443        Ok(Self {
444            number: legacy.number.and_then(|v| v.to_u256()).unwrap_or(U256::ZERO),
445            beneficiary: legacy.beneficiary.unwrap_or(Address::ZERO),
446            timestamp: legacy.timestamp.and_then(|v| v.to_u256()).unwrap_or(U256::ONE),
447            gas_limit: legacy.gas_limit.and_then(|v| v.to_u64()).unwrap_or(u64::MAX),
448            basefee: legacy.basefee.and_then(|v| v.to_u64()).unwrap_or(0),
449            difficulty: legacy.difficulty.and_then(|v| v.to_u256()).unwrap_or(U256::ZERO),
450            prevrandao: legacy.prevrandao.or(Some(B256::ZERO)),
451            blob_excess_gas_and_price: legacy
452                .blob_excess_gas_and_price
453                .map(|v| BlobExcessGasAndPrice::new(v.excess_blob_gas, v.blob_gasprice))
454                .or_else(|| {
455                    Some(BlobExcessGasAndPrice::new(0, BLOB_BASE_FEE_UPDATE_FRACTION_PRAGUE))
456                }),
457        })
458    }
459}

461/// Custom deserializer for `BlockEnv` that handles both v1.2 and v1.3+ formats.
462fn deserialize_block_env_compat<'de, D>(deserializer: D) -> Result<Option<BlockEnv>, D::Error>
463where
464    D: Deserializer<'de>,
465{
466    let value: Option<Value> = Option::deserialize(deserializer)?;
467    let Some(value) = value else {
468        return Ok(None);
469    };
470
471    if let Ok(env) = BlockEnv::deserialize(&value) {
472        return Ok(Some(env));
473    }
474
475    let legacy: LegacyBlockEnv = serde_json::from_value(value).map_err(|e| {
476        D::Error::custom(format!("Legacy deserialization of `BlockEnv` failed: {e}"))
477    })?;
478
479    Ok(Some(BlockEnv::try_from(legacy).map_err(D::Error::custom)?))
480}

482/// Custom deserializer for `best_block_number` that handles both v1.2 and v1.3+ formats.
483fn deserialize_best_block_number_compat<'de, D>(deserializer: D) -> Result<Option<u64>, D::Error>
484where
485    D: Deserializer<'de>,
486{
487    let value: Option<Value> = Option::deserialize(deserializer)?;
488    let Some(value) = value else {
489        return Ok(None);
490    };
491
492    let number = match value {
493        Value::Number(n) => n.as_u64(),
494        Value::String(s) => {
495            if let Some(s) = s.strip_prefix("0x") {
496                u64::from_str_radix(s, 16).ok()
497            } else {
498                s.parse().ok()
499            }
500        }
501        _ => None,
502    };
503
504    Ok(number)
505}

529impl SerializableState {

531    pub fn load(path: impl AsRef<Path>) -> Result<Self, FsPathError> {
532        let path = path.as_ref();
533        if path.is_dir() {
534            foundry_common::fs::read_json_file(&path.join("state.json"))
535        } else {
536            foundry_common::fs::read_json_file(path)
537        }
538    }

541    #[allow(dead_code)]

543        Self::load(path).map_err(|err| err.to_string())
544    }

557fn deserialize_btree<'de, D>(deserializer: D) -> Result<BTreeMap<B256, B256>, D::Error>
558where
559    D: Deserializer<'de>,
560{
561    struct BTreeVisitor;
562
563    impl<'de> Visitor<'de> for BTreeVisitor {
564        type Value = BTreeMap<B256, B256>;
565
566        fn expecting(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
567            formatter.write_str("a mapping of hex encoded storage slots to hex encoded state data")
568        }
569
570        fn visit_map<M>(self, mut mapping: M) -> Result<BTreeMap<B256, B256>, M::Error>
571        where
572            M: MapAccess<'de>,
573        {
574            let mut btree = BTreeMap::new();
575            while let Some((key, value)) = mapping.next_entry::<U256, U256>()? {
576                btree.insert(B256::from(key), B256::from(value));
577            }
578
579            Ok(btree)
580        }
581    }
582
583    deserializer.deserialize_map(BTreeVisitor)
584}

586/// Defines a backwards-compatible enum for transactions.
587/// This is essential for maintaining compatibility with state dumps
588/// created before the changes introduced in PR #8411.
589///
590/// The enum can represent either a `TypedTransaction` or a `MaybeImpersonatedTransaction`,
591/// depending on the data being deserialized. This flexibility ensures that older state
592/// dumps can still be loaded correctly, even after the changes in #8411.
593#[derive(Clone, Debug, Serialize, Deserialize)]
594#[serde(untagged)]
595pub enum SerializableTransactionType {
596    TypedTransaction(TypedTransaction),
597    MaybeImpersonatedTransaction(MaybeImpersonatedTransaction),
598}
