16/// A coverage reporter.
17pub trait CoverageReporter {
18    /// Returns a debug string for the reporter.
19    fn name(&self) -> &'static str;
20
21    /// Returns `true` if the reporter needs source maps for the final report.
22    fn needs_source_maps(&self) -> bool {
23        false
24    }
25
26    /// Runs the reporter.
27    fn report(&mut self, report: &CoverageReport) -> eyre::Result<()>;
28}

30/// A simple summary reporter that prints the coverage results in a table.
31pub struct CoverageSummaryReporter {
32    /// The summary table.
33    table: Table,
34    /// The total coverage of the entire project.
35    total: CoverageSummary,
36}
37
38impl Default for CoverageSummaryReporter {
39    fn default() -> Self {
40        let mut table = Table::new();
41        table.apply_modifier(UTF8_ROUND_CORNERS);
42
43        table.set_header(vec![
44            Cell::new("File"),
45            Cell::new("% Lines"),
46            Cell::new("% Statements"),
47            Cell::new("% Branches"),
48            Cell::new("% Funcs"),
49        ]);
50
51        Self { table, total: CoverageSummary::default() }
52    }
53}
54
55impl CoverageSummaryReporter {
56    fn add_row(&mut self, name: impl Into<Cell>, summary: CoverageSummary) {
57        let mut row = Row::new();
58        row.add_cell(name.into())
59            .add_cell(format_cell(summary.line_hits, summary.line_count))
60            .add_cell(format_cell(summary.statement_hits, summary.statement_count))
61            .add_cell(format_cell(summary.branch_hits, summary.branch_count))
62            .add_cell(format_cell(summary.function_hits, summary.function_count));
63        self.table.add_row(row);
64    }
65}
66
67impl CoverageReporter for CoverageSummaryReporter {
68    fn name(&self) -> &'static str {
69        "summary"
70    }
71
72    fn report(&mut self, report: &CoverageReport) -> eyre::Result<()> {
73        for (path, summary) in report.summary_by_file() {
74            self.total.merge(&summary);
75            self.add_row(path.display(), summary);
76        }
77
78        self.add_row("Total", self.total.clone());
79        sh_println!("\n{}", self.table)?;
80        Ok(())
81    }
82}

84fn format_cell(hits: usize, total: usize) -> Cell {
85    let percentage = if total == 0 { 1. } else { hits as f64 / total as f64 };
86
87    let mut cell =
88        Cell::new(format!("{:.2}% ({hits}/{total})", percentage * 100.)).fg(match percentage {
89            _ if total == 0 => Color::Grey,
90            _ if percentage < 0.5 => Color::Red,
91            _ if percentage < 0.75 => Color::Yellow,
92            _ => Color::Green,
93        });
94
95    if total == 0 {
96        cell = cell.add_attribute(Attribute::Dim);
97    }
98    cell
99}

101/// Writes the coverage report in [LCOV]'s [tracefile format].
102///
103/// [LCOV]: https://github.com/linux-test-project/lcov
104/// [tracefile format]: https://man.archlinux.org/man/geninfo.1.en#TRACEFILE_FORMAT
105pub struct LcovReporter {
106    path: PathBuf,
107    version: Version,
108}
109
110impl LcovReporter {
111    /// Create a new LCOV reporter.
112    pub fn new(path: PathBuf, version: Version) -> Self {
113        Self { path, version }
114    }
115}
116
117impl CoverageReporter for LcovReporter {
118    fn name(&self) -> &'static str {
119        "lcov"
120    }
121
122    fn report(&mut self, report: &CoverageReport) -> eyre::Result<()> {
123        let mut out = std::io::BufWriter::new(fs::create_file(&self.path)?);
124
125        let mut fn_index = 0usize;
126        for (path, items) in report.items_by_file() {
127            let summary = CoverageSummary::from_items(items.iter().copied());
128
129            writeln!(out, "TN:")?;
130            writeln!(out, "SF:{}", path.display())?;
131
132            let mut recorded_lines = HashSet::new();
133
134            for item in items {
135                let line = item.loc.lines.start;
136                // `lines` is half-open, so we need to subtract 1 to get the last included line.
137                let end_line = item.loc.lines.end - 1;
138                let hits = item.hits;
139                match item.kind {
140                    CoverageItemKind::Function { ref name } => {
141                        let name = format!("{}.{name}", item.loc.contract_name);
142                        if self.version >= Version::new(2, 2, 0) {
143                            // v2.2 changed the FN format.
144                            writeln!(out, "FNL:{fn_index},{line},{end_line}")?;
145                            writeln!(out, "FNA:{fn_index},{hits},{name}")?;
146                            fn_index += 1;
147                        } else if self.version >= Version::new(2, 0, 0) {
148                            // v2.0 added end_line to FN.
149                            writeln!(out, "FN:{line},{end_line},{name}")?;
150                            writeln!(out, "FNDA:{hits},{name}")?;
151                        } else {
152                            writeln!(out, "FN:{line},{name}")?;
153                            writeln!(out, "FNDA:{hits},{name}")?;
154                        }
155                    }
156                    // Add lines / statement hits only once.
157                    CoverageItemKind::Line | CoverageItemKind::Statement => {
158                        if recorded_lines.insert(line) {
159                            writeln!(out, "DA:{line},{hits}")?;
160                        }
161                    }
162                    CoverageItemKind::Branch { branch_id, path_id, .. } => {
163                        writeln!(
164                            out,
165                            "BRDA:{line},{branch_id},{path_id},{}",
166                            if hits == 0 { "-".to_string() } else { hits.to_string() }
167                        )?;
168                    }
169                }
170            }
171
172            // Function summary
173            writeln!(out, "FNF:{}", summary.function_count)?;
174            writeln!(out, "FNH:{}", summary.function_hits)?;
175
176            // Line summary
177            writeln!(out, "LF:{}", summary.line_count)?;
178            writeln!(out, "LH:{}", summary.line_hits)?;
179
180            // Branch summary
181            writeln!(out, "BRF:{}", summary.branch_count)?;
182            writeln!(out, "BRH:{}", summary.branch_hits)?;
183
184            writeln!(out, "end_of_record")?;
185        }
186
187        out.flush()?;
188        sh_println!("Wrote LCOV report.")?;
189
190        Ok(())
191    }
192}

194/// A super verbose reporter for debugging coverage while it is still unstable.
195pub struct DebugReporter;
196
197impl CoverageReporter for DebugReporter {
198    fn name(&self) -> &'static str {
199        "debug"
200    }
201
202    fn report(&mut self, report: &CoverageReport) -> eyre::Result<()> {
203        for (path, items) in report.items_by_file() {
204            let uncovered = items.iter().copied().filter(|item| item.hits == 0);
205            if uncovered.clone().count() == 0 {
206                continue;
207            }
208
209            sh_println!("Uncovered for {}:", path.display())?;
210            for item in uncovered {
211                sh_println!("- {item}")?;
212            }
213            sh_println!()?;
214        }
215
216        for (contract_id, (cta, rta)) in &report.anchors {
217            if cta.is_empty() && rta.is_empty() {
218                continue;
219            }
220
221            sh_println!("Anchors for {contract_id}:")?;
222            let anchors = cta
223                .iter()
224                .map(|anchor| (false, anchor))
225                .chain(rta.iter().map(|anchor| (true, anchor)));
226            for (is_runtime, anchor) in anchors {
227                let kind = if is_runtime { " runtime" } else { "creation" };
228                sh_println!(
229                    "- {kind} {anchor}: {}",
230                    report
231                        .analyses
232                        .get(&contract_id.version)
233                        .and_then(|items| items.get(anchor.item_id))
234                        .map_or_else(|| "None".to_owned(), |item| item.to_string())
235                )?;
236            }
237            sh_println!()?;
238        }
239
240        Ok(())
241    }
242}

244pub struct BytecodeReporter {
245    root: PathBuf,
246    destdir: PathBuf,
247}
248
249impl BytecodeReporter {
250    pub fn new(root: PathBuf, destdir: PathBuf) -> Self {
251        Self { root, destdir }
252    }
253}
254
255impl CoverageReporter for BytecodeReporter {
256    fn name(&self) -> &'static str {
257        "bytecode"
258    }
259
260    fn needs_source_maps(&self) -> bool {
261        true
262    }
263
264    fn report(&mut self, report: &CoverageReport) -> eyre::Result<()> {
265        use std::fmt::Write;
266
267        fs::create_dir_all(&self.destdir)?;
268
269        let no_source_elements = Vec::new();
270        let mut line_number_cache = LineNumberCache::new(self.root.clone());
271
272        for (contract_id, hits) in &report.bytecode_hits {
273            let ops = disassemble_bytes(hits.bytecode().to_vec())?;
274            let mut formatted = String::new();
275
276            let source_elements =
277                report.source_maps.get(contract_id).map(|sm| &sm.1).unwrap_or(&no_source_elements);
278
279            for (code, source_element) in std::iter::zip(ops.iter(), source_elements) {
280                let hits = hits
281                    .get(code.offset)
282                    .map(|h| format!("[{h:03}]"))
283                    .unwrap_or("     ".to_owned());
284                let source_id = source_element.index();
285                let source_path = source_id.and_then(|i| {
286                    report.source_paths.get(&(contract_id.version.clone(), i as usize))
287                });
288
289                let code = format!("{code:?}");
290                let start = source_element.offset() as usize;
291                let end = (source_element.offset() + source_element.length()) as usize;
292
293                if let Some(source_path) = source_path {
294                    let (sline, spos) = line_number_cache.get_position(source_path, start)?;
295                    let (eline, epos) = line_number_cache.get_position(source_path, end)?;
296                    writeln!(
297                        formatted,
298                        "{} {:40} // {}: {}:{}-{}:{} ({}-{})",
299                        hits,
300                        code,
301                        source_path.display(),
302                        sline,
303                        spos,
304                        eline,
305                        epos,
306                        start,
307                        end
308                    )?;
309                } else if let Some(source_id) = source_id {
310                    writeln!(formatted, "{hits} {code:40} // SRCID{source_id}: ({start}-{end})")?;
311                } else {
312                    writeln!(formatted, "{hits} {code:40}")?;
313                }
314            }
315            fs::write(
316                self.destdir.join(&*contract_id.contract_name).with_extension("asm"),
317                formatted,
318            )?;
319        }
320
321        Ok(())
322    }
323}

325/// Cache line number offsets for source files
326struct LineNumberCache {
327    root: PathBuf,
328    line_offsets: HashMap<PathBuf, Vec<usize>>,
329}
330
331impl LineNumberCache {
332    pub fn new(root: PathBuf) -> Self {
333        Self { root, line_offsets: HashMap::default() }
334    }
335
336    pub fn get_position(&mut self, path: &Path, offset: usize) -> eyre::Result<(usize, usize)> {
337        let line_offsets = match self.line_offsets.entry(path.to_path_buf()) {
338            hash_map::Entry::Occupied(o) => o.into_mut(),
339            hash_map::Entry::Vacant(v) => {
340                let text = fs::read_to_string(self.root.join(path))?;
341                let mut line_offsets = vec![0];
342                for line in text.lines() {
343                    let line_offset = line.as_ptr() as usize - text.as_ptr() as usize;
344                    line_offsets.push(line_offset);
345                }
346                v.insert(line_offsets)
347            }
348        };
349        let lo = match line_offsets.binary_search(&offset) {
350            Ok(lo) => lo,
351            Err(lo) => lo - 1,
352        };
353        let pos = offset - line_offsets.get(lo).unwrap() + 1;
354        Ok((lo, pos))
355    }
