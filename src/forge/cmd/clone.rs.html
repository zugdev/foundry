55/// CLI arguments for `forge clone`.
56///
57/// `forge clone` clones an on-chain contract from block explorers (e.g., Etherscan) in the
58/// following steps:
59/// 1. Fetch the contract source code from the block explorer.
60/// 2. Initialize a empty foundry project at the `root` directory specified in `CloneArgs`.
61/// 3. Dump the contract sources to the source directory.
62/// 4. Update the `foundry.toml` configuration file with the compiler settings from Etherscan.
63/// 5. Try compile the cloned contract, so that we can get the original storage layout. This
64///    original storage layout is preserved in the `CloneMetadata` so that if the user later
65///    modifies the contract, it is possible to quickly check the storage layout compatibility with
66///    the original on-chain contract.
67/// 6. Dump the `CloneMetadata` to the root directory of the cloned project as `.clone.meta` file.
68#[derive(Clone, Debug, Parser)]
69pub struct CloneArgs {
70    /// The contract address to clone.
71    pub address: Address,
72
73    /// The root directory of the cloned project.
74    #[arg(value_hint = ValueHint::DirPath, default_value = ".", value_name = "PATH")]
75    pub root: PathBuf,
76
77    /// Do not generate the remappings.txt file. Instead, keep the remappings in the configuration.
78    #[arg(long)]
79    pub no_remappings_txt: bool,
80
81    /// Keep the original directory structure collected from Etherscan.
82    ///
83    /// If this flag is set, the directory structure of the cloned project will be kept as is.
84    /// By default, the directory structure is re-orgnized to increase the readability, but may
85    /// risk some compilation failures.
86    #[arg(long)]
87    pub keep_directory_structure: bool,
88
89    #[command(flatten)]
90    pub etherscan: EtherscanOpts,
91
92    #[command(flatten)]
93    pub install: DependencyInstallOpts,
94}

97    pub async fn run(self) -> Result<()> {
98        let Self { address, root, install, etherscan, no_remappings_txt, keep_directory_structure } =
99            self;
100
101        // step 0. get the chain and api key from the config
102        let config = etherscan.load_config()?;
103        let chain = config.chain.unwrap_or_default();
104        let etherscan_api_version = config.get_etherscan_api_version(Some(chain));
105        let etherscan_api_key = config.get_etherscan_api_key(Some(chain)).unwrap_or_default();
106        let client =
107            Client::new_with_api_version(chain, etherscan_api_key.clone(), etherscan_api_version)?;
108
109        // step 1. get the metadata from client
110        sh_println!("Downloading the source code of {address} from Etherscan...")?;
111
112        let meta = Self::collect_metadata_from_client(address, &client).await?;
113
114        // step 2. initialize an empty project
115        Self::init_an_empty_project(&root, install)?;
116        // canonicalize the root path
117        // note that at this point, the root directory must have been created
118        let root = dunce::canonicalize(&root)?;
119
120        // step 3. parse the metadata
121        Self::parse_metadata(&meta, chain, &root, no_remappings_txt, keep_directory_structure)
122            .await?;
123
124        // step 4. collect the compilation metadata
125        // if the etherscan api key is not set, we need to wait for 3 seconds between calls
126        sh_println!("Collecting the creation information of {address} from Etherscan...")?;
127
128        if etherscan_api_key.is_empty() {
129            sh_warn!("Waiting for 5 seconds to avoid rate limit...")?;
130            tokio::time::sleep(Duration::from_secs(5)).await;
131        }
132        Self::collect_compilation_metadata(&meta, chain, address, &root, &client).await?;
133
134        // step 5. git add and commit the changes if needed
135        if install.commit {
136            let git = Git::new(&root);
137            git.add(Some("--all"))?;
138            let msg = format!("chore: forge clone {address}");
139            git.commit(&msg)?;
140        }
141
142        Ok(())
143    }

180    /// Collect the compilation metadata of the cloned contract.
181    /// This function compiles the cloned contract and collects the compilation metadata.
182    ///
183    /// * `meta` - the metadata of the contract (from Etherscan).
184    /// * `chain` - the chain where the contract to be cloned locates.
185    /// * `address` - the address of the contract to be cloned.
186    /// * `root` - the root directory of the cloned project.
187    /// * `client` - the client of the block explorer.
188    pub(crate) async fn collect_compilation_metadata<C: EtherscanClient>(
189        meta: &Metadata,
190        chain: Chain,
191        address: Address,
192        root: &PathBuf,
193        client: &C,
194    ) -> Result<()> {
195        // compile the cloned contract
196        let compile_output = compile_project(root)?;
197        let (main_file, main_artifact) = find_main_contract(&compile_output, &meta.contract_name)?;
198        let main_file = main_file.strip_prefix(root)?.to_path_buf();
199        let storage_layout =
200            main_artifact.storage_layout.to_owned().expect("storage layout not found");
201
202        // dump the metadata to the root directory
203        let creation_tx = client.contract_creation_data(address).await?;
204        let clone_meta = CloneMetadata {
205            path: main_file,
206            target_contract: meta.contract_name.clone(),
207            address,
208            chain_id: chain.id(),
209            creation_transaction: creation_tx.transaction_hash,
210            deployer: creation_tx.contract_creator,
211            constructor_arguments: meta.constructor_arguments.clone(),
212            storage_layout,
213        };
214        let metadata_content = serde_json::to_string(&clone_meta)?;
215        let metadata_file = root.join(".clone.meta");
216        fs::write(&metadata_file, metadata_content)?;
217        let mut perms = std::fs::metadata(&metadata_file)?.permissions();
218        perms.set_readonly(true);
219        std::fs::set_permissions(&metadata_file, perms)?;
220
221        Ok(())
222    }

224    /// Download and parse the source code from Etherscan.
225    ///
226    /// * `chain` - the chain where the contract to be cloned locates.
227    /// * `address` - the address of the contract to be cloned.
228    /// * `root` - the root directory to clone the contract into as a foundry project.
229    /// * `client` - the client of the block explorer.
230    /// * `no_remappings_txt` - whether to generate the remappings.txt file.
231    pub(crate) async fn parse_metadata(
232        meta: &Metadata,
233        chain: Chain,
234        root: &PathBuf,
235        no_remappings_txt: bool,
236        keep_directory_structure: bool,
237    ) -> Result<()> {
238        // dump sources and update the remapping in configuration
239        let remappings = dump_sources(meta, root, keep_directory_structure)?;
240        Config::update_at(root, |config, doc| {
241            let profile = config.profile.as_str().as_str();
242
243            // update the remappings in the configuration
244            let mut remapping_array = toml_edit::Array::new();
245            for r in remappings {
246                remapping_array.push(r.to_string());
247            }
248            doc[Config::PROFILE_SECTION][profile]["remappings"] = toml_edit::value(remapping_array);
249
250            // make sure auto_detect_remappings is false (it is very important because cloned
251            // project may not follow the common remappings)
252            doc[Config::PROFILE_SECTION][profile]["auto_detect_remappings"] =
253                toml_edit::value(false);
254            true
255        })?;
256
257        // update configuration
258        Config::update_at(root, |config, doc| {
259            update_config_by_metadata(config, doc, meta, chain).is_ok()
260        })?;
261
262        // write remappings to remappings.txt if necessary
263        if !no_remappings_txt {
264            let remappings_txt = root.join("remappings.txt");
265            eyre::ensure!(
266                !remappings_txt.exists(),
267                "remappings.txt already exists, please remove it first"
268            );
269
270            Config::update_at(root, |config, doc| {
271                let remappings_txt_content =
272                    config.remappings.iter().map(|r| r.to_string()).collect::<Vec<_>>().join("\n");
273                if fs::write(&remappings_txt, remappings_txt_content).is_err() {
274                    return false;
275                }
276
277                let profile = config.profile.as_str().as_str();
278                if let Some(elem) = doc[Config::PROFILE_SECTION][profile].as_table_mut() {
279                    elem.remove_entry("remappings");
280                    true
281                } else {
282                    false
283                }
284            })?;
285        }
286
287        Ok(())
288    }

291/// Update the configuration file with the metadata.
292/// This function will update the configuration file with the metadata from the contract.
293/// It will update the following fields:
294/// - `auto_detect_solc` to `false`
295/// - `solc_version` to the value from the metadata
296/// - `evm_version` to the value from the metadata, if the metadata's evm_version is "Default", then
297///   this is derived from the solc version this contract was compiled with.
298/// - `via_ir` to the value from the metadata
299/// - `libraries` to the value from the metadata
300/// - `metadata` to the value from the metadata
301///     - `cbor_metadata`, `use_literal_content`, and `bytecode_hash`
302/// - `optimizer` to the value from the metadata
303/// - `optimizer_runs` to the value from the metadata
304/// - `optimizer_details` to the value from the metadata
305///     - `yul_details`, `yul`, etc.
306///     - `simpleCounterForLoopUncheckedIncrement` is ignored for now
307/// - `remappings` and `stop_after` are pre-validated to be empty and None, respectively
308/// - `model_checker`, `debug`, and `output_selection` are ignored for now
309///
310/// Detailed information can be found from the following link:
311/// - <https://github.com/foundry-rs/foundry/blob/master/crates/config/README.md#all-options>
312/// - <https://docs.soliditylang.org/en/latest/using-the-compiler.html#compiler-input-and-output-json-description>
313fn update_config_by_metadata(
314    config: &Config,
315    doc: &mut toml_edit::DocumentMut,
316    meta: &Metadata,
317    chain: Chain,
318) -> Result<()> {
319    let profile = config.profile.as_str().as_str();
320
321    // macro to update the config if the value exists
322    macro_rules! update_if_needed {
323        ([$($key:expr),+], $value:expr) => {
324            {
325                if let Some(value) = $value {
326                    let mut current = &mut doc[Config::PROFILE_SECTION][profile];
327                    $(
328                        if let Some(nested_doc) = current.get_mut(&$key) {
329                            current = nested_doc;
330                        } else {
331                            return Err(eyre::eyre!("cannot find the key: {}", $key));
332                        }
333                    )+
334                    *current = toml_edit::value(value);
335                }
336            }
337        };
338    }
339
340    // update the chain id
341    doc[Config::PROFILE_SECTION][profile]["chain_id"] = toml_edit::value(chain.id() as i64);
342
343    // disable auto detect solc and set the solc version
344    doc[Config::PROFILE_SECTION][profile]["auto_detect_solc"] = toml_edit::value(false);
345    let version = meta.compiler_version()?;
346    doc[Config::PROFILE_SECTION][profile]["solc_version"] =
347        toml_edit::value(format!("{}.{}.{}", version.major, version.minor, version.patch));
348
349    // get optimizer settings
350    // we ignore `model_checker`, `debug`, and `output_selection` for now,
351    // it seems they do not have impacts on the actual compilation
352    let Settings { optimizer, libraries, evm_version, via_ir, stop_after, metadata, .. } =
353        meta.settings()?;
354    eyre::ensure!(stop_after.is_none(), "stop_after should be None");
355
356    update_if_needed!(["evm_version"], evm_version.map(|v| v.to_string()));
357    update_if_needed!(["via_ir"], via_ir);
358
359    // update metadata if needed
360    if let Some(metadata) = metadata {
361        update_if_needed!(["cbor_metadata"], metadata.cbor_metadata);
362        update_if_needed!(["use_literal_content"], metadata.use_literal_content);
363        update_if_needed!(["bytecode_hash"], metadata.bytecode_hash.map(|v| v.to_string()));
364    }
365
366    // update optimizer settings if needed
367    update_if_needed!(["optimizer"], optimizer.enabled);
368    update_if_needed!(["optimizer_runs"], optimizer.runs.map(|v| v as i64));
369    // update optimizer details if needed
370    if let Some(detail) = optimizer.details {
371        doc[Config::PROFILE_SECTION][profile]["optimizer_details"] = toml_edit::table();
372
373        update_if_needed!(["optimizer_details", "peephole"], detail.peephole);
374        update_if_needed!(["optimizer_details", "inliner"], detail.inliner);
375        update_if_needed!(["optimizer_details", "jumpdestRemover"], detail.jumpdest_remover);
376        update_if_needed!(["optimizer_details", "orderLiterals"], detail.order_literals);
377        update_if_needed!(["optimizer_details", "deduplicate"], detail.deduplicate);
378        update_if_needed!(["optimizer_details", "cse"], detail.cse);
379        update_if_needed!(["optimizer_details", "constantOptimizer"], detail.constant_optimizer);
380        update_if_needed!(
381            ["optimizer_details", "simpleCounterForLoopUncheckedIncrement"],
382            detail.simple_counter_for_loop_unchecked_increment
383        );
384        update_if_needed!(["optimizer_details", "yul"], detail.yul);
385
386        if let Some(yul_detail) = detail.yul_details {
387            doc[Config::PROFILE_SECTION][profile]["optimizer_details"]["yulDetails"] =
388                toml_edit::table();
389            update_if_needed!(
390                ["optimizer_details", "yulDetails", "stackAllocation"],
391                yul_detail.stack_allocation
392            );
393            update_if_needed!(
394                ["optimizer_details", "yulDetails", "optimizerSteps"],
395                yul_detail.optimizer_steps
396            );
397        }
398    }
399
400    // apply remapping on libraries
401    let path_config: ProjectPathsConfig = config.project_paths();
402    let libraries = libraries
403        .apply(|libs| path_config.apply_lib_remappings(libs))
404        .with_stripped_file_prefixes(&path_config.root);
405
406    // update libraries
407    let mut lib_array = toml_edit::Array::new();
408    for (path_to_lib, info) in libraries.libs {
409        for (lib_name, address) in info {
410            lib_array.push(format!("{}:{}:{}", path_to_lib.to_str().unwrap(), lib_name, address));
411        }
412    }
413    doc[Config::PROFILE_SECTION][profile]["libraries"] = toml_edit::value(lib_array);
414
415    Ok(())
416}

418/// Dump the contract sources to the root directory.
419/// The sources are dumped to the `src` directory.
420/// IO errors may be returned.
421/// A list of remappings is returned
422fn dump_sources(meta: &Metadata, root: &PathBuf, no_reorg: bool) -> Result<Vec<RelativeRemapping>> {
423    // get config
424    let path_config = ProjectPathsConfig::builder().build_with_root::<Solc>(root);
425    // we will canonicalize the sources directory later
426    let src_dir = &path_config.sources;
427    let lib_dir = &path_config.libraries[0];
428    // Optional dir, if found in src
429    let node_modules_dir = &root.join("node_modules");
430    let contract_name = &meta.contract_name;
431    let source_tree = meta.source_tree();
432
433    // then we move the sources to the correct directories
434    // we will first load existing remappings if necessary
435    //  make sure this happens before dumping sources
436    let mut remappings: Vec<Remapping> = Remapping::find_many(root);
437
438    // first we dump the sources to a temporary directory
439    let tmp_dump_dir = root.join("raw_sources");
440    source_tree
441        .write_to(&tmp_dump_dir)
442        .map_err(|e| eyre::eyre!("failed to dump sources: {}", e))?;
443
444    // check whether we need to re-organize directories in the original sources, since we do not
445    // want to put all the sources in the `src` directory if the original directory structure is
446    // well organized, e.g., a standard foundry project containing `src` and `lib`
447    //
448    // * if the user wants to keep the original directory structure, we should not re-organize.
449    // * if there is any other directory other than `src`, `contracts`, `lib`, `hardhat`,
450    //   `forge-std`,
451    // or not started with `@`, we should not re-organize.
452    let to_reorg = !no_reorg
453        && std::fs::read_dir(tmp_dump_dir.join(contract_name))?.all(|e| {
454            let Ok(e) = e else { return false };
455            let folder_name = e.file_name();
456            folder_name == "src"
457                || folder_name == "lib"
458                || folder_name == "node_modules"
459                || folder_name == "contracts"
460                || folder_name == "hardhat"
461                || folder_name == "forge-std"
462                || folder_name.to_string_lossy().starts_with('@')
463        });
464
465    // ensure `src` and `lib` directories exist
466    eyre::ensure!(Path::exists(&root.join(src_dir)), "`src` directory must exists");
467    eyre::ensure!(Path::exists(&root.join(lib_dir)), "`lib` directory must exists");
468
469    // move source files
470    for entry in std::fs::read_dir(tmp_dump_dir.join(contract_name))? {
471        let entry = entry?;
472        let folder_name = entry.file_name();
473        // special handling when we need to re-organize the directories: we flatten them.
474        if to_reorg {
475            if folder_name == "contracts"
476                || folder_name == "src"
477                || folder_name == "lib"
478                || folder_name == "node_modules"
479            {
480                // move all sub folders in contracts to src or lib
481                let new_dir = if folder_name == "lib" {
482                    lib_dir
483                } else if folder_name == "node_modules" {
484                    // Create node_modules dir if it exists in raw sources.
485                    std::fs::create_dir(node_modules_dir)?;
486                    node_modules_dir
487                } else {
488                    src_dir
489                };
490                for e in read_dir(entry.path())? {
491                    let e = e?;
492                    let dest = new_dir.join(e.file_name());
493                    eyre::ensure!(!Path::exists(&dest), "destination already exists: {:?}", dest);
494                    std::fs::rename(e.path(), &dest)?;
495                    remappings.push(Remapping {
496                        context: None,
497                        name: format!(
498                            "{}/{}",
499                            folder_name.to_string_lossy(),
500                            e.file_name().to_string_lossy()
501                        ),
502                        path: dest.to_string_lossy().to_string(),
503                    });
504                }
505            } else {
506                assert!(
507                    folder_name == "hardhat"
508                        || folder_name == "forge-std"
509                        || folder_name.to_string_lossy().starts_with('@')
510                );
511                // move these other folders to lib
512                let dest = lib_dir.join(&folder_name);
513                if folder_name == "forge-std" {
514                    // let's use the provided forge-std directory
515                    std::fs::remove_dir_all(&dest)?;
516                }
517                eyre::ensure!(!Path::exists(&dest), "destination already exists: {:?}", dest);
518                std::fs::rename(entry.path(), &dest)?;
519                remappings.push(Remapping {
520                    context: None,
521                    name: folder_name.to_string_lossy().to_string(),
522                    path: dest.to_string_lossy().to_string(),
523                });
524            }
525        } else {
526            // directly move the all folders into src
527            let dest = src_dir.join(&folder_name);
528            eyre::ensure!(!Path::exists(&dest), "destination already exists: {:?}", dest);
529            std::fs::rename(entry.path(), &dest)?;
530            if folder_name != "src" {
531                remappings.push(Remapping {
532                    context: None,
533                    name: folder_name.to_string_lossy().to_string(),
534                    path: dest.to_string_lossy().to_string(),
535                });
536            }
537        }
538    }
539
540    // remove the temporary directory
541    std::fs::remove_dir_all(tmp_dump_dir)?;
542
543    // add remappings in the metadata
544    for mut r in meta.settings()?.remappings {
545        if to_reorg {
546            // we should update its remapped path in the same way as we dump sources
547            // i.e., remove prefix `contracts` (if any) and add prefix `src`
548            let new_path = if r.path.starts_with("contracts") {
549                PathBuf::from("src").join(PathBuf::from(&r.path).strip_prefix("contracts")?)
550            } else if r.path.starts_with('@')
551                || r.path.starts_with("hardhat/")
552                || r.path.starts_with("forge-std/")
553            {
554                PathBuf::from("lib").join(PathBuf::from(&r.path))
555            } else {
556                PathBuf::from(&r.path)
557            };
558            r.path = new_path.to_string_lossy().to_string();
559        }
560        remappings.push(r);
561    }
562
563    Ok(remappings.into_iter().map(|r| r.into_relative(root)).collect())
564}

575/// Find the artifact of the contract with the specified name.
576/// This function returns the path to the source file and the artifact.
577pub fn find_main_contract<'a>(
578    compile_output: &'a ProjectCompileOutput,
579    contract: &str,
580) -> Result<(PathBuf, &'a ConfigurableContractArtifact)> {
581    let mut rv = None;
582    for (f, c, a) in compile_output.artifacts_with_files() {
583        if contract == c {
584            // it is possible that we have multiple contracts with the same name
585            // in different files
586            // instead of throwing an error, we should handle this case in the future
587            if rv.is_some() {
588                return Err(eyre::eyre!("multiple contracts with the same name found"));
589            }
590            rv = Some((PathBuf::from(f), a));
591        }
592    }
593    rv.ok_or_else(|| eyre::eyre!("contract not found"))
594}
