30/// CLI arguments for `forge build`.
31///
32/// CLI arguments take the highest precedence in the Config/Figment hierarchy.
33/// In order to override them in the foundry `Config` they need to be merged into an existing
34/// `figment::Provider`, like `foundry_config::Config` is.
35///
36/// `BuildArgs` implements `figment::Provider` in which all config related fields are serialized and
37/// then merged into an existing `Config`, effectively overwriting them.
38///
39/// Some arguments are marked as `#[serde(skip)]` and require manual processing in
40/// `figment::Provider` implementation
41#[derive(Clone, Debug, Default, Serialize, Parser)]
42#[command(next_help_heading = "Build options", about = None, long_about = None)] // override doc
43pub struct BuildArgs {
44    /// Build source files from specified paths.
45    #[serde(skip)]
46    pub paths: Option<Vec<PathBuf>>,
47
48    /// Print compiled contract names.
49    #[arg(long)]
50    #[serde(skip)]
51    pub names: bool,
52
53    /// Print compiled contract sizes.
54    /// Constructor argument length is not included in the calculation of initcode size.
55    #[arg(long)]
56    #[serde(skip)]
57    pub sizes: bool,
58
59    /// Ignore initcode contract bytecode size limit introduced by EIP-3860.
60    #[arg(long, alias = "ignore-initcode-size")]
61    #[serde(skip)]
62    pub ignore_eip_3860: bool,
63
64    #[command(flatten)]
65    #[serde(flatten)]
66    pub build: BuildOpts,
67
68    #[command(flatten)]
69    #[serde(skip)]
70    pub watch: WatchArgs,
71}

73impl BuildArgs {
74    pub fn run(self) -> Result<ProjectCompileOutput> {
75        let mut config = self.load_config()?;
76
77        if install::install_missing_dependencies(&mut config) && config.auto_detect_remappings {
78            // need to re-configure here to also catch additional remappings
79            config = self.load_config()?;
80        }
81
82        let project = config.project()?;
83
84        // Collect sources to compile if build subdirectories specified.
85        let mut files = vec![];
86        if let Some(paths) = &self.paths {
87            for path in paths {
88                let joined = project.root().join(path);
89                let path = if joined.exists() { &joined } else { path };
90                files.extend(source_files_iter(path, MultiCompilerLanguage::FILE_EXTENSIONS));
91            }
92            if files.is_empty() {
93                eyre::bail!("No source files found in specified build paths.")
94            }
95        }
96
97        let format_json = shell::is_json();
98        let compiler = ProjectCompiler::new()
99            .files(files)
100            .dynamic_test_linking(config.dynamic_test_linking)
101            .print_names(self.names)
102            .print_sizes(self.sizes)
103            .ignore_eip_3860(self.ignore_eip_3860)
104            .bail(!format_json);
105
106        let output = compiler.compile(&project)?;
107
108        // Cache project selectors.
109        cache_local_signatures(&output)?;
110
111        if format_json && !self.names && !self.sizes {
112            sh_println!("{}", serde_json::to_string_pretty(&output.output())?)?;
113        }
114
115        if !config.lint.lint_on_build {
116            return Ok(output);
117        }
118
119        // Only run the `SolidityLinter` if there are no compilation errors
120        if !output.output().errors.iter().any(|e| e.is_error()) {
121            self.lint(&project, &config).map_err(|err| eyre!("Lint failed: {err}"))?;
122        }
123
124        Ok(output)
125    }
126
127    fn lint(&self, project: &Project, config: &Config) -> Result<()> {
128        let format_json = shell::is_json();
129        if project.compiler.solc.is_some() && !shell::is_quiet() {
130            let linter = SolidityLinter::new(config.project_paths())
131                .with_json_emitter(format_json)
132                .with_description(!format_json)
133                .with_severity(if config.lint.severity.is_empty() {
134                    None
135                } else {
136                    Some(config.lint.severity.clone())
137                })
138                .without_lints(if config.lint.exclude_lints.is_empty() {
139                    None
140                } else {
141                    Some(
142                        config
143                            .lint
144                            .exclude_lints
145                            .iter()
146                            .filter_map(|s| forge_lint::sol::SolLint::try_from(s.as_str()).ok())
147                            .collect(),
148                    )
149                });
150
151            // Expand ignore globs and canonicalize from the get go
152            let ignored = expand_globs(&config.root, config.lint.ignore.iter())?
153                .iter()
154                .flat_map(foundry_common::fs::canonicalize_path)
155                .collect::<Vec<_>>();
156
157            let skip = SkipBuildFilters::new(config.skip.clone(), config.root.clone());
158            let curr_dir = std::env::current_dir()?;
159            let input_files = config
160                .project_paths::<SolcLanguage>()
161                .input_files_iter()
162                .filter(|p| {
163                    skip.is_match(p)
164                        && !(ignored.contains(p) || ignored.contains(&curr_dir.join(p)))
165                })
166                .collect::<Vec<_>>();
167
168            if !input_files.is_empty() {
169                let sess = linter.init();
170
171                let pcx = solar_pcx_from_build_opts(
172                    &sess,
173                    &self.build,
174                    Some(project),
175                    Some(&input_files),
176                )?;
177                linter.early_lint(&input_files, pcx);
178
179                let pcx = solar_pcx_from_build_opts(
180                    &sess,
181                    &self.build,
182                    Some(project),
183                    Some(&input_files),
184                )?;
185                linter.late_lint(&input_files, pcx);
186            }
187        }
188
189        Ok(())
190    }
191
192    /// Returns the `Project` for the current workspace
193    ///
194    /// This loads the `foundry_config::Config` for the current workspace (see
195    /// [`foundry_config::utils::find_project_root`] and merges the cli `BuildArgs` into it before
196    /// returning [`foundry_config::Config::project()`]
197    pub fn project(&self) -> Result<Project> {
198        self.build.project()
199    }
200
201    /// Returns whether `BuildArgs` was configured with `--watch`
202    pub fn is_watch(&self) -> bool {
203        self.watch.watch.is_some()
204    }
205
206    /// Returns the [`watchexec::Config`] necessary to bootstrap a new watch loop.
207    pub(crate) fn watchexec_config(&self) -> Result<watchexec::Config> {
208        // Use the path arguments or if none where provided the `src`, `test` and `script`
209        // directories as well as the `foundry.toml` configuration file.
210        self.watch.watchexec_config(|| {
211            let config = self.load_config()?;
212            let foundry_toml: PathBuf = config.root.join(Config::FILE_NAME);
213            Ok([config.src, config.test, config.script, foundry_toml])
214        })
215    }
216}

218// Make this args a `figment::Provider` so that it can be merged into the `Config`
219impl Provider for BuildArgs {
220    fn metadata(&self) -> Metadata {
221        Metadata::named("Build Args Provider")
222    }
223
224    fn data(&self) -> Result<Map<Profile, Dict>, figment::Error> {
225        let value = Value::serialize(self)?;
226        let error = InvalidType(value.to_actual(), "map".into());
227        let mut dict = value.into_dict().ok_or(error)?;
228
229        if self.names {
230            dict.insert("names".to_string(), true.into());
231        }
232
233        if self.sizes {
234            dict.insert("sizes".to_string(), true.into());
235        }
236
237        if self.ignore_eip_3860 {
238            dict.insert("ignore_eip_3860".to_string(), true.into());
239        }
240
241        Ok(Map::from([(Config::selected_profile(), dict)]))
242    }
