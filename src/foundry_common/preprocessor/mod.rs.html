28/// Preprocessor that replaces static bytecode linking in tests and scripts (`new Contract`) with
29/// dynamic linkage through (`Vm.create*`).
30///
31/// This allows for more efficient caching when iterating on tests.
32///
33/// See <https://github.com/foundry-rs/foundry/pull/10010>.
34#[derive(Debug)]
35pub struct DynamicTestLinkingPreprocessor;
36
37impl Preprocessor<SolcCompiler> for DynamicTestLinkingPreprocessor {
38    #[instrument(name = "DynamicTestLinkingPreprocessor::preprocess", skip_all)]
39    fn preprocess(
40        &self,
41        _solc: &SolcCompiler,
42        input: &mut SolcVersionedInput,
43        paths: &ProjectPathsConfig<SolcLanguage>,
44        mocks: &mut HashSet<PathBuf>,
45    ) -> Result<()> {
46        // Skip if we are not preprocessing any tests or scripts. Avoids unnecessary AST parsing.
47        if !input.input.sources.iter().any(|(path, _)| paths.is_test_or_script(path)) {
48            trace!("no tests or sources to preprocess");
49            return Ok(());
50        }
51
52        let sess = solar_session_from_solc(input);
53        let _ = sess.enter_parallel(|| -> solar_parse::interface::Result {
54            // Set up the parsing context with the project paths.
55            let mut parsing_context = solar_pcx_from_solc_no_sources(&sess, input, paths);
56
57            // Add the sources into the context.
58            // Include all sources in the source map so as to not re-load them from disk, but only
59            // parse and preprocess tests and scripts.
60            let mut preprocessed_paths = vec![];
61            let sources = &mut input.input.sources;
62            for (path, source) in sources.iter() {
63                if let Ok(src_file) =
64                    sess.source_map().new_source_file(path.clone(), source.content.as_str())
65                    && paths.is_test_or_script(path)
66                {
67                    parsing_context.add_file(src_file);
68                    preprocessed_paths.push(path.clone());
69                }
70            }
71
72            // Parse and preprocess.
73            let hir_arena = ThreadLocal::new();
74            if let Some(gcx) = parsing_context.parse_and_lower(&hir_arena)? {
75                let hir = &gcx.get().hir;
76                // Collect tests and scripts dependencies and identify mock contracts.
77                let deps = PreprocessorDependencies::new(
78                    &sess,
79                    hir,
80                    &preprocessed_paths,
81                    &paths.paths_relative().sources,
82                    &paths.root,
83                    mocks,
84                );
85                // Collect data of source contracts referenced in tests and scripts.
86                let data = collect_preprocessor_data(&sess, hir, &deps.referenced_contracts);
87
88                // Extend existing sources with preprocessor deploy helper sources.
89                sources.extend(create_deploy_helpers(&data));
90
91                // Generate and apply preprocessor source updates.
92                apply_updates(sources, remove_bytecode_dependencies(hir, &deps, &data));
93            }
94
95            Ok(())
96        });
97
98        // Warn if any diagnostics emitted during content parsing.
99        if let Err(err) = sess.emitted_errors().unwrap() {
100            warn!("failed preprocessing {err}");
101        }
102
103        Ok(())
104    }
105}

107impl Preprocessor<MultiCompiler> for DynamicTestLinkingPreprocessor {
108    fn preprocess(
109        &self,
110        compiler: &MultiCompiler,
111        input: &mut <MultiCompiler as Compiler>::Input,
112        paths: &ProjectPathsConfig<MultiCompilerLanguage>,
113        mocks: &mut HashSet<PathBuf>,
114    ) -> Result<()> {
115        // Preprocess only Solc compilers.
116        let MultiCompilerInput::Solc(input) = input else { return Ok(()) };
117
118        let Some(solc) = &compiler.solc else { return Ok(()) };
119
120        let paths = paths.clone().with_language::<SolcLanguage>();
121        self.preprocess(solc, input, &paths, mocks)
122    }
123}

125fn solar_session_from_solc(solc: &SolcVersionedInput) -> Session {
126    use solar_parse::interface::config;
127
128    Session::builder()
129        .with_buffer_emitter(Default::default())
130        .opts(config::Opts {
131            language: match solc.input.language {
132                SolcLanguage::Solidity => config::Language::Solidity,
133                SolcLanguage::Yul => config::Language::Yul,
134                _ => unimplemented!(),
135            },
136
137            // TODO: ...
138            /*
139            evm_version: solc.input.settings.evm_version,
140            */
141            ..Default::default()
142        })
143        .build()
144}

146fn solar_pcx_from_solc_no_sources<'sess>(
147    sess: &'sess Session,
148    solc: &SolcVersionedInput,
149    paths: &ProjectPathsConfig<impl Language>,
150) -> ParsingContext<'sess> {
151    let mut pcx = ParsingContext::new(sess);
152    pcx.file_resolver.set_current_dir(solc.cli_settings.base_path.as_ref().unwrap_or(&paths.root));
153    for remapping in &paths.remappings {
154        pcx.file_resolver.add_import_remapping(solar_sema::interface::config::ImportRemapping {
155            context: remapping.context.clone().unwrap_or_default(),
156            prefix: remapping.name.clone(),
157            path: remapping.path.clone(),
158        });
159    }
160    pcx.file_resolver.add_include_paths(solc.cli_settings.include_paths.iter().cloned());
161    pcx
162}
