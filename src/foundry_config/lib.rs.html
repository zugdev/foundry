133/// Foundry configuration
134///
135/// # Defaults
136///
137/// All configuration values have a default, documented in the [fields](#fields)
138/// section below. [`Config::default()`] returns the default values for
139/// the default profile while [`Config::with_root()`] returns the values based on the given
140/// directory. [`Config::load()`] starts with the default profile and merges various providers into
141/// the config, same for [`Config::load_with_root()`], but there the default values are determined
142/// by [`Config::with_root()`]
143///
144/// # Provider Details
145///
146/// `Config` is a Figment [`Provider`] with the following characteristics:
147///
148///   * **Profile**
149///
150///     The profile is set to the value of the `profile` field.
151///
152///   * **Metadata**
153///
154///     This provider is named `Foundry Config`. It does not specify a
155///     [`Source`](figment::Source) and uses default interpolation.
156///
157///   * **Data**
158///
159///     The data emitted by this provider are the keys and values corresponding
160///     to the fields and values of the structure. The dictionary is emitted to
161///     the "default" meta-profile.
162///
163/// Note that these behaviors differ from those of [`Config::figment()`].
164#[derive(Clone, Debug, PartialEq, Serialize, Deserialize)]

170    /// selected on the `Figment`. When a `Config` is extracted, this field is
171    /// set to the extracting Figment's selected `Profile`.

679    fn from_figment(figment: Figment) -> Result<Self, ExtractConfigError> {
680        let mut config = figment.extract::<Self>().map_err(ExtractConfigError::new)?;
681        config.profile = figment.profile().clone();
682
683        // The `"profile"` profile contains all the profiles as keys.
684        let mut add_profile = |profile: &Profile| {
685            if !config.profiles.contains(profile) {
686                config.profiles.push(profile.clone());
687            }
688        };
689        let figment = figment.select(Self::PROFILE_SECTION);
690        if let Ok(data) = figment.data()
691            && let Some(profiles) = data.get(&Profile::new(Self::PROFILE_SECTION))
692        {
693            for profile in profiles.keys() {
694                add_profile(&Profile::new(profile));
695            }
696        }
697        add_profile(&Self::DEFAULT_PROFILE);
698        add_profile(&config.profile);
699
700        config.normalize_optimizer_settings();
701
702        Ok(config)
703    }

705    /// Returns the populated [Figment] using the requested [FigmentProviders] preset.
706    ///
707    /// This will merge various providers, such as env,toml,remappings into the figment if
708    /// requested.
709    pub fn to_figment(&self, providers: FigmentProviders) -> Figment {
710        // Note that `Figment::from` here is a method on `Figment` rather than the `From` impl below
711
712        if providers.is_none() {
713            return Figment::from(self);
714        }
715
716        let root = self.root.as_path();
717        let profile = Self::selected_profile();
718        let mut figment = Figment::default().merge(DappHardhatDirProvider(root));
719
720        // merge global foundry.toml file
721        if let Some(global_toml) = Self::foundry_dir_toml().filter(|p| p.exists()) {
722            figment = Self::merge_toml_provider(
723                figment,
724                TomlFileProvider::new(None, global_toml).cached(),
725                profile.clone(),
726            );
727        }
728        // merge local foundry.toml file
729        figment = Self::merge_toml_provider(
730            figment,
731            TomlFileProvider::new(Some("FOUNDRY_CONFIG"), root.join(Self::FILE_NAME)).cached(),
732            profile.clone(),
733        );
734
735        // merge environment variables
736        figment = figment
737            .merge(
738                Env::prefixed("DAPP_")
739                    .ignore(&["REMAPPINGS", "LIBRARIES", "FFI", "FS_PERMISSIONS"])
740                    .global(),
741            )
742            .merge(
743                Env::prefixed("DAPP_TEST_")
744                    .ignore(&["CACHE", "FUZZ_RUNS", "DEPTH", "FFI", "FS_PERMISSIONS"])
745                    .global(),
746            )
747            .merge(DappEnvCompatProvider)
748            .merge(EtherscanEnvProvider::default())
749            .merge(
750                Env::prefixed("FOUNDRY_")
751                    .ignore(&["PROFILE", "REMAPPINGS", "LIBRARIES", "FFI", "FS_PERMISSIONS"])
752                    .map(|key| {
753                        let key = key.as_str();
754                        if Self::STANDALONE_SECTIONS.iter().any(|section| {
755                            key.starts_with(&format!("{}_", section.to_ascii_uppercase()))
756                        }) {
757                            key.replacen('_', ".", 1).into()
758                        } else {
759                            key.into()
760                        }
761                    })
762                    .global(),
763            )
764            .select(profile.clone());
765
766        // only resolve remappings if all providers are requested
767        if providers.is_all() {
768            // we try to merge remappings after we've merged all other providers, this prevents
769            // redundant fs lookups to determine the default remappings that are eventually updated
770            // by other providers, like the toml file
771            let remappings = RemappingsProvider {
772                auto_detect_remappings: figment
773                    .extract_inner::<bool>("auto_detect_remappings")
774                    .unwrap_or(true),
775                lib_paths: figment
776                    .extract_inner::<Vec<PathBuf>>("libs")
777                    .map(Cow::Owned)
778                    .unwrap_or_else(|_| Cow::Borrowed(&self.libs)),
779                root,
780                remappings: figment.extract_inner::<Vec<Remapping>>("remappings"),
781            };
782            figment = figment.merge(remappings);
783        }
784
785        // normalize defaults
786        figment = self.normalize_defaults(figment);
787
788        Figment::from(self).merge(figment).select(profile)
789    }

801    /// Joins all relative paths with the given root so that paths that are defined as:
802    ///
803    /// ```toml
804    /// [profile.default]
805    /// src = "src"
806    /// out = "./out"
807    /// libs = ["lib", "/var/lib"]
808    /// ```
809    ///
810    /// Will be made canonic with the given root:
811    ///
812    /// ```toml
813    /// [profile.default]
814    /// src = "<root>/src"
815    /// out = "<root>/out"
816    /// libs = ["<root>/lib", "/var/lib"]
817    /// ```
818    #[must_use]
819    pub fn canonic_at(mut self, root: impl Into<PathBuf>) -> Self {
820        let root = canonic(root);
821
822        fn p(root: &Path, rem: &Path) -> PathBuf {
823            canonic(root.join(rem))
824        }
825
826        self.src = p(&root, &self.src);
827        self.test = p(&root, &self.test);
828        self.script = p(&root, &self.script);
829        self.out = p(&root, &self.out);
830        self.broadcast = p(&root, &self.broadcast);
831        self.cache_path = p(&root, &self.cache_path);
832        self.snapshots = p(&root, &self.snapshots);
833
834        if let Some(build_info_path) = self.build_info_path {
835            self.build_info_path = Some(p(&root, &build_info_path));
836        }
837
838        self.libs = self.libs.into_iter().map(|lib| p(&root, &lib)).collect();
839
840        self.remappings =
841            self.remappings.into_iter().map(|r| RelativeRemapping::new(r.into(), &root)).collect();
842
843        self.allow_paths = self.allow_paths.into_iter().map(|allow| p(&root, &allow)).collect();
844
845        self.include_paths = self.include_paths.into_iter().map(|allow| p(&root, &allow)).collect();
846
847        self.fs_permissions.join_all(&root);
848
849        if let Some(model_checker) = &mut self.model_checker {
850            model_checker.contracts = std::mem::take(&mut model_checker.contracts)
851                .into_iter()
852                .map(|(path, contracts)| {
853                    (format!("{}", p(&root, path.as_ref()).display()), contracts)
854                })
855                .collect();
856        }
857
858        self
859    }

881    /// - if optimizer is set and optimizer runs not specified, then optimizer runs is set to 200

884        match (self.optimizer, self.optimizer_runs) {

893            (None, Some(runs)) => self.optimizer = Some(runs > 0),
894            _ => {}
895        }

989    /// Resolves globs and builds a mapping from individual source files to their restrictions
990    #[expect(clippy::disallowed_macros)]
991    fn restrictions(
992        &self,
993        paths: &ProjectPathsConfig,
994    ) -> Result<BTreeMap<PathBuf, RestrictionsWithVersion<MultiCompilerRestrictions>>, SolcError>
995    {
996        let mut map = BTreeMap::new();
997        if self.compilation_restrictions.is_empty() {
998            return Ok(BTreeMap::new());
999        }
1000
1001        let graph = Graph::<MultiCompilerParsedSource>::resolve(paths)?;
1002        let (sources, _) = graph.into_sources();
1003
1004        for res in &self.compilation_restrictions {
1005            for source in sources.keys().filter(|path| {
1006                if res.paths.is_match(path) {
1007                    true
1008                } else if let Ok(path) = path.strip_prefix(&paths.root) {
1009                    res.paths.is_match(path)
1010                } else {
1011                    false
1012                }
1013            }) {
1014                let res: RestrictionsWithVersion<_> =
1015                    res.clone().try_into().map_err(SolcError::msg)?;
1016                if !map.contains_key(source) {
1017                    map.insert(source.clone(), res);
1018                } else {
1019                    let value = map.remove(source.as_path()).unwrap();
1020                    if let Some(merged) = value.clone().merge(res) {
1021                        map.insert(source.clone(), merged);
1022                    } else {
1023                        // `sh_warn!` is a circular dependency, preventing us from using it here.
1024                        eprintln!(
1025                            "{}",
1026                            yansi::Paint::yellow(&format!(
1027                                "Failed to merge compilation restrictions for {}",
1028                                source.display()
1029                            ))
1030                        );
1031                        map.insert(source.clone(), value);
1032                    }
1033                }
1034            }
1035        }
1036
1037        Ok(map)
1038    }

1040    /// Creates a [`Project`] with the given `cached` and `no_artifacts` flags.
1041    ///
1042    /// Prefer using [`Self::project`] or [`Self::ephemeral_project`] instead.
1043    pub fn create_project(&self, cached: bool, no_artifacts: bool) -> Result<Project, SolcError> {
1044        let settings = self.compiler_settings()?;
1045        let paths = self.project_paths();
1046        let mut builder = Project::builder()
1047            .artifacts(self.configured_artifacts_handler())
1048            .additional_settings(self.additional_settings(&settings))
1049            .restrictions(self.restrictions(&paths)?)
1050            .settings(settings)
1051            .paths(paths)
1052            .ignore_error_codes(self.ignored_error_codes.iter().copied().map(Into::into))
1053            .ignore_paths(self.ignored_file_paths.clone())
1054            .set_compiler_severity_filter(if self.deny_warnings {
1055                Severity::Warning
1056            } else {
1057                Severity::Error
1058            })
1059            .set_offline(self.offline)
1060            .set_cached(cached)
1061            .set_build_info(!no_artifacts && self.build_info)
1062            .set_no_artifacts(no_artifacts);
1063
1064        if !self.skip.is_empty() {
1065            let filter = SkipBuildFilters::new(self.skip.clone(), self.root.clone());
1066            builder = builder.sparse_output(filter);
1067        }
1068
1069        let project = builder.build(self.compiler()?)?;
1070
1071        if self.force {
1072            self.cleanup(&project)?;
1073        }
1074
1075        Ok(project)
1076    }

1122                        }

1124                    }
1125                }
1126                SolcReq::Local(solc) => {

1128                        return Err(SolcError::msg(format!(
1129                            "`solc` {} does not exist",
1130                            solc.display()

1132                    }
1133                    Solc::new(solc)?
1134                }
1135            };

1287    ///
1288    /// # Returns
1289    ///
1290    /// In order of resolution:
1291    ///
1292    /// - the matching, resolved url of `rpc_endpoints` if `maybe_alias` is an alias
1293    /// - a mesc resolved url if `maybe_alias` is a known alias in mesc
1294    /// - `None` otherwise
1295    ///
1296    /// # Note on mesc
1297    ///
1298    /// The endpoint is queried for in mesc under the `foundry` profile, allowing users to customize
1299    /// endpoints for Foundry specifically.
1300    ///
1301    /// # Example
1302    ///
1303    /// ```
1304    /// use foundry_config::Config;
1305    /// # fn t() {

1311        &self,
1312        maybe_alias: &str,
1313    ) -> Option<Result<Cow<'_, str>, UnresolvedEnvVarError>> {
1314        let mut endpoints = self.rpc_endpoints.clone().resolved();
1315        if let Some(endpoint) = endpoints.remove(maybe_alias) {
1316            return Some(endpoint.url().map(Cow::Owned));
1317        }
1318
1319        if let Some(mesc_url) = self.get_rpc_url_from_mesc(maybe_alias) {
1320            return Some(Ok(Cow::Owned(mesc_url)));
1321        }
1322
1323        None
1324    }

1326    /// Attempts to resolve the URL for the given alias from [`mesc`](https://github.com/paradigmxyz/mesc)
1327    pub fn get_rpc_url_from_mesc(&self, maybe_alias: &str) -> Option<String> {
1328        // Note: mesc requires a MESC_PATH in the env, which the user can configure and is expected
1329        // to be part of the shell profile, default is ~/mesc.json
1330        let mesc_config = mesc::load::load_config_data()
1331            .inspect_err(|err| debug!(%err, "failed to load mesc config"))
1332            .ok()?;
1333
1334        if let Ok(Some(endpoint)) =
1335            mesc::query::get_endpoint_by_query(&mesc_config, maybe_alias, Some("foundry"))
1336        {
1337            return Some(endpoint.url);
1338        }
1339
1340        if maybe_alias.chars().all(|c| c.is_numeric()) {
1341            // try to lookup the mesc network by chain id if alias is numeric
1342            // This only succeeds if the chain id has a default:
1343            // "network_defaults": {
1344            //    "50104": "sophon_50104"
1345            // }
1346            if let Ok(Some(endpoint)) =
1347                mesc::query::get_endpoint_by_network(&mesc_config, maybe_alias, Some("foundry"))
1348            {
1349                return Some(endpoint.url);
1350            }
1351        }
1352
1353        None
1354    }

1414    /// Same as [`Self::get_etherscan_config()`] but optionally updates the config with the given
1415    /// `chain`, and `etherscan_api_key`
1416    ///
1417    /// If not matching alias was found, then this will try to find the first entry in the table
1418    /// with a matching chain id. If an etherscan_api_key is already set it will take precedence
1419    /// over the chain's entry in the table.
1420    pub fn get_etherscan_config_with_chain(
1421        &self,
1422        chain: Option<Chain>,
1423    ) -> Result<Option<ResolvedEtherscanConfig>, EtherscanConfigError> {
1424        let default_api_version = self.etherscan_api_version.unwrap_or_default();
1425
1426        if let Some(maybe_alias) = self.etherscan_api_key.as_ref().or(self.eth_rpc_url.as_ref())
1427            && self.etherscan.contains_key(maybe_alias)
1428        {
1429            return self
1430                .etherscan
1431                .clone()
1432                .resolved(default_api_version)
1433                .remove(maybe_alias)
1434                .transpose();
1435        }
1436
1437        // try to find by comparing chain IDs after resolving
1438        if let Some(res) = chain.or(self.chain).and_then(|chain| {
1439            self.etherscan.clone().resolved(default_api_version).find_chain(chain)
1440        }) {
1441            match (res, self.etherscan_api_key.as_ref()) {
1442                (Ok(mut config), Some(key)) => {
1443                    // we update the key, because if an etherscan_api_key is set, it should take
1444                    // precedence over the entry, since this is usually set via env var or CLI args.
1445                    config.key.clone_from(key);
1446                    return Ok(Some(config));
1447                }
1448                (Ok(config), None) => return Ok(Some(config)),
1449                (Err(err), None) => return Err(err),
1450                (Err(_), Some(_)) => {
1451                    // use the etherscan key as fallback
1452                }
1453            }
1454        }
1455
1456        // etherscan fallback via API key
1457        if let Some(key) = self.etherscan_api_key.as_ref() {
1458            match ResolvedEtherscanConfig::create(
1459                key,
1460                chain.or(self.chain).unwrap_or_default(),
1461                default_api_version,
1462            ) {
1463                Some(config) => return Ok(Some(config)),
1464                None => {
1465                    return Err(EtherscanConfigError::UnknownChain(
1466                        String::new(),
1467                        chain.unwrap_or_default(),
1468                    ));
1469                }
1470            }
1471        }
1472        Ok(None)
1473    }

1803    /// Serialize the config type as a String of TOML.
1804    ///
1805    /// This serializes to a table with the name of the profile
1806    ///
1807    /// ```toml
1808    /// [profile.default]
1809    /// src = "src"
1810    /// out = "out"
1811    /// libs = ["lib"]
1812    /// # ...
1813    /// ```
1814    pub fn to_string_pretty(&self) -> Result<String, toml::ser::Error> {
1815        // serializing to value first to prevent `ValueAfterTable` errors
1816        let mut value = toml::Value::try_from(self)?;
1817        // Config map always gets serialized as a table
1818        let value_table = value.as_table_mut().unwrap();
1819        // remove standalone sections from inner table
1820        let standalone_sections = Self::STANDALONE_SECTIONS
1821            .iter()

1835        for (section, value) in standalone_sections {

1837        }

2488/// Wrapper for the config's `gas_limit` value necessary because toml-rs can't handle larger number
2489/// because integers are stored signed: <https://github.com/alexcrichton/toml-rs/issues/256>
2490///
2491/// Due to this limitation this type will be serialized/deserialized as String if it's larger than
2492/// `i64`
2493#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Deserialize)]
2494pub struct GasLimit(#[serde(deserialize_with = "crate::deserialize_u64_or_max")] pub u64);
2495
2496impl From<u64> for GasLimit {
2497    fn from(gas: u64) -> Self {
2498        Self(gas)
2499    }
2500}
2501
2502impl From<GasLimit> for u64 {
2503    fn from(gas: GasLimit) -> Self {
2504        gas.0
2505    }
2506}
2507
2508impl Serialize for GasLimit {
2509    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
2510    where
2511        S: Serializer,
2512    {
2513        if self.0 == u64::MAX {
2514            serializer.serialize_str("max")
2515        } else if self.0 > i64::MAX as u64 {
2516            serializer.serialize_str(&self.0.to_string())
2517        } else {
2518            serializer.serialize_u64(self.0)
2519        }
2520    }
2521}

2523/// Variants for selecting the [`Solc`] instance
2524#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
2525#[serde(untagged)]
2526pub enum SolcReq {
2527    /// Requires a specific solc version, that's either already installed (via `svm`) or will be
2528    /// auto installed (via `svm`)
2529    Version(Version),
2530    /// Path to an existing local solc installation
2531    Local(PathBuf),
2532}
2533
2534impl SolcReq {
2535    /// Tries to get the solc version from the `SolcReq`
2536    ///
2537    /// If the `SolcReq` is a `Version` it will return the version, if it's a path to a binary it
2538    /// will try to get the version from the binary.
2539    fn try_version(&self) -> Result<Version, SolcError> {
2540        match self {
2541            Self::Version(version) => Ok(version.clone()),
2542            Self::Local(path) => Solc::new(path).map(|solc| solc.version),
2543        }
2544    }
2545}
2546
2547impl<T: AsRef<str>> From<T> for SolcReq {
2548    fn from(s: T) -> Self {
2549        let s = s.as_ref();
2550        if let Ok(v) = Version::from_str(s) { Self::Version(v) } else { Self::Local(s.into()) }
2551    }
2552}
