46/// A convenience provider to retrieve a toml file.
47/// This will return an error if the env var is set but the file does not exist
48pub(crate) struct TomlFileProvider {
49    pub env_var: Option<&'static str>,
50    pub default: PathBuf,
51    pub cache: Option<Result<Map<Profile, Dict>, Error>>,
52}

54impl TomlFileProvider {
55    pub(crate) fn new(env_var: Option<&'static str>, default: impl Into<PathBuf>) -> Self {
56        Self { env_var, default: default.into(), cache: None }
57    }
58
59    fn env_val(&self) -> Option<String> {
60        self.env_var.and_then(Env::var)
61    }
62
63    fn file(&self) -> PathBuf {
64        self.env_val().map(PathBuf::from).unwrap_or_else(|| self.default.clone())
65    }
66
67    fn is_missing(&self) -> bool {
68        if let Some(file) = self.env_val() {
69            let path = Path::new(&file);
70            if !path.exists() {
71                return true;
72            }
73        }
74        false
75    }

77    pub(crate) fn cached(mut self) -> Self {
78        self.cache = Some(self.read());

82    /// Reads and processes the TOML configuration file, handling inheritance if configured.
83    fn read(&self) -> Result<Map<Profile, Dict>, Error> {
84        use serde::de::Error as _;
85
86        // Get the config file path and validate it exists
87        let local_path = self.file();
88        if !local_path.exists() {
89            if let Some(file) = self.env_val() {
90                return Err(Error::custom(format!(
91                    "Config file `{}` set in env var `{}` does not exist",
92                    file,
93                    self.env_var.unwrap()
94                )));
95            }
96            return Ok(Map::new());
97        }
98
99        // Create a provider for the local config file
100        let local_provider = Toml::file(local_path.clone()).nested();
101
102        // Parse the local config to check for extends field
103        let local_path_str = local_path.to_string_lossy();
104        let local_content = std::fs::read_to_string(&local_path)
105            .map_err(|e| Error::custom(e.to_string()).with_path(&local_path_str))?;
106        let partial_config: extend::ExtendsPartialConfig = toml::from_str(&local_content)
107            .map_err(|e| Error::custom(e.to_string()).with_path(&local_path_str))?;
108
109        // Check if the currently active profile has an 'extends' field
110        let selected_profile = Config::selected_profile();
111        let extends_config = partial_config.profile.as_ref().and_then(|profiles| {
112            let profile_str = selected_profile.to_string();
113            profiles.get(&profile_str).and_then(|cfg| cfg.extends.as_ref())
114        });
115
116        // If inheritance is configured, load and merge the base config
117        if let Some(extends_config) = extends_config {
118            let extends_path = extends_config.path();
119            let extends_strategy = extends_config.strategy();
120            let relative_base_path = PathBuf::from(extends_path);
121            let local_dir = local_path.parent().ok_or_else(|| {
122                Error::custom(format!(
123                    "Could not determine parent directory of config file: {}",
124                    local_path.display()
125                ))
126            })?;
127
128            let base_path =
129                foundry_compilers::utils::canonicalize(local_dir.join(&relative_base_path))
130                    .map_err(|e| {
131                        Error::custom(format!(
132                            "Failed to resolve inherited config path: {}: {e}",
133                            relative_base_path.display()
134                        ))
135                    })?;
136
137            // Validate the base config file exists
138            if !base_path.is_file() {
139                return Err(Error::custom(format!(
140                    "Inherited config file does not exist or is not a file: {}",
141                    base_path.display()
142                )));
143            }
144
145            // Prevent self-inheritance which would cause infinite recursion
146            if foundry_compilers::utils::canonicalize(&local_path).ok().as_ref() == Some(&base_path)
147            {
148                return Err(Error::custom(format!(
149                    "Config file {} cannot inherit from itself.",
150                    local_path.display()
151                )));
152            }
153
154            // Parse the base config to check for nested inheritance
155            let base_path_str = base_path.to_string_lossy();
156            let base_content = std::fs::read_to_string(&base_path)
157                .map_err(|e| Error::custom(e.to_string()).with_path(&base_path_str))?;
158            let base_partial: extend::ExtendsPartialConfig = toml::from_str(&base_content)
159                .map_err(|e| Error::custom(e.to_string()).with_path(&base_path_str))?;
160
161            // Check if the base file's same profile also has extends (nested inheritance)
162            let base_extends = base_partial
163                .profile
164                .as_ref()
165                .and_then(|profiles| {
166                    let profile_str = selected_profile.to_string();
167                    profiles.get(&profile_str)
168                })
169                .and_then(|profile| profile.extends.as_ref());
170
171            // Prevent nested inheritance to avoid complexity and potential cycles
172            if base_extends.is_some() {
173                return Err(Error::custom(format!(
174                    "Nested inheritance is not allowed. Base file '{}' cannot have an 'extends' field in profile '{selected_profile}'.",
175                    base_path.display()
176                )));
177            }
178
179            // Load base configuration as a Figment provider
180            let base_provider = Toml::file(base_path).nested();
181
182            // Apply the selected merge strategy
183            match extends_strategy {
184                extend::ExtendStrategy::ExtendArrays => {
185                    // Using 'admerge' strategy:
186                    // - Arrays are concatenated (base elements + local elements)
187                    // - Other values are replaced (local values override base values)
188                    // - The extends field is preserved in the final configuration
189                    Figment::new().merge(base_provider).admerge(local_provider).data()
190                }
191                extend::ExtendStrategy::ReplaceArrays => {
192                    // Using 'merge' strategy:
193                    // - Arrays are replaced entirely (local arrays replace base arrays)
194                    // - Other values are replaced (local values override base values)
195                    Figment::new().merge(base_provider).merge(local_provider).data()
196                }
197                extend::ExtendStrategy::NoCollision => {
198                    // Check for key collisions between base and local configs
199                    let base_data = base_provider.data()?;
200                    let local_data = local_provider.data()?;
201
202                    let profile_key = Profile::new("profile");
203                    if let (Some(local_profiles), Some(base_profiles)) =
204                        (local_data.get(&profile_key), base_data.get(&profile_key))
205                    {
206                        // Extract dicts for the selected profile
207                        let profile_str = selected_profile.to_string();
208                        let base_dict = base_profiles.get(&profile_str).and_then(|v| v.as_dict());
209                        let local_dict = local_profiles.get(&profile_str).and_then(|v| v.as_dict());
210
211                        // Find colliding keys
212                        if let (Some(local_dict), Some(base_dict)) = (local_dict, base_dict) {
213                            let collisions: Vec<&String> = local_dict
214                                .keys()
215                                .filter(|key| {
216                                    // Ignore the "extends" key as it's expected
217                                    *key != "extends" && base_dict.contains_key(*key)
218                                })
219                                .collect();
220
221                            if !collisions.is_empty() {
222                                return Err(Error::custom(format!(
223                                    "Key collision detected in profile '{profile_str}' when extending '{extends_path}'. \
224                                    Conflicting keys: {collisions:?}. Use 'extends.strategy' or 'extends_strategy' to specify how to handle conflicts."
225                                )));
226                            }
227                        }
228                    }
229
230                    // Safe to merge the configs without collisions
231                    Figment::new().merge(base_provider).merge(local_provider).data()
232                }
233            }
234        } else {
235            // No inheritance - return the local config as-is
236            local_provider.data()
237        }
238    }

255/// A Provider that ensures all keys are snake case if they're not standalone sections, See
256/// `Config::STANDALONE_SECTIONS`
257pub(crate) struct ForcedSnakeCaseData<P>(pub(crate) P);
258
259impl<P: Provider> Provider for ForcedSnakeCaseData<P> {
260    fn metadata(&self) -> Metadata {
261        self.0.metadata()
262    }
263
264    fn data(&self) -> Result<Map<Profile, Dict>, Error> {
265        let mut map = Map::new();
266        for (profile, dict) in self.0.data()? {
267            if Config::STANDALONE_SECTIONS.contains(&profile.as_ref()) {
268                // don't force snake case for keys in standalone sections
269                map.insert(profile, dict);
270                continue;
271            }
272            map.insert(profile, dict.into_iter().map(|(k, v)| (k.to_snake_case(), v)).collect());
273        }
274        Ok(map)
275    }
276}

278/// A Provider that handles breaking changes in toml files
279pub(crate) struct BackwardsCompatTomlProvider<P>(pub(crate) P);
280
281impl<P: Provider> Provider for BackwardsCompatTomlProvider<P> {
282    fn metadata(&self) -> Metadata {
283        self.0.metadata()
284    }
285
286    fn data(&self) -> Result<Map<Profile, Dict>, Error> {
287        let mut map = Map::new();
288        let solc_env = std::env::var("FOUNDRY_SOLC_VERSION")
289            .or_else(|_| std::env::var("DAPP_SOLC_VERSION"))
290            .map(Value::from)
291            .ok();
292        for (profile, mut dict) in self.0.data()? {
293            if let Some(v) = solc_env.clone() {
294                // ENV var takes precedence over config file
295                dict.insert("solc".to_string(), v);
296            } else if let Some(v) = dict.remove("solc_version") {
297                // only insert older variant if not already included
298                if !dict.contains_key("solc") {
299                    dict.insert("solc".to_string(), v);
300                }
301            }
302
303            if let Some(v) = dict.remove("odyssey") {
304                dict.insert("odyssey".to_string(), v);
305            }
306            map.insert(profile, dict);
307        }
308        Ok(map)
309    }
310}

312/// A provider that sets the `src` and `output` path depending on their existence.
313pub(crate) struct DappHardhatDirProvider<'a>(pub(crate) &'a Path);
314
315impl Provider for DappHardhatDirProvider<'_> {
316    fn metadata(&self) -> Metadata {
317        Metadata::named("Dapp Hardhat dir compat")
318    }
319
320    fn data(&self) -> Result<Map<Profile, Dict>, Error> {
321        let mut dict = Dict::new();
322        dict.insert(
323            "src".to_string(),
324            ProjectPathsConfig::find_source_dir(self.0)
325                .file_name()
326                .unwrap()
327                .to_string_lossy()
328                .to_string()
329                .into(),
330        );
331        dict.insert(
332            "out".to_string(),
333            ProjectPathsConfig::find_artifacts_dir(self.0)
334                .file_name()
335                .unwrap()
336                .to_string_lossy()
337                .to_string()
338                .into(),
339        );
340
341        // detect libs folders:
342        //   if `lib` _and_ `node_modules` exists: include both
343        //   if only `node_modules` exists: include `node_modules`
344        //   include `lib` otherwise
345        let mut libs = vec![];
346        let node_modules = self.0.join("node_modules");
347        let lib = self.0.join("lib");
348        if node_modules.exists() {
349            if lib.exists() {
350                libs.push(lib.file_name().unwrap().to_string_lossy().to_string());
351            }
352            libs.push(node_modules.file_name().unwrap().to_string_lossy().to_string());
353        } else {
354            libs.push(lib.file_name().unwrap().to_string_lossy().to_string());
355        }
356
357        dict.insert("libs".to_string(), libs.into());
358
359        Ok(Map::from([(Config::selected_profile(), dict)]))
360    }
361}

427                "runs".to_string(),
428                val.parse::<u32>().map_err(figment::Error::custom)?.into(),
429            );
430        }

432

435            invariant_dict.insert(
436                "depth".to_string(),
437                val.parse::<u32>().map_err(figment::Error::custom)?.into(),
438            );
439        }
440        dict.insert("invariant".to_string(), invariant_dict.into());
441
442        Ok(Map::from([(Config::selected_profile(), dict)]))
443    }

489/// Unwraps a profile reducing the key depth
490///
491/// For example given:
492///
493/// ```toml
494/// [wrapping_key.profile]
495/// key = "value"
496/// ```
497///
498/// UnwrapProfileProvider will output:
499///
500/// ```toml
501/// [profile]
502/// key = "value"
503/// ```
504struct UnwrapProfileProvider<P> {
505    provider: P,
506    wrapping_key: Profile,
507    profile: Profile,
508}
509
510impl<P> UnwrapProfileProvider<P> {
511    pub fn new(provider: P, wrapping_key: impl Into<Profile>, profile: impl Into<Profile>) -> Self {
512        Self { provider, wrapping_key: wrapping_key.into(), profile: profile.into() }
513    }
514}
515
516impl<P: Provider> Provider for UnwrapProfileProvider<P> {
517    fn metadata(&self) -> Metadata {
518        self.provider.metadata()
519    }
520    fn data(&self) -> Result<Map<Profile, Dict>, Error> {
521        self.provider.data().and_then(|mut data| {
522            if let Some(profiles) = data.remove(&self.wrapping_key) {
523                for (profile_str, profile_val) in profiles {
524                    let profile = Profile::new(&profile_str);
525                    if profile != self.profile {
526                        continue;
527                    }
528                    match profile_val {
529                        Value::Dict(_, dict) => return Ok(profile.collect(dict)),
530                        bad_val => {
531                            let mut err = Error::from(figment::error::Kind::InvalidType(
532                                bad_val.to_actual(),
533                                "dict".into(),
534                            ));
535                            err.metadata = Some(self.provider.metadata());
536                            err.profile = Some(self.profile.clone());
537                            return Err(err);
538                        }
539                    }
540                }
541            }
542            Ok(Default::default())
543        })
544    }
545    fn profile(&self) -> Option<Profile> {
546        Some(self.profile.clone())
547    }
548}

595/// Extracts the profile from the `profile` key and using the original key as backup, merging
596/// values where necessary
597///
598/// For example given:
599///
600/// ```toml
601/// [profile.cool]
602/// key = "value"
603///
604/// [cool]
605/// key2 = "value2"
606/// ```
607///
608/// OptionalStrictProfileProvider will output:
609///
610/// ```toml
611/// [cool]
612/// key = "value"
613/// key2 = "value2"
614/// ```
615///
616/// And emit a deprecation warning
617pub(crate) struct OptionalStrictProfileProvider<P> {
618    provider: P,
619    profiles: Vec<Profile>,
620}
621
622impl<P> OptionalStrictProfileProvider<P> {
623    pub const PROFILE_PROFILE: Profile = Profile::const_new("profile");
624
625    pub fn new(provider: P, profiles: impl IntoIterator<Item = impl Into<Profile>>) -> Self {
626        Self { provider, profiles: profiles.into_iter().map(|profile| profile.into()).collect() }
627    }
628}
629
630impl<P: Provider> Provider for OptionalStrictProfileProvider<P> {
631    fn metadata(&self) -> Metadata {
632        self.provider.metadata()
633    }
634    fn data(&self) -> Result<Map<Profile, Dict>, Error> {
635        let mut figment = Figment::from(&self.provider);
636        for profile in &self.profiles {
637            figment = figment.merge(UnwrapProfileProvider::new(
638                &self.provider,
639                Self::PROFILE_PROFILE,
640                profile.clone(),
641            ));
642        }
643        figment.data().map_err(|err| {
644            // figment does tag metadata and tries to map metadata to an error, since we use a new
645            // figment in this provider this new figment does not know about the metadata of the
646            // provider and can't map the metadata to the error. Therefore we return the root error
647            // if this error originated in the provider's data.
648            if let Err(root_err) = self.provider.data() {
649                return root_err;
650            }
651            err
652        })
653    }
654    fn profile(&self) -> Option<Profile> {
655        self.profiles.last().cloned()
656    }
657}

659/// Extracts the profile from the `profile` key and sets unset values according to the fallback
660/// provider
661pub struct FallbackProfileProvider<P> {
662    provider: P,
663    profile: Profile,
664    fallback: Profile,
665}
666
667impl<P> FallbackProfileProvider<P> {
668    /// Creates a new fallback profile provider.
669    pub fn new(provider: P, profile: impl Into<Profile>, fallback: impl Into<Profile>) -> Self {
670        Self { provider, profile: profile.into(), fallback: fallback.into() }
671    }
672}
673
674impl<P: Provider> Provider for FallbackProfileProvider<P> {
675    fn metadata(&self) -> Metadata {
676        self.provider.metadata()
677    }
678
679    fn data(&self) -> Result<Map<Profile, Dict>, Error> {
680        let data = self.provider.data()?;
681        if let Some(fallback) = data.get(&self.fallback) {
682            let mut inner = data.get(&self.profile).cloned().unwrap_or_default();
683            for (k, v) in fallback {
684                if !inner.contains_key(k) {
685                    inner.insert(k.to_owned(), v.clone());
686                }
687            }
688            Ok(self.profile.collect(inner))
689        } else {
690            Ok(data)
691        }
692    }
693
694    fn profile(&self) -> Option<Profile> {
695        Some(self.profile.clone())
696    }
697}
