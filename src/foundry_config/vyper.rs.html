1//! Vyper specific configuration types.
2
3use foundry_compilers::artifacts::vyper::VyperOptimizationMode;
4use serde::{Deserialize, Serialize};
5use std::path::PathBuf;
6
7#[derive(Default, Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
8pub struct VyperConfig {
9    /// Vyper optimization mode. "gas", "none" or "codesize"
10    #[serde(default, skip_serializing_if = "Option::is_none")]
11    pub optimize: Option<VyperOptimizationMode>,
12    /// The Vyper instance to use if any.
13    #[serde(default, skip_serializing_if = "Option::is_none")]
14    pub path: Option<PathBuf>,
15    /// Optionally enables experimental Venom pipeline
16    #[serde(default, skip_serializing_if = "Option::is_none")]
17    pub experimental_codegen: Option<bool>,
18}
