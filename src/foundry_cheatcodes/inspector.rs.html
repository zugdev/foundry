79/// Helper trait for obtaining complete [revm::Inspector] instance from mutable reference to
80/// [Cheatcodes].
81///
82/// This is needed for cases when inspector itself needs mutable access to [Cheatcodes] state and
83/// allows us to correctly execute arbitrary EVM frames from inside cheatcode implementations.
84pub trait CheatcodesExecutor {
85    /// Core trait method accepting mutable reference to [Cheatcodes] and returning
86    /// [revm::Inspector].
87    fn get_inspector<'a>(&'a mut self, cheats: &'a mut Cheatcodes) -> Box<dyn InspectorExt + 'a>;
88
89    /// Obtains [FoundryEvm] instance and executes the given CREATE frame.
90    fn exec_create(
91        &mut self,
92        inputs: CreateInputs,
93        ccx: &mut CheatsCtxt,
94    ) -> Result<CreateOutcome, EVMError<DatabaseError>> {
95        with_evm(self, ccx, |evm| {
96            evm.inner.ctx.journaled_state.depth += 1;
97
98            let frame = FrameInput::Create(Box::new(inputs));
99
100            let outcome = match evm.run_execution(frame)? {
101                FrameResult::Call(_) => unreachable!(),
102                FrameResult::Create(create) => create,
103            };
104
105            evm.inner.ctx.journaled_state.depth -= 1;
106
107            Ok(outcome)
108        })
109    }
110
111    fn console_log(&mut self, ccx: &mut CheatsCtxt, msg: &str) {
112        self.get_inspector(ccx.state).console_log(msg);
113    }
114
115    /// Returns a mutable reference to the tracing inspector if it is available.
116    fn tracing_inspector(&mut self) -> Option<&mut TracingInspector> {
117        None
118    }
119}

121/// Constructs [FoundryEvm] and runs a given closure with it.
122fn with_evm<E, F, O>(
123    executor: &mut E,
124    ccx: &mut CheatsCtxt,
125    f: F,
126) -> Result<O, EVMError<DatabaseError>>
127where
128    E: CheatcodesExecutor + ?Sized,
129    F: for<'a, 'b> FnOnce(
130        &mut FoundryEvm<'a, &'b mut dyn InspectorExt>,
131    ) -> Result<O, EVMError<DatabaseError>>,
132{
133    let mut inspector = executor.get_inspector(ccx.state);
134    let error = std::mem::replace(&mut ccx.ecx.error, Ok(()));
135
136    let ctx = EthEvmContext {
137        block: ccx.ecx.block.clone(),
138        cfg: ccx.ecx.cfg.clone(),
139        tx: ccx.ecx.tx.clone(),
140        journaled_state: Journal {
141            inner: ccx.ecx.journaled_state.inner.clone(),
142            database: &mut *ccx.ecx.journaled_state.database as &mut dyn DatabaseExt,
143        },
144        local: LocalContext::default(),
145        chain: (),
146        error,
147    };
148
149    let mut evm = new_evm_with_existing_context(ctx, &mut *inspector);
150
151    let res = f(&mut evm)?;
152
153    ccx.ecx.journaled_state.inner = evm.inner.ctx.journaled_state.inner;
154    ccx.ecx.block = evm.inner.ctx.block;
155    ccx.ecx.tx = evm.inner.ctx.tx;
156    ccx.ecx.cfg = evm.inner.ctx.cfg;
157    ccx.ecx.error = evm.inner.ctx.error;
158
159    Ok(res)
160}

278/// Holds data about arbitrary storage.
279#[derive(Clone, Debug, Default)]
280pub struct ArbitraryStorage {
281    /// Mapping of arbitrary storage addresses to generated values (slot, arbitrary value).
282    /// (SLOADs return random value if storage slot wasn't accessed).
283    /// Changed values are recorded and used to copy storage to different addresses.
284    pub values: HashMap<Address, HashMap<U256, U256>>,
285    /// Mapping of address with storage copied to arbitrary storage address source.
286    pub copies: HashMap<Address, Address>,
287    /// Address with storage slots that should be overwritten even if previously set.
288    pub overwrites: HashSet<Address>,
289}
290
291impl ArbitraryStorage {
292    /// Marks an address with arbitrary storage.
293    pub fn mark_arbitrary(&mut self, address: &Address, overwrite: bool) {
294        self.values.insert(*address, HashMap::default());
295        if overwrite {
296            self.overwrites.insert(*address);
297        } else {
298            self.overwrites.remove(address);
299        }
300    }
301
302    /// Maps an address that copies storage with the arbitrary storage address.
303    pub fn mark_copy(&mut self, from: &Address, to: &Address) {
304        if self.values.contains_key(from) {
305            self.copies.insert(*to, *from);
306        }
307    }
308
309    /// Saves arbitrary storage value for a given address:
310    /// - store value in changed values cache.
311    /// - update account's storage with given value.
312    pub fn save(&mut self, ecx: Ecx, address: Address, slot: U256, data: U256) {
313        self.values.get_mut(&address).expect("missing arbitrary address entry").insert(slot, data);
314        if let Ok(mut account) = ecx.journaled_state.load_account(address) {
315            account.storage.insert(slot, EvmStorageSlot::new(data, 0));
316        }
317    }
318
319    /// Copies arbitrary storage value from source address to the given target address:
320    /// - if a value is present in arbitrary values cache, then update target storage and return
321    ///   existing value.
322    /// - if no value was yet generated for given slot, then save new value in cache and update both
323    ///   source and target storages.
324    pub fn copy(&mut self, ecx: Ecx, target: Address, slot: U256, new_value: U256) -> U256 {
325        let source = self.copies.get(&target).expect("missing arbitrary copy target entry");
326        let storage_cache = self.values.get_mut(source).expect("missing arbitrary source storage");
327        let value = match storage_cache.get(&slot) {
328            Some(value) => *value,
329            None => {
330                storage_cache.insert(slot, new_value);
331                // Update source storage with new value.
332                if let Ok(mut source_account) = ecx.journaled_state.load_account(*source) {
333                    source_account.storage.insert(slot, EvmStorageSlot::new(new_value, 0));
334                }
335                new_value
336            }
337        };
338        // Update target storage with new value.
339        if let Ok(mut target_account) = ecx.journaled_state.load_account(target) {
340            target_account.storage.insert(slot, EvmStorageSlot::new(value, 0));
341        }
342        value
343    }
344}

349/// An EVM inspector that handles calls to various cheatcodes, each with their own behavior.
350///
351/// Cheatcodes can be called by contracts during execution to modify the VM environment, such as
352/// mocking addresses, signatures and altering call reverts.
353///
354/// Executing cheatcodes can be very powerful. Most cheatcodes are limited to evm internals, but
355/// there are also cheatcodes like `ffi` which can execute arbitrary commands or `writeFile` and
356/// `readFile` which can manipulate files of the filesystem. Therefore, several restrictions are
357/// implemented for these cheatcodes:
358/// - `ffi`, and file cheatcodes are _always_ opt-in (via foundry config) and never enabled by
359///   default: all respective cheatcode handlers implement the appropriate checks
360/// - File cheatcodes require explicit permissions which paths are allowed for which operation, see
361///   `Config.fs_permission`
362/// - Only permitted accounts are allowed to execute cheatcodes in forking mode, this ensures no
363///   contract deployed on the live network is able to execute cheatcodes by simply calling the

580    /// Decodes the input data and applies the cheatcode.
581    fn apply_cheatcode(
582        &mut self,
583        ecx: Ecx,
584        call: &CallInputs,
585        executor: &mut dyn CheatcodesExecutor,
586    ) -> Result {
587        // decode the cheatcode call
588        let decoded = Vm::VmCalls::abi_decode(&call.input.bytes(ecx)).map_err(|e| {
589            if let alloy_sol_types::Error::UnknownSelector { name: _, selector } = e {
590                let msg = format!(
591                    "unknown cheatcode with selector {selector}; \
592                     you may have a mismatch between the `Vm` interface (likely in `forge-std`) \
593                     and the `forge` version"
594                );
595                return alloy_sol_types::Error::Other(std::borrow::Cow::Owned(msg));
596            }
597            e
598        })?;
599
600        let caller = call.caller;
601
602        // ensure the caller is allowed to execute cheatcodes,
603        // but only if the backend is in forking mode
604        ecx.journaled_state.database.ensure_cheatcode_access_forking_mode(&caller)?;
605
606        apply_dispatch(
607            &decoded,
608            &mut CheatsCtxt { state: self, ecx, gas_limit: call.gas_limit, caller },
609            executor,
610        )
611    }

641    /// Called when there was a revert.
642    ///
643    /// Cleanup any previously applied cheatcodes that altered the state in such a way that revm's
644    /// revert would run into issues.
645    pub fn on_revert(&mut self, ecx: Ecx) {
646        trace!(deals=?self.eth_deals.len(), "rolling back deals");
647
648        // Delay revert clean up until expected revert is handled, if set.
649        if self.expected_revert.is_some() {
650            return;
651        }
652
653        // we only want to apply cleanup top level
654        if ecx.journaled_state.depth() > 0 {
655            return;
656        }
657
658        // Roll back all previously applied deals
659        // This will prevent overflow issues in revm's [`JournaledState::journal_revert`] routine
660        // which rolls back any transfers.
661        while let Some(record) = self.eth_deals.pop() {
662            if let Some(acc) = ecx.journaled_state.state.get_mut(&record.address) {
663                acc.info.balance = record.old_balance;
664            }
665        }
666    }

668    pub fn call_with_executor(
669        &mut self,
670        ecx: Ecx,
671        call: &mut CallInputs,
672        executor: &mut dyn CheatcodesExecutor,
673    ) -> Option<CallOutcome> {
674        let gas = Gas::new(call.gas_limit);
675        let curr_depth = ecx.journaled_state.depth();
676
677        // At the root call to test function or script `run()`/`setUp()` functions, we are
678        // decreasing sender nonce to ensure that it matches on-chain nonce once we start
679        // broadcasting.
680        if curr_depth == 0 {
681            let sender = ecx.tx.caller;
682            let account = match super::evm::journaled_account(ecx, sender) {
683                Ok(account) => account,
684                Err(err) => {
685                    return Some(CallOutcome {
686                        result: InterpreterResult {
687                            result: InstructionResult::Revert,
688                            output: err.abi_encode().into(),
689                            gas,
690                        },
691                        memory_offset: call.return_memory_offset.clone(),
692                    });
693                }
694            };
695            let prev = account.info.nonce;
696            account.info.nonce = prev.saturating_sub(1);
697
698            trace!(target: "cheatcodes", %sender, nonce=account.info.nonce, prev, "corrected nonce");
699        }
700
701        if call.target_address == CHEATCODE_ADDRESS {
702            return match self.apply_cheatcode(ecx, call, executor) {
703                Ok(retdata) => Some(CallOutcome {
704                    result: InterpreterResult {
705                        result: InstructionResult::Return,
706                        output: retdata.into(),
707                        gas,
708                    },
709                    memory_offset: call.return_memory_offset.clone(),
710                }),
711                Err(err) => Some(CallOutcome {
712                    result: InterpreterResult {
713                        result: InstructionResult::Revert,
714                        output: err.abi_encode().into(),
715                        gas,
716                    },
717                    memory_offset: call.return_memory_offset.clone(),
718                }),
719            };
720        }
721
722        if call.target_address == HARDHAT_CONSOLE_ADDRESS {
723            return None;
724        }
725
726        // Handle expected calls
727
728        // Grab the different calldatas expected.
729        if let Some(expected_calls_for_target) = self.expected_calls.get_mut(&call.bytecode_address)
730        {
731            // Match every partial/full calldata
732            for (calldata, (expected, actual_count)) in expected_calls_for_target {
733                // Increment actual times seen if...
734                // The calldata is at most, as big as this call's input, and
735                if calldata.len() <= call.input.len() &&
736                    // Both calldata match, taking the length of the assumed smaller one (which will have at least the selector), and
737                    *calldata == call.input.bytes(ecx)[..calldata.len()] &&
738                    // The value matches, if provided
739                    expected
740                        .value.is_none_or(|value| Some(value) == call.transfer_value()) &&
741                    // The gas matches, if provided
742                    expected.gas.is_none_or(|gas| gas == call.gas_limit) &&
743                    // The minimum gas matches, if provided
744                    expected.min_gas.is_none_or(|min_gas| min_gas <= call.gas_limit)
745                {
746                    *actual_count += 1;
747                }
748            }
749        }
750
751        // Handle mocked calls
752        if let Some(mocks) = self.mocked_calls.get_mut(&call.bytecode_address) {
753            let ctx = MockCallDataContext {
754                calldata: call.input.bytes(ecx),
755                value: call.transfer_value(),
756            };
757
758            if let Some(return_data_queue) = match mocks.get_mut(&ctx) {
759                Some(queue) => Some(queue),
760                None => mocks
761                    .iter_mut()
762                    .find(|(mock, _)| {
763                        call.input.bytes(ecx).get(..mock.calldata.len()) == Some(&mock.calldata[..])
764                            && mock.value.is_none_or(|value| Some(value) == call.transfer_value())
765                    })
766                    .map(|(_, v)| v),
767            } && let Some(return_data) = if return_data_queue.len() == 1 {
768                // If the mocked calls stack has a single element in it, don't empty it
769                return_data_queue.front().map(|x| x.to_owned())
770            } else {
771                // Else, we pop the front element
772                return_data_queue.pop_front()
773            } {
774                return Some(CallOutcome {
775                    result: InterpreterResult {
776                        result: return_data.ret_type,
777                        output: return_data.data,
778                        gas,
779                    },
780                    memory_offset: call.return_memory_offset.clone(),
781                });
782            }
783        }
784
785        // Apply our prank
786        if let Some(prank) = &self.get_prank(curr_depth) {
787            // Apply delegate call, `call.caller`` will not equal `prank.prank_caller`
788            if prank.delegate_call
789                && curr_depth == prank.depth
790                && let CallScheme::DelegateCall = call.scheme
791            {
792                call.target_address = prank.new_caller;
793                call.caller = prank.new_caller;
794                if let Some(new_origin) = prank.new_origin {
795                    ecx.tx.caller = new_origin;
796                }
797            }
798
799            if curr_depth >= prank.depth && call.caller == prank.prank_caller {
800                let mut prank_applied = false;
801
802                // At the target depth we set `msg.sender`
803                if curr_depth == prank.depth {
804                    call.caller = prank.new_caller;
805                    prank_applied = true;
806                }
807
808                // At the target depth, or deeper, we set `tx.origin`
809                if let Some(new_origin) = prank.new_origin {
810                    ecx.tx.caller = new_origin;
811                    prank_applied = true;
812                }
813
814                // If prank applied for first time, then update
815                if prank_applied && let Some(applied_prank) = prank.first_time_applied() {
816                    self.pranks.insert(curr_depth, applied_prank);
817                }
818            }
819        }
820
821        // Apply EIP-2930 access list
822        self.apply_accesslist(ecx);
823
824        // Apply our broadcast
825        if let Some(broadcast) = &self.broadcast {
826            // We only apply a broadcast *to a specific depth*.
827            //
828            // We do this because any subsequent contract calls *must* exist on chain and
829            // we only want to grab *this* call, not internal ones
830            if curr_depth == broadcast.depth && call.caller == broadcast.original_caller {
831                // At the target depth we set `msg.sender` & tx.origin.
832                // We are simulating the caller as being an EOA, so *both* must be set to the
833                // broadcast.origin.
834                ecx.tx.caller = broadcast.new_origin;
835
836                call.caller = broadcast.new_origin;
837                // Add a `legacy` transaction to the VecDeque. We use a legacy transaction here
838                // because we only need the from, to, value, and data. We can later change this
839                // into 1559, in the cli package, relatively easily once we
840                // know the target chain supports EIP-1559.
841                if !call.is_static {
842                    if let Err(err) = ecx.journaled_state.load_account(broadcast.new_origin) {
843                        return Some(CallOutcome {
844                            result: InterpreterResult {
845                                result: InstructionResult::Revert,
846                                output: Error::encode(err),
847                                gas,
848                            },
849                            memory_offset: call.return_memory_offset.clone(),
850                        });
851                    }
852
853                    let is_fixed_gas_limit = check_if_fixed_gas_limit(&ecx, call.gas_limit);
854
855                    let input = TransactionInput::new(call.input.bytes(ecx));
856
857                    let account =
858                        ecx.journaled_state.inner.state().get_mut(&broadcast.new_origin).unwrap();
859
860                    let mut tx_req = TransactionRequest {
861                        from: Some(broadcast.new_origin),
862                        to: Some(TxKind::from(Some(call.target_address))),
863                        value: call.transfer_value(),
864                        input,
865                        nonce: Some(account.info.nonce),
866                        chain_id: Some(ecx.cfg.chain_id),
867                        gas: if is_fixed_gas_limit { Some(call.gas_limit) } else { None },
868                        ..Default::default()
869                    };
870
871                    let active_delegations = std::mem::take(&mut self.active_delegations);
872                    // Set active blob sidecar, if any.
873                    if let Some(blob_sidecar) = self.active_blob_sidecar.take() {
874                        // Ensure blob and delegation are not set for the same tx.
875                        if !active_delegations.is_empty() {
876                            let msg = "both delegation and blob are active; `attachBlob` and `attachDelegation` are not compatible";
877                            return Some(CallOutcome {
878                                result: InterpreterResult {
879                                    result: InstructionResult::Revert,
880                                    output: Error::encode(msg),
881                                    gas,
882                                },
883                                memory_offset: call.return_memory_offset.clone(),
884                            });
885                        }
886                        tx_req.set_blob_sidecar(blob_sidecar);
887                    }
888
889                    // Apply active EIP-7702 delegations, if any.
890                    if !active_delegations.is_empty() {
891                        for auth in &active_delegations {
892                            let Ok(authority) = auth.recover_authority() else {
893                                continue;
894                            };
895                            if authority == broadcast.new_origin {
896                                // Increment nonce of broadcasting account to reflect signed
897                                // authorization.
898                                account.info.nonce += 1;
899                            }
900                        }
901                        tx_req.authorization_list = Some(active_delegations);
902                    }
903
904                    self.broadcastable_transactions.push_back(BroadcastableTransaction {
905                        rpc: ecx.journaled_state.database.active_fork_url(),
906                        transaction: tx_req.into(),
907                    });
908                    debug!(target: "cheatcodes", tx=?self.broadcastable_transactions.back().unwrap(), "broadcastable call");
909
910                    // Explicitly increment nonce if calls are not isolated.
911                    if !self.config.evm_opts.isolate {
912                        let prev = account.info.nonce;
913                        account.info.nonce += 1;
914                        debug!(target: "cheatcodes", address=%broadcast.new_origin, nonce=prev+1, prev, "incremented nonce");
915                    }
916                } else if broadcast.single_call {
917                    let msg = "`staticcall`s are not allowed after `broadcast`; use `startBroadcast` instead";
918                    return Some(CallOutcome {
919                        result: InterpreterResult {
920                            result: InstructionResult::Revert,
921                            output: Error::encode(msg),
922                            gas,
923                        },
924                        memory_offset: call.return_memory_offset.clone(),
925                    });
926                }
927            }
928        }
929
930        // Record called accounts if `startStateDiffRecording` has been called
931        if let Some(recorded_account_diffs_stack) = &mut self.recorded_account_diffs_stack {
932            // Determine if account is "initialized," ie, it has a non-zero balance, a non-zero
933            // nonce, a non-zero KECCAK_EMPTY codehash, or non-empty code
934            let initialized;
935            let old_balance;
936            let old_nonce;
937            if let Ok(acc) = ecx.journaled_state.load_account(call.target_address) {
938                initialized = acc.info.exists();
939                old_balance = acc.info.balance;
940                old_nonce = acc.info.nonce;
941            } else {
942                initialized = false;
943                old_balance = U256::ZERO;
944                old_nonce = 0;
945            }
946            let kind = match call.scheme {
947                CallScheme::Call => crate::Vm::AccountAccessKind::Call,
948                CallScheme::CallCode => crate::Vm::AccountAccessKind::CallCode,
949                CallScheme::DelegateCall => crate::Vm::AccountAccessKind::DelegateCall,
950                CallScheme::StaticCall => crate::Vm::AccountAccessKind::StaticCall,
951            };
952            // Record this call by pushing it to a new pending vector; all subsequent calls at
953            // that depth will be pushed to the same vector. When the call ends, the
954            // RecordedAccountAccess (and all subsequent RecordedAccountAccesses) will be
955            // updated with the revert status of this call, since the EVM does not mark accounts
956            // as "warm" if the call from which they were accessed is reverted
957            recorded_account_diffs_stack.push(vec![AccountAccess {
958                chainInfo: crate::Vm::ChainInfo {
959                    forkId: ecx.journaled_state.db().active_fork_id().unwrap_or_default(),
960                    chainId: U256::from(ecx.cfg.chain_id),
961                },
962                accessor: call.caller,
963                account: call.bytecode_address,
964                kind,
965                initialized,
966                oldBalance: old_balance,
967                newBalance: U256::ZERO, // updated on call_end
968                oldNonce: old_nonce,
969                newNonce: 0, // updated on call_end
970                value: call.call_value(),
971                data: call.input.bytes(ecx),
972                reverted: false,
973                deployedCode: Bytes::new(),
974                storageAccesses: vec![], // updated on step
975                depth: ecx
976                    .journaled_state
977                    .depth()
978                    .try_into()
979                    .expect("journaled state depth exceeds u64"),
980            }]);
981        }
982
983        None
984    }

1056        if let Some(block) = self.block.take() {

1061        }
1062
1063        // Record gas for current frame.
1064        if self.gas_metering.paused {
1065            self.gas_metering.paused_frames.push(interpreter.gas);
1066        }
1067
1068        // `expectRevert`: track the max call depth during `expectRevert`

1070            expected.max_depth = max(ecx.journaled_state.depth(), expected.max_depth);

1074    fn step(&mut self, interpreter: &mut Interpreter, ecx: Ecx) {
1075        self.pc = interpreter.bytecode.pc();
1076
1077        // `pauseGasMetering`: pause / resume interpreter gas.
1078        if self.gas_metering.paused {
1079            self.meter_gas(interpreter);
1080        }
1081
1082        // `resetGasMetering`: reset interpreter gas.
1083        if self.gas_metering.reset {
1084            self.meter_gas_reset(interpreter);
1085        }
1086
1087        // `record`: record storage reads and writes.
1088        if self.recording_accesses {
1089            self.record_accesses(interpreter);
1090        }
1091
1092        // `startStateDiffRecording`: record granular ordered storage accesses.
1093        if self.recorded_account_diffs_stack.is_some() {
1094            self.record_state_diffs(interpreter, ecx);
1095        }
1096
1097        // `expectSafeMemory`: check if the current opcode is allowed to interact with memory.
1098        if !self.allowed_mem_writes.is_empty() {
1099            self.check_mem_opcodes(
1100                interpreter,
1101                ecx.journaled_state.depth().try_into().expect("journaled state depth exceeds u64"),
1102            );
1103        }
1104
1105        // `startMappingRecording`: record SSTORE and KECCAK256.
1106        if let Some(mapping_slots) = &mut self.mapping_slots {
1107            mapping::step(mapping_slots, interpreter);
1108        }
1109
1110        // `snapshotGas*`: take a snapshot of the current gas.
1111        if self.gas_metering.recording {
1112            self.meter_gas_record(interpreter, ecx);
1113        }
1114    }

1150    fn call_end(&mut self, ecx: Ecx, call: &CallInputs, outcome: &mut CallOutcome) {
1151        let cheatcode_call = call.target_address == CHEATCODE_ADDRESS
1152            || call.target_address == HARDHAT_CONSOLE_ADDRESS;
1153
1154        // Clean up pranks/broadcasts if it's not a cheatcode call end. We shouldn't do
1155        // it for cheatcode calls because they are not applied for cheatcodes in the `call` hook.
1156        // This should be placed before the revert handling, because we might exit early there
1157        if !cheatcode_call {
1158            // Clean up pranks
1159            let curr_depth = ecx.journaled_state.depth();
1160            if let Some(prank) = &self.get_prank(curr_depth)
1161                && curr_depth == prank.depth
1162            {
1163                ecx.tx.caller = prank.prank_origin;
1164
1165                // Clean single-call prank once we have returned to the original depth
1166                if prank.single_call {
1167                    self.pranks.remove(&curr_depth);
1168                }
1169            }
1170
1171            // Clean up broadcast
1172            if let Some(broadcast) = &self.broadcast
1173                && curr_depth == broadcast.depth
1174            {
1175                ecx.tx.caller = broadcast.original_origin;
1176
1177                // Clean single-call broadcast once we have returned to the original depth
1178                if broadcast.single_call {
1179                    let _ = self.broadcast.take();
1180                }
1181            }
1182        }
1183
1184        // Handle assume no revert cheatcode.
1185        if let Some(assume_no_revert) = &mut self.assume_no_revert {
1186            // Record current reverter address before processing the expect revert if call reverted,
1187            // expect revert is set with expected reverter address and no actual reverter set yet.
1188            if outcome.result.is_revert() && assume_no_revert.reverted_by.is_none() {
1189                assume_no_revert.reverted_by = Some(call.target_address);
1190            }
1191
1192            // allow multiple cheatcode calls at the same depth
1193            let curr_depth = ecx.journaled_state.depth();
1194            if curr_depth <= assume_no_revert.depth && !cheatcode_call {
1195                // Discard run if we're at the same depth as cheatcode, call reverted, and no
1196                // specific reason was supplied
1197                if outcome.result.is_revert() {
1198                    let assume_no_revert = std::mem::take(&mut self.assume_no_revert).unwrap();
1199                    return match revert_handlers::handle_assume_no_revert(
1200                        &assume_no_revert,
1201                        outcome.result.result,
1202                        &outcome.result.output,
1203                        &self.config.available_artifacts,
1204                    ) {
1205                        // if result is Ok, it was an anticipated revert; return an "assume" error
1206                        // to reject this run
1207                        Ok(_) => {
1208                            outcome.result.output = Error::from(MAGIC_ASSUME).abi_encode().into();
1209                        }
1210                        // if result is Error, it was an unanticipated revert; should revert
1211                        // normally
1212                        Err(error) => {
1213                            trace!(expected=?assume_no_revert, ?error, status=?outcome.result.result, "Expected revert mismatch");
1214                            outcome.result.result = InstructionResult::Revert;
1215                            outcome.result.output = error.abi_encode().into();
1216                        }
1217                    };
1218                } else {
1219                    // Call didn't revert, reset `assume_no_revert` state.
1220                    self.assume_no_revert = None;
1221                }
1222            }
1223        }
1224
1225        // Handle expected reverts.
1226        if let Some(expected_revert) = &mut self.expected_revert {
1227            // Record current reverter address and call scheme before processing the expect revert
1228            // if call reverted.
1229            if outcome.result.is_revert() {
1230                // Record current reverter address if expect revert is set with expected reverter
1231                // address and no actual reverter was set yet or if we're expecting more than one
1232                // revert.
1233                if expected_revert.reverter.is_some()
1234                    && (expected_revert.reverted_by.is_none() || expected_revert.count > 1)
1235                {
1236                    expected_revert.reverted_by = Some(call.target_address);
1237                }
1238            }
1239
1240            let curr_depth = ecx.journaled_state.depth();
1241            if curr_depth <= expected_revert.depth {
1242                let needs_processing = match expected_revert.kind {
1243                    ExpectedRevertKind::Default => !cheatcode_call,
1244                    // `pending_processing` == true means that we're in the `call_end` hook for
1245                    // `vm.expectCheatcodeRevert` and shouldn't expect revert here
1246                    ExpectedRevertKind::Cheatcode { pending_processing } => {
1247                        cheatcode_call && !pending_processing
1248                    }
1249                };
1250
1251                if needs_processing {
1252                    let mut expected_revert = std::mem::take(&mut self.expected_revert).unwrap();
1253                    return match revert_handlers::handle_expect_revert(
1254                        cheatcode_call,
1255                        false,
1256                        self.config.internal_expect_revert,
1257                        &expected_revert,
1258                        outcome.result.result,
1259                        outcome.result.output.clone(),
1260                        &self.config.available_artifacts,
1261                    ) {
1262                        Err(error) => {
1263                            trace!(expected=?expected_revert, ?error, status=?outcome.result.result, "Expected revert mismatch");
1264                            outcome.result.result = InstructionResult::Revert;
1265                            outcome.result.output = error.abi_encode().into();
1266                        }
1267                        Ok((_, retdata)) => {
1268                            expected_revert.actual_count += 1;
1269                            if expected_revert.actual_count < expected_revert.count {
1270                                self.expected_revert = Some(expected_revert.clone());
1271                            }
1272                            outcome.result.result = InstructionResult::Return;
1273                            outcome.result.output = retdata;
1274                        }
1275                    };
1276                }
1277
1278                // Flip `pending_processing` flag for cheatcode revert expectations, marking that
1279                // we've exited the `expectCheatcodeRevert` call scope
1280                if let ExpectedRevertKind::Cheatcode { pending_processing } =
1281                    &mut self.expected_revert.as_mut().unwrap().kind
1282                {
1283                    *pending_processing = false;
1284                }
1285            }
1286        }
1287
1288        // Exit early for calls to cheatcodes as other logic is not relevant for cheatcode
1289        // invocations
1290        if cheatcode_call {
1291            return;
1292        }
1293
1294        // Record the gas usage of the call, this allows the `lastCallGas` cheatcode to
1295        // retrieve the gas usage of the last call.
1296        let gas = outcome.result.gas;
1297        self.gas_metering.last_call_gas = Some(crate::Vm::Gas {
1298            gasLimit: gas.limit(),
1299            gasTotalUsed: gas.spent(),
1300            gasMemoryUsed: 0,
1301            gasRefunded: gas.refunded(),
1302            gasRemaining: gas.remaining(),
1303        });
1304
1305        // If `startStateDiffRecording` has been called, update the `reverted` status of the
1306        // previous call depth's recorded accesses, if any
1307        if let Some(recorded_account_diffs_stack) = &mut self.recorded_account_diffs_stack {
1308            // The root call cannot be recorded.
1309            if ecx.journaled_state.depth() > 0
1310                && let Some(last_recorded_depth) = &mut recorded_account_diffs_stack.pop()
1311            {
1312                // Update the reverted status of all deeper calls if this call reverted, in
1313                // accordance with EVM behavior
1314                if outcome.result.is_revert() {
1315                    last_recorded_depth.iter_mut().for_each(|element| {
1316                        element.reverted = true;
1317                        element
1318                            .storageAccesses
1319                            .iter_mut()
1320                            .for_each(|storage_access| storage_access.reverted = true);
1321                    })
1322                }
1323
1324                if let Some(call_access) = last_recorded_depth.first_mut() {
1325                    // Assert that we're at the correct depth before recording post-call state
1326                    // changes. Depending on the depth the cheat was
1327                    // called at, there may not be any pending
1328                    // calls to update if execution has percolated up to a higher depth.
1329                    let curr_depth = ecx.journaled_state.depth();
1330                    if call_access.depth == curr_depth as u64
1331                        && let Ok(acc) = ecx.journaled_state.load_account(call.target_address)
1332                    {
1333                        debug_assert!(access_is_call(call_access.kind));
1334                        call_access.newBalance = acc.info.balance;
1335                        call_access.newNonce = acc.info.nonce;
1336                    }
1337                    // Merge the last depth's AccountAccesses into the AccountAccesses at the
1338                    // current depth, or push them back onto the pending
1339                    // vector if higher depths were not recorded. This
1340                    // preserves ordering of accesses.
1341                    if let Some(last) = recorded_account_diffs_stack.last_mut() {
1342                        last.append(last_recorded_depth);
1343                    } else {
1344                        recorded_account_diffs_stack.push(last_recorded_depth.clone());
1345                    }
1346                }
1347            }
1348        }
1349
1350        // At the end of the call,
1351        // we need to check if we've found all the emits.
1352        // We know we've found all the expected emits in the right order
1353        // if the queue is fully matched.
1354        // If it's not fully matched, then either:
1355        // 1. Not enough events were emitted (we'll know this because the amount of times we
1356        // inspected events will be less than the size of the queue) 2. The wrong events
1357        // were emitted (The inspected events should match the size of the queue, but still some
1358        // events will not be matched)
1359
1360        // First, check that we're at the call depth where the emits were declared from.
1361        let should_check_emits = self
1362            .expected_emits
1363            .iter()
1364            .any(|(expected, _)| {
1365                let curr_depth = ecx.journaled_state.depth();
1366                expected.depth == curr_depth
1367            }) &&
1368            // Ignore staticcalls
1369            !call.is_static;
1370        if should_check_emits {
1371            let expected_counts = self
1372                .expected_emits
1373                .iter()
1374                .filter_map(|(expected, count_map)| {
1375                    let count = match expected.address {
1376                        Some(emitter) => match count_map.get(&emitter) {
1377                            Some(log_count) => expected
1378                                .log
1379                                .as_ref()
1380                                .map(|l| log_count.count(l))
1381                                .unwrap_or_else(|| log_count.count_unchecked()),
1382                            None => 0,
1383                        },
1384                        None => match &expected.log {
1385                            Some(log) => count_map.values().map(|logs| logs.count(log)).sum(),
1386                            None => count_map.values().map(|logs| logs.count_unchecked()).sum(),
1387                        },
1388                    };
1389
1390                    if count != expected.count { Some((expected, count)) } else { None }
1391                })
1392                .collect::<Vec<_>>();
1393
1394            // Revert if not all emits expected were matched.
1395            if let Some((expected, _)) = self
1396                .expected_emits
1397                .iter()
1398                .find(|(expected, _)| !expected.found && expected.count > 0)
1399            {
1400                outcome.result.result = InstructionResult::Revert;
1401                let error_msg = expected.mismatch_error.as_deref().unwrap_or("log != expected log");
1402                outcome.result.output = error_msg.abi_encode().into();
1403                return;
1404            }
1405
1406            if !expected_counts.is_empty() {
1407                let msg = if outcome.result.is_ok() {
1408                    let (expected, count) = expected_counts.first().unwrap();
1409                    format!("log emitted {count} times, expected {}", expected.count)
1410                } else {
1411                    "expected an emit, but the call reverted instead. \
1412                     ensure you're testing the happy path when using `expectEmit`"
1413                        .to_string()
1414                };
1415
1416                outcome.result.result = InstructionResult::Revert;
1417                outcome.result.output = Error::encode(msg);
1418                return;
1419            }
1420
1421            // All emits were found, we're good.
1422            // Clear the queue, as we expect the user to declare more events for the next call
1423            // if they wanna match further events.
1424            self.expected_emits.clear()
1425        }
1426
1427        // this will ensure we don't have false positives when trying to diagnose reverts in fork
1428        // mode
1429        let diag = self.fork_revert_diagnostic.take();
1430
1431        // if there's a revert and a previous call was diagnosed as fork related revert then we can
1432        // return a better error here
1433        if outcome.result.is_revert()
1434            && let Some(err) = diag
1435        {
1436            outcome.result.output = Error::encode(err.to_error_msg(&self.labels));
1437            return;
1438        }
1439
1440        // try to diagnose reverts in multi-fork mode where a call is made to an address that does
1441        // not exist
1442        if let TxKind::Call(test_contract) = ecx.tx.kind {
1443            // if a call to a different contract than the original test contract returned with
1444            // `Stop` we check if the contract actually exists on the active fork
1445            if ecx.journaled_state.db().is_forked_mode()
1446                && outcome.result.result == InstructionResult::Stop
1447                && call.target_address != test_contract
1448            {
1449                let journaled_state = ecx.journaled_state.clone();
1450                self.fork_revert_diagnostic =
1451                    ecx.journaled_state.db().diagnose_revert(call.target_address, &journaled_state);
1452            }
1453        }
1454
1455        // If the depth is 0, then this is the root call terminating
1456        if ecx.journaled_state.depth() == 0 {
1457            // If we already have a revert, we shouldn't run the below logic as it can obfuscate an
1458            // earlier error that happened first with unrelated information about
1459            // another error when using cheatcodes.
1460            if outcome.result.is_revert() {
1461                return;
1462            }
1463
1464            // If there's not a revert, we can continue on to run the last logic for expect*
1465            // cheatcodes.
1466
1467            // Match expected calls
1468            for (address, calldatas) in &self.expected_calls {
1469                // Loop over each address, and for each address, loop over each calldata it expects.
1470                for (calldata, (expected, actual_count)) in calldatas {
1471                    // Grab the values we expect to see
1472                    let ExpectedCallData { gas, min_gas, value, count, call_type } = expected;
1473
1474                    let failed = match call_type {
1475                        // If the cheatcode was called with a `count` argument,
1476                        // we must check that the EVM performed a CALL with this calldata exactly
1477                        // `count` times.
1478                        ExpectedCallType::Count => *count != *actual_count,
1479                        // If the cheatcode was called without a `count` argument,
1480                        // we must check that the EVM performed a CALL with this calldata at least
1481                        // `count` times. The amount of times to check was
1482                        // the amount of time the cheatcode was called.
1483                        ExpectedCallType::NonCount => *count > *actual_count,
1484                    };
1485                    if failed {
1486                        let expected_values = [
1487                            Some(format!("data {}", hex::encode_prefixed(calldata))),
1488                            value.as_ref().map(|v| format!("value {v}")),
1489                            gas.map(|g| format!("gas {g}")),
1490                            min_gas.map(|g| format!("minimum gas {g}")),
1491                        ]
1492                        .into_iter()
1493                        .flatten()
1494                        .join(", ");
1495                        let but = if outcome.result.is_ok() {
1496                            let s = if *actual_count == 1 { "" } else { "s" };
1497                            format!("was called {actual_count} time{s}")
1498                        } else {
1499                            "the call reverted instead; \
1500                             ensure you're testing the happy path when using `expectCall`"
1501                                .to_string()
1502                        };
1503                        let s = if *count == 1 { "" } else { "s" };
1504                        let msg = format!(
1505                            "expected call to {address} with {expected_values} \
1506                             to be called {count} time{s}, but {but}"
1507                        );
1508                        outcome.result.result = InstructionResult::Revert;
1509                        outcome.result.output = Error::encode(msg);
1510
1511                        return;
1512                    }
1513                }
1514            }
1515
1516            // Check if we have any leftover expected emits
1517            // First, if any emits were found at the root call, then we its ok and we remove them.
1518            self.expected_emits.retain(|(expected, _)| expected.count > 0 && !expected.found);
1519            // If not empty, we got mismatched emits
1520            if !self.expected_emits.is_empty() {
1521                let msg = if outcome.result.is_ok() {
1522                    "expected an emit, but no logs were emitted afterwards. \
1523                     you might have mismatched events or not enough events were emitted"
1524                } else {
1525                    "expected an emit, but the call reverted instead. \
1526                     ensure you're testing the happy path when using `expectEmit`"
1527                };
1528                outcome.result.result = InstructionResult::Revert;
1529                outcome.result.output = Error::encode(msg);
1530                return;
1531            }
1532
1533            // Check for leftover expected creates
1534            if let Some(expected_create) = self.expected_creates.first() {
1535                let msg = format!(
1536                    "expected {} call by address {} for bytecode {} but not found",
1537                    expected_create.create_scheme,
1538                    hex::encode_prefixed(expected_create.deployer),
1539                    hex::encode_prefixed(&expected_create.bytecode),
1540                );
1541                outcome.result.result = InstructionResult::Revert;
1542                outcome.result.output = Error::encode(msg);
1543            }
1544        }
1545    }

1547    fn create(&mut self, ecx: Ecx, mut input: &mut CreateInputs) -> Option<CreateOutcome> {
1548        let gas = Gas::new(input.gas_limit());
1549        // Check if we should intercept this create
1550        if self.intercept_next_create_call {
1551            // Reset the flag
1552            self.intercept_next_create_call = false;
1553
1554            // Get initcode from the input
1555            let output = input.init_code();
1556
1557            // Return a revert with the initcode as error data
1558            return Some(CreateOutcome {
1559                result: InterpreterResult { result: InstructionResult::Revert, output, gas },
1560                address: None,
1561            });
1562        }
1563
1564        let curr_depth = ecx.journaled_state.depth();
1565
1566        // Apply our prank
1567        if let Some(prank) = &self.get_prank(curr_depth)
1568            && curr_depth >= prank.depth
1569            && input.caller() == prank.prank_caller
1570        {
1571            let mut prank_applied = false;
1572
1573            // At the target depth we set `msg.sender`
1574            if curr_depth == prank.depth {
1575                input.set_caller(prank.new_caller);
1576                prank_applied = true;
1577            }
1578
1579            // At the target depth, or deeper, we set `tx.origin`
1580            if let Some(new_origin) = prank.new_origin {
1581                ecx.tx.caller = new_origin;
1582                prank_applied = true;
1583            }
1584
1585            // If prank applied for first time, then update
1586            if prank_applied && let Some(applied_prank) = prank.first_time_applied() {
1587                self.pranks.insert(curr_depth, applied_prank);
1588            }
1589        }
1590
1591        // Apply EIP-2930 access list
1592        self.apply_accesslist(ecx);
1593
1594        // Apply our broadcast
1595        if let Some(broadcast) = &self.broadcast
1596            && curr_depth >= broadcast.depth
1597            && input.caller() == broadcast.original_caller
1598        {
1599            if let Err(err) = ecx.journaled_state.load_account(broadcast.new_origin) {
1600                return Some(CreateOutcome {
1601                    result: InterpreterResult {
1602                        result: InstructionResult::Revert,
1603                        output: Error::encode(err),
1604                        gas,
1605                    },
1606                    address: None,
1607                });
1608            }
1609
1610            ecx.tx.caller = broadcast.new_origin;
1611
1612            if curr_depth == broadcast.depth {
1613                input.set_caller(broadcast.new_origin);
1614                let is_fixed_gas_limit = check_if_fixed_gas_limit(&ecx, input.gas_limit());
1615
1616                let account = &ecx.journaled_state.inner.state()[&broadcast.new_origin];
1617                self.broadcastable_transactions.push_back(BroadcastableTransaction {
1618                    rpc: ecx.journaled_state.database.active_fork_url(),
1619                    transaction: TransactionRequest {
1620                        from: Some(broadcast.new_origin),
1621                        to: None,
1622                        value: Some(input.value()),
1623                        input: TransactionInput::new(input.init_code()),
1624                        nonce: Some(account.info.nonce),
1625                        gas: if is_fixed_gas_limit { Some(input.gas_limit()) } else { None },
1626                        ..Default::default()
1627                    }
1628                    .into(),
1629                });
1630
1631                input.log_debug(self, &input.scheme().unwrap_or(CreateScheme::Create));
1632            }
1633        }
1634
1635        // Allow cheatcodes from the address of the new contract
1636        let address = input.allow_cheatcodes(self, ecx);
1637
1638        // If `recordAccountAccesses` has been called, record the create
1639        if let Some(recorded_account_diffs_stack) = &mut self.recorded_account_diffs_stack {
1640            recorded_account_diffs_stack.push(vec![AccountAccess {
1641                chainInfo: crate::Vm::ChainInfo {
1642                    forkId: ecx.journaled_state.db().active_fork_id().unwrap_or_default(),
1643                    chainId: U256::from(ecx.cfg.chain_id),
1644                },
1645                accessor: input.caller(),
1646                account: address,
1647                kind: crate::Vm::AccountAccessKind::Create,
1648                initialized: true,
1649                oldBalance: U256::ZERO, // updated on create_end
1650                newBalance: U256::ZERO, // updated on create_end
1651                oldNonce: 0,            // new contract starts with nonce 0
1652                newNonce: 1,            // updated on create_end (contracts start with nonce 1)
1653                value: input.value(),
1654                data: input.init_code(),
1655                reverted: false,
1656                deployedCode: Bytes::new(), // updated on create_end
1657                storageAccesses: vec![],    // updated on create_end
1658                depth: curr_depth as u64,
1659            }]);
1660        }
1661
1662        None
1663    }

1665    fn create_end(&mut self, ecx: Ecx, call: &CreateInputs, outcome: &mut CreateOutcome) {
1666        let call = Some(call);
1667        let curr_depth = ecx.journaled_state.depth();
1668
1669        // Clean up pranks
1670        if let Some(prank) = &self.get_prank(curr_depth)
1671            && curr_depth == prank.depth
1672        {
1673            ecx.tx.caller = prank.prank_origin;
1674
1675            // Clean single-call prank once we have returned to the original depth
1676            if prank.single_call {
1677                std::mem::take(&mut self.pranks);
1678            }
1679        }
1680
1681        // Clean up broadcasts
1682        if let Some(broadcast) = &self.broadcast
1683            && curr_depth == broadcast.depth
1684        {
1685            ecx.tx.caller = broadcast.original_origin;
1686
1687            // Clean single-call broadcast once we have returned to the original depth
1688            if broadcast.single_call {
1689                std::mem::take(&mut self.broadcast);
1690            }
1691        }
1692
1693        // Handle expected reverts
1694        if let Some(expected_revert) = &self.expected_revert
1695            && curr_depth <= expected_revert.depth
1696            && matches!(expected_revert.kind, ExpectedRevertKind::Default)
1697        {
1698            let mut expected_revert = std::mem::take(&mut self.expected_revert).unwrap();
1699            return match revert_handlers::handle_expect_revert(
1700                false,
1701                true,
1702                self.config.internal_expect_revert,
1703                &expected_revert,
1704                outcome.result.result,
1705                outcome.result.output.clone(),
1706                &self.config.available_artifacts,
1707            ) {
1708                Ok((address, retdata)) => {
1709                    expected_revert.actual_count += 1;
1710                    if expected_revert.actual_count < expected_revert.count {
1711                        self.expected_revert = Some(expected_revert.clone());
1712                    }
1713
1714                    outcome.result.result = InstructionResult::Return;
1715                    outcome.result.output = retdata;
1716                    outcome.address = address;
1717                }
1718                Err(err) => {
1719                    outcome.result.result = InstructionResult::Revert;
1720                    outcome.result.output = err.abi_encode().into();
1721                }
1722            };
1723        }
1724
1725        // If `startStateDiffRecording` has been called, update the `reverted` status of the
1726        // previous call depth's recorded accesses, if any
1727        if let Some(recorded_account_diffs_stack) = &mut self.recorded_account_diffs_stack {
1728            // The root call cannot be recorded.
1729            if curr_depth > 0
1730                && let Some(last_depth) = &mut recorded_account_diffs_stack.pop()
1731            {
1732                // Update the reverted status of all deeper calls if this call reverted, in
1733                // accordance with EVM behavior
1734                if outcome.result.is_revert() {
1735                    last_depth.iter_mut().for_each(|element| {
1736                        element.reverted = true;
1737                        element
1738                            .storageAccesses
1739                            .iter_mut()
1740                            .for_each(|storage_access| storage_access.reverted = true);
1741                    })
1742                }
1743
1744                if let Some(create_access) = last_depth.first_mut() {
1745                    // Assert that we're at the correct depth before recording post-create state
1746                    // changes. Depending on what depth the cheat was called at, there
1747                    // may not be any pending calls to update if execution has
1748                    // percolated up to a higher depth.
1749                    let depth = ecx.journaled_state.depth();
1750                    if create_access.depth == depth as u64 {
1751                        debug_assert_eq!(
1752                            create_access.kind as u8,
1753                            crate::Vm::AccountAccessKind::Create as u8
1754                        );
1755                        if let Some(address) = outcome.address
1756                            && let Ok(created_acc) = ecx.journaled_state.load_account(address)
1757                        {
1758                            create_access.newBalance = created_acc.info.balance;
1759                            create_access.newNonce = created_acc.info.nonce;
1760                            create_access.deployedCode =
1761                                created_acc.info.code.clone().unwrap_or_default().original_bytes();
1762                        }
1763                    }
1764                    // Merge the last depth's AccountAccesses into the AccountAccesses at the
1765                    // current depth, or push them back onto the pending
1766                    // vector if higher depths were not recorded. This
1767                    // preserves ordering of accesses.
1768                    if let Some(last) = recorded_account_diffs_stack.last_mut() {
1769                        last.append(last_depth);
1770                    } else {
1771                        recorded_account_diffs_stack.push(last_depth.clone());
1772                    }
1773                }
1774            }
1775        }
1776
1777        // Match the create against expected_creates
1778        if !self.expected_creates.is_empty()
1779            && let (Some(address), Some(call)) = (outcome.address, call)
1780            && let Ok(created_acc) = ecx.journaled_state.load_account(address)
1781        {
1782            let bytecode = created_acc.info.code.clone().unwrap_or_default().original_bytes();
1783            if let Some((index, _)) =
1784                self.expected_creates.iter().find_position(|expected_create| {
1785                    expected_create.deployer == call.caller
1786                        && expected_create.create_scheme.eq(call.scheme.into())
1787                        && expected_create.bytecode == bytecode
1788                })
1789            {
1790                self.expected_creates.swap_remove(index);
1791            }
1792        }
1793    }

1821    #[cold]
1822    fn meter_gas(&mut self, interpreter: &mut Interpreter) {
1823        if let Some(paused_gas) = self.gas_metering.paused_frames.last() {
1824            // Keep gas constant if paused.
1825            // Make sure we record the memory changes so that memory expansion is not paused.
1826            let memory = *interpreter.gas.memory();
1827            interpreter.gas = *paused_gas;
1828            interpreter.gas.memory_mut().words_num = memory.words_num;
1829            interpreter.gas.memory_mut().expansion_cost = memory.expansion_cost;
1830        } else {
1831            // Record frame paused gas.
1832            self.gas_metering.paused_frames.push(interpreter.gas);
1833        }
1834    }

1836    #[cold]
1837    fn meter_gas_record(&mut self, interpreter: &mut Interpreter, ecx: Ecx) {
1838        if interpreter.bytecode.action.as_ref().and_then(|i| i.instruction_result()).is_none() {
1839            self.gas_metering.gas_records.iter_mut().for_each(|record| {
1840                let curr_depth = ecx.journaled_state.depth();
1841                if curr_depth == record.depth {
1842                    // Skip the first opcode of the first call frame as it includes the gas cost of
1843                    // creating the snapshot.
1844                    if self.gas_metering.last_gas_used != 0 {
1845                        let gas_diff =
1846                            interpreter.gas.spent().saturating_sub(self.gas_metering.last_gas_used);
1847                        record.gas_used = record.gas_used.saturating_add(gas_diff);
1848                    }
1849
1850                    // Update `last_gas_used` to the current spent gas for the next iteration to
1851                    // compare against.
1852                    self.gas_metering.last_gas_used = interpreter.gas.spent();
1853                }
1854            });
1855        }
1856    }

1891    /// Invoked in inspector `step_end` (when the current opcode is not executed), if current opcode
1892    /// to execute is `SLOAD` and storage slot is cold.
1893    /// Ensures that in next step (when `SLOAD` opcode is executed) an arbitrary value is returned:
1894    /// - copies the existing arbitrary storage value (or the new generated one if no value in
1895    ///   cache) from mapped source address to the target address.
1896    /// - generates arbitrary value and saves it in target address storage.
1897    #[cold]
1898    fn arbitrary_storage_end(&mut self, interpreter: &mut Interpreter, ecx: Ecx) {
1899        let (key, target_address) = if interpreter.bytecode.opcode() == op::SLOAD {
1900            (try_or_return!(interpreter.stack.peek(0)), interpreter.input.target_address)
1901        } else {
1902            return;
1903        };
1904
1905        let Some(value) = ecx.sload(target_address, key) else {
1906            return;
1907        };
1908
1909        if (value.is_cold && value.data.is_zero())
1910            || self.should_overwrite_arbitrary_storage(&target_address, key)
1911        {
1912            if self.has_arbitrary_storage(&target_address) {
1913                let arbitrary_value = self.rng().random();
1914                self.arbitrary_storage.as_mut().unwrap().save(
1915                    ecx,
1916                    target_address,
1917                    key,
1918                    arbitrary_value,
1919                );
1920            } else if self.is_arbitrary_storage_copy(&target_address) {
1921                let arbitrary_value = self.rng().random();
1922                self.arbitrary_storage.as_mut().unwrap().copy(
1923                    ecx,
1924                    target_address,
1925                    key,
1926                    arbitrary_value,
1927                );
1928            }
1929        }
1930    }

1951        let Some(account_accesses) = &mut self.recorded_account_diffs_stack else { return };
1952        match interpreter.bytecode.opcode() {
1953            op::SELFDESTRUCT => {
1954                // Ensure that we're not selfdestructing a context recording was initiated on
1955                let Some(last) = account_accesses.last_mut() else { return };
1956
1957                // get previous balance, nonce and initialized status of the target account
1958                let target = try_or_return!(interpreter.stack.peek(0));
1959                let target = Address::from_word(B256::from(target));
1960                let (initialized, old_balance, old_nonce) = ecx
1961                    .journaled_state
1962                    .load_account(target)
1963                    .map(|account| {
1964                        (account.info.exists(), account.info.balance, account.info.nonce)
1965                    })
1966                    .unwrap_or_default();
1967
1968                // load balance of this account
1969                let value = ecx
1970                    .balance(interpreter.input.target_address)
1971                    .map(|b| b.data)
1972                    .unwrap_or(U256::ZERO);
1973
1974                // register access for the target account
1975                last.push(crate::Vm::AccountAccess {
1976                    chainInfo: crate::Vm::ChainInfo {
1977                        forkId: ecx.journaled_state.database.active_fork_id().unwrap_or_default(),

1980                    accessor: interpreter.input.target_address,
1981                    account: target,
1982                    kind: crate::Vm::AccountAccessKind::SelfDestruct,

1984                    oldBalance: old_balance,
1985                    newBalance: old_balance + value,
1986                    oldNonce: old_nonce,
1987                    newNonce: old_nonce, // nonce doesn't change on selfdestruct
1988                    value,
1989                    data: Bytes::new(),
1990                    reverted: false,
1991                    deployedCode: Bytes::new(),

1999            }
2000

2002                let Some(last) = account_accesses.last_mut() else { return };

2005                let address = interpreter.input.target_address;
2006
2007                // Try to include present value for informational purposes, otherwise assume
2008                // it's not set (zero value)
2009                let mut present_value = U256::ZERO;
2010                // Try to load the account and the slot's present value
2011                if ecx.journaled_state.load_account(address).is_ok()
2012                    && let Some(previous) = ecx.sload(address, key)

2017                    account: interpreter.input.target_address,
2018                    slot: key.into(),
2019                    isWrite: false,

2022                    reverted: false,

2025                    .journaled_state
2026                    .depth()
2027                    .try_into()
2028                    .expect("journaled state depth exceeds u64");
2029                append_storage_access(last, access, curr_depth);
2030            }
2031            op::SSTORE => {
2032                let Some(last) = account_accesses.last_mut() else { return };
2033
2034                let key = try_or_return!(interpreter.stack.peek(0));
2035                let value = try_or_return!(interpreter.stack.peek(1));
2036                let address = interpreter.input.target_address;
2037                // Try to load the account and the slot's previous value, otherwise, assume it's
2038                // not set (zero value)
2039                let mut previous_value = U256::ZERO;
2040                if ecx.journaled_state.load_account(address).is_ok()
2041                    && let Some(previous) = ecx.sload(address, key)
2042                {
2043                    previous_value = previous.data;
2044                }
2045
2046                let access = crate::Vm::StorageAccess {
2047                    account: address,
2048                    slot: key.into(),
2049                    isWrite: true,
2050                    previousValue: previous_value.into(),
2051                    newValue: value.into(),
2052                    reverted: false,
2053                };
2054                let curr_depth = ecx
2055                    .journaled_state
2056                    .depth()
2057                    .try_into()
2058                    .expect("journaled state depth exceeds u64");
2059                append_storage_access(last, access, curr_depth);
2060            }
2061
2062            // Record account accesses via the EXT family of opcodes
2063            op::EXTCODECOPY | op::EXTCODESIZE | op::EXTCODEHASH | op::BALANCE => {

2065                    op::EXTCODECOPY => crate::Vm::AccountAccessKind::Extcodecopy,
2066                    op::EXTCODESIZE => crate::Vm::AccountAccessKind::Extcodesize,
2067                    op::EXTCODEHASH => crate::Vm::AccountAccessKind::Extcodehash,
2068                    op::BALANCE => crate::Vm::AccountAccessKind::Balance,
2069                    _ => unreachable!(),
2070                };
2071                let address =
2072                    Address::from_word(B256::from(try_or_return!(interpreter.stack.peek(0))));
2073                let initialized;
2074                let balance;
2075                let nonce;
2076                if let Ok(acc) = ecx.journaled_state.load_account(address) {
2077                    initialized = acc.info.exists();
2078                    balance = acc.info.balance;
2079                    nonce = acc.info.nonce;
2080                } else {
2081                    initialized = false;
2082                    balance = U256::ZERO;
2083                    nonce = 0;
2084                }
2085                let curr_depth = ecx
2086                    .journaled_state
2087                    .depth()
2088                    .try_into()
2089                    .expect("journaled state depth exceeds u64");
2090                let account_access = crate::Vm::AccountAccess {
2091                    chainInfo: crate::Vm::ChainInfo {
2092                        forkId: ecx.journaled_state.database.active_fork_id().unwrap_or_default(),
2093                        chainId: U256::from(ecx.cfg.chain_id),
2094                    },
2095                    accessor: interpreter.input.target_address,
2096                    account: address,
2097                    kind,
2098                    initialized,
2099                    oldBalance: balance,
2100                    newBalance: balance,
2101                    oldNonce: nonce,
2102                    newNonce: nonce, // EXT* operations don't change nonce
2103                    value: U256::ZERO,
2104                    data: Bytes::new(),
2105                    reverted: false,
2106                    deployedCode: Bytes::new(),
2107                    storageAccesses: vec![],
2108                    depth: curr_depth,
2109                };
2110                // Record the EXT* call as an account access at the current depth
2111                // (future storage accesses will be recorded in a new "Resume" context)
2112                if let Some(last) = account_accesses.last_mut() {
2113                    last.push(account_access);
2114                } else {
2115                    account_accesses.push(vec![account_access]);
2116                }
2117            }
2118            _ => {}
2119        }
2120    }

2122    /// Checks to see if the current opcode can either mutate directly or expand memory.
2123    ///
2124    /// If the opcode at the current program counter is a match, check if the modified memory lies
2125    /// within the allowed ranges. If not, revert and fail the test.
2126    #[cold]
2127    fn check_mem_opcodes(&self, interpreter: &mut Interpreter, depth: u64) {
2128        let Some(ranges) = self.allowed_mem_writes.get(&depth) else {
2129            return;
2130        };
2131
2132        // The `mem_opcode_match` macro is used to match the current opcode against a list of
2133        // opcodes that can mutate memory (either directly or expansion via reading). If the
2134        // opcode is a match, the memory offsets that are being written to are checked to be
2135        // within the allowed ranges. If not, the test is failed and the transaction is
2136        // reverted. For all opcodes that can mutate memory aside from MSTORE,
2137        // MSTORE8, and MLOAD, the size and destination offset are on the stack, and
2138        // the macro expands all of these cases. For MSTORE, MSTORE8, and MLOAD, the
2139        // size of the memory write is implicit, so these cases are hard-coded.
2140        macro_rules! mem_opcode_match {
2141            ($(($opcode:ident, $offset_depth:expr, $size_depth:expr, $writes:expr)),* $(,)?) => {
2142                match interpreter.bytecode.opcode() {
2143                    ////////////////////////////////////////////////////////////////
2144                    //    OPERATIONS THAT CAN EXPAND/MUTATE MEMORY BY WRITING     //
2145                    ////////////////////////////////////////////////////////////////
2146
2147                    op::MSTORE => {
2148                        // The offset of the mstore operation is at the top of the stack.
2149                        let offset = try_or_return!(interpreter.stack.peek(0)).saturating_to::<u64>();
2150
2151                        // If none of the allowed ranges contain [offset, offset + 32), memory has been
2152                        // unexpectedly mutated.
2153                        if !ranges.iter().any(|range| {
2154                            range.contains(&offset) && range.contains(&(offset + 31))
2155                        }) {
2156                            // SPECIAL CASE: When the compiler attempts to store the selector for
2157                            // `stopExpectSafeMemory`, this is allowed. It will do so at the current free memory
2158                            // pointer, which could have been updated to the exclusive upper bound during
2159                            // execution.
2160                            let value = try_or_return!(interpreter.stack.peek(1)).to_be_bytes::<32>();
2161                            if value[..SELECTOR_LEN] == stopExpectSafeMemoryCall::SELECTOR {
2162                                return
2163                            }
2164
2165                            disallowed_mem_write(offset, 32, interpreter, ranges);
2166                            return
2167                        }
2168                    }
2169                    op::MSTORE8 => {
2170                        // The offset of the mstore8 operation is at the top of the stack.
2171                        let offset = try_or_return!(interpreter.stack.peek(0)).saturating_to::<u64>();
2172
2173                        // If none of the allowed ranges contain the offset, memory has been
2174                        // unexpectedly mutated.
2175                        if !ranges.iter().any(|range| range.contains(&offset)) {
2176                            disallowed_mem_write(offset, 1, interpreter, ranges);
2177                            return
2178                        }
2179                    }
2180
2181                    ////////////////////////////////////////////////////////////////
2182                    //        OPERATIONS THAT CAN EXPAND MEMORY BY READING        //
2183                    ////////////////////////////////////////////////////////////////
2184
2185                    op::MLOAD => {
2186                        // The offset of the mload operation is at the top of the stack
2187                        let offset = try_or_return!(interpreter.stack.peek(0)).saturating_to::<u64>();
2188
2189                        // If the offset being loaded is >= than the memory size, the
2190                        // memory is being expanded. If none of the allowed ranges contain
2191                        // [offset, offset + 32), memory has been unexpectedly mutated.
2192                        if offset >= interpreter.memory.size() as u64 && !ranges.iter().any(|range| {
2193                            range.contains(&offset) && range.contains(&(offset + 31))
2194                        }) {
2195                            disallowed_mem_write(offset, 32, interpreter, ranges);
2196                            return
2197                        }
2198                    }
2199
2200                    ////////////////////////////////////////////////////////////////
2201                    //          OPERATIONS WITH OFFSET AND SIZE ON STACK          //
2202                    ////////////////////////////////////////////////////////////////
2203
2204                    op::CALL => {
2205                        // The destination offset of the operation is the fifth element on the stack.
2206                        let dest_offset = try_or_return!(interpreter.stack.peek(5)).saturating_to::<u64>();
2207
2208                        // The size of the data that will be copied is the sixth element on the stack.
2209                        let size = try_or_return!(interpreter.stack.peek(6)).saturating_to::<u64>();
2210
2211                        // If none of the allowed ranges contain [dest_offset, dest_offset + size),
2212                        // memory outside of the expected ranges has been touched. If the opcode
2213                        // only reads from memory, this is okay as long as the memory is not expanded.
2214                        let fail_cond = !ranges.iter().any(|range| {
2215                            range.contains(&dest_offset) &&
2216                                range.contains(&(dest_offset + size.saturating_sub(1)))
2217                        });
2218
2219                        // If the failure condition is met, set the output buffer to a revert string
2220                        // that gives information about the allowed ranges and revert.
2221                        if fail_cond {
2222                            // SPECIAL CASE: When a call to `stopExpectSafeMemory` is performed, this is allowed.
2223                            // It allocated calldata at the current free memory pointer, and will attempt to read
2224                            // from this memory region to perform the call.
2225                            let to = Address::from_word(try_or_return!(interpreter.stack.peek(1)).to_be_bytes::<32>().into());
2226                            if to == CHEATCODE_ADDRESS {
2227                                let args_offset = try_or_return!(interpreter.stack.peek(3)).saturating_to::<usize>();
2228                                let args_size = try_or_return!(interpreter.stack.peek(4)).saturating_to::<usize>();
2229                                let memory_word = interpreter.memory.slice_len(args_offset, args_size);
2230                                if memory_word[..SELECTOR_LEN] == stopExpectSafeMemoryCall::SELECTOR {
2231                                    return
2232                                }
2233                            }
2234
2235                            disallowed_mem_write(dest_offset, size, interpreter, ranges);
2236                            return
2237                        }
2238                    }
2239
2240                    $(op::$opcode => {
2241                        // The destination offset of the operation.
2242                        let dest_offset = try_or_return!(interpreter.stack.peek($offset_depth)).saturating_to::<u64>();
2243
2244                        // The size of the data that will be copied.
2245                        let size = try_or_return!(interpreter.stack.peek($size_depth)).saturating_to::<u64>();
2246
2247                        // If none of the allowed ranges contain [dest_offset, dest_offset + size),
2248                        // memory outside of the expected ranges has been touched. If the opcode
2249                        // only reads from memory, this is okay as long as the memory is not expanded.
2250                        let fail_cond = !ranges.iter().any(|range| {
2251                                range.contains(&dest_offset) &&
2252                                    range.contains(&(dest_offset + size.saturating_sub(1)))
2253                            }) && ($writes ||
2254                                [dest_offset, (dest_offset + size).saturating_sub(1)].into_iter().any(|offset| {
2255                                    offset >= interpreter.memory.size() as u64
2256                                })
2257                            );
2258
2259                        // If the failure condition is met, set the output buffer to a revert string
2260                        // that gives information about the allowed ranges and revert.
2261                        if fail_cond {
2262                            disallowed_mem_write(dest_offset, size, interpreter, ranges);
2263                            return
2264                        }
2265                    })*
2266
2267                    _ => {}
2268                }
2269            }
2270        }
2271
2272        // Check if the current opcode can write to memory, and if so, check if the memory
2273        // being written to is registered as safe to modify.
2274        mem_opcode_match!(
2275            (CALLDATACOPY, 0, 2, true),
2276            (CODECOPY, 0, 2, true),
2277            (RETURNDATACOPY, 0, 2, true),
2278            (EXTCODECOPY, 1, 3, true),
2279            (CALLCODE, 5, 6, true),
2280            (STATICCALL, 4, 5, true),
2281            (DELEGATECALL, 4, 5, true),
2282            (KECCAK256, 0, 1, false),
2283            (LOG0, 0, 1, false),
2284            (LOG1, 0, 1, false),
2285            (LOG2, 0, 1, false),
2286            (LOG3, 0, 1, false),
2287            (LOG4, 0, 1, false),
2288            (CREATE, 1, 2, false),
2289            (CREATE2, 1, 2, false),
2290            (RETURN, 0, 1, false),
2291            (REVERT, 0, 1, false),
2292        );
2293    }
2294}

2296/// Helper that expands memory, stores a revert string pertaining to a disallowed memory write,
2297/// and sets the return range to the revert string's location in memory.
2298///
2299/// This will set the interpreter's next action to a return with the revert string as the output.
2300/// And trigger a revert.
2301fn disallowed_mem_write(
2302    dest_offset: u64,
2303    size: u64,
2304    interpreter: &mut Interpreter,
2305    ranges: &[Range<u64>],
2306) {

2308        "memory write at offset 0x{:02X} of size 0x{:02X} not allowed; safe range: {}",

2310        size,
2311        ranges.iter().map(|r| format!("(0x{:02X}, 0x{:02X}]", r.start, r.end)).join(" U ")
2312    );
2313
2314    interpreter.bytecode.set_action(InterpreterAction::new_return(
2315        InstructionResult::Revert,
2316        Bytes::from(revert_string.into_bytes()),
2317        interpreter.gas,
2318    ));
2319}

2346/// Appends an AccountAccess that resumes the recording of the current context.
2347fn append_storage_access(
2348    last: &mut Vec<AccountAccess>,
2349    storage_access: crate::Vm::StorageAccess,
2350    storage_depth: u64,
2351) {
2352    // Assert that there's an existing record for the current context.
2353    if !last.is_empty() && last.first().unwrap().depth < storage_depth {
2354        // Three cases to consider:
2355        // 1. If there hasn't been a context switch since the start of this context, then add the
2356        //    storage access to the current context record.
2357        // 2. If there's an existing Resume record, then add the storage access to it.
2358        // 3. Otherwise, create a new Resume record based on the current context.
2359        if last.len() == 1 {
2360            last.first_mut().unwrap().storageAccesses.push(storage_access);
2361        } else {
2362            let last_record = last.last_mut().unwrap();
2363            if last_record.kind as u8 == crate::Vm::AccountAccessKind::Resume as u8 {
2364                last_record.storageAccesses.push(storage_access);
2365            } else {
2366                let entry = last.first().unwrap();
2367                let resume_record = crate::Vm::AccountAccess {
2368                    chainInfo: crate::Vm::ChainInfo {
2369                        forkId: entry.chainInfo.forkId,
2370                        chainId: entry.chainInfo.chainId,
2371                    },
2372                    accessor: entry.accessor,
2373                    account: entry.account,
2374                    kind: crate::Vm::AccountAccessKind::Resume,
2375                    initialized: entry.initialized,
2376                    storageAccesses: vec![storage_access],
2377                    reverted: entry.reverted,
2378                    // The remaining fields are defaults
2379                    oldBalance: U256::ZERO,
2380                    newBalance: U256::ZERO,
2381                    oldNonce: 0,
2382                    newNonce: 0,
2383                    value: U256::ZERO,
2384                    data: Bytes::new(),
2385                    deployedCode: Bytes::new(),
2386                    depth: entry.depth,
2387                };
2388                last.push(resume_record);
2389            }
2390        }
2391    }
2392}

2394/// Dispatches the cheatcode call to the appropriate function.
2395fn apply_dispatch(
2396    calls: &Vm::VmCalls,
2397    ccx: &mut CheatsCtxt,
2398    executor: &mut dyn CheatcodesExecutor,
2399) -> Result {
2400    let cheat = calls_as_dyn_cheatcode(calls);
2401
2402    let _guard = debug_span!(target: "cheatcodes", "apply", id = %cheat.id()).entered();
2403    trace!(target: "cheatcodes", cheat = ?cheat.as_debug(), "applying");
2404
2405    if let spec::Status::Deprecated(replacement) = *cheat.status() {
2406        ccx.state.deprecated.insert(cheat.signature(), replacement);
2407    }
2408
2409    // Apply the cheatcode.
2410    let mut result = cheat.dyn_apply(ccx, executor);
2411
2412    // Format the error message to include the cheatcode name.
2413    if let Err(e) = &mut result
2414        && e.is_str()
2415    {
2416        let name = cheat.name();
2417        // Skip showing the cheatcode name for:
2418        // - assertions: too verbose, and can already be inferred from the error message
2419        // - `rpcUrl`: forge-std relies on it in `getChainWithUpdatedRpcUrl`
2420        if !name.contains("assert") && name != "rpcUrl" {
2421            *e = fmt_err!("vm.{name}: {e}");
2422        }
2423    }
2424
2425    trace!(
2426        target: "cheatcodes",
2427        return = %match &result {
2428            Ok(b) => hex::encode(b),
2429            Err(e) => e.to_string(),
2430        }
2431    );
2432
2433    result
2434}

2436fn calls_as_dyn_cheatcode(calls: &Vm::VmCalls) -> &dyn DynCheatcode {
2437    macro_rules! as_dyn {
2438        ($($variant:ident),*) => {
2439            match calls {
2440                $(Vm::VmCalls::$variant(cheat) => cheat,)*
2441            }
2442        };
2443    }
2444    vm_calls!(as_dyn)
2445}
