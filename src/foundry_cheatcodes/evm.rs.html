44/// Records storage slots reads and writes.
45#[derive(Clone, Debug, Default)]
46pub struct RecordAccess {
47    /// Storage slots reads.
48    pub reads: HashMap<Address, Vec<U256>>,
49    /// Storage slots writes.
50    pub writes: HashMap<Address, Vec<U256>>,
51}
52
53impl RecordAccess {
54    /// Records a read access to a storage slot.
55    pub fn record_read(&mut self, target: Address, slot: U256) {
56        self.reads.entry(target).or_default().push(slot);
57    }
58
59    /// Records a write access to a storage slot.
60    ///
61    /// This also records a read internally as `SSTORE` does an implicit `SLOAD`.
62    pub fn record_write(&mut self, target: Address, slot: U256) {
63        self.record_read(target, slot);
64        self.writes.entry(target).or_default().push(slot);
65    }
66
67    /// Clears the recorded reads and writes.
68    pub fn clear(&mut self) {
69        // Also frees memory.
70        *self = Default::default();
71    }
72}

128/// Account state diff info.
129#[derive(Serialize, Default)]
130#[serde(rename_all = "camelCase")]
131struct AccountStateDiffs {
132    /// Address label, if any set.
133    label: Option<String>,
134    /// Contract identifier from artifact. e.g "src/Counter.sol:Counter"
135    contract: Option<String>,
136    /// Account balance changes.
137    balance_diff: Option<BalanceDiff>,
138    /// Account nonce changes.
139    nonce_diff: Option<NonceDiff>,
140    /// State changes, per slot.
141    state_diff: BTreeMap<B256, SlotStateDiff>,
142}
143
144impl Display for AccountStateDiffs {
145    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> eyre::Result<(), std::fmt::Error> {
146        // Print changed account.
147        if let Some(label) = &self.label {
148            writeln!(f, "label: {label}")?;
149        }
150        if let Some(contract) = &self.contract {
151            writeln!(f, "contract: {contract}")?;
152        }
153        // Print balance diff if changed.
154        if let Some(balance_diff) = &self.balance_diff
155            && balance_diff.previous_value != balance_diff.new_value
156        {
157            writeln!(
158                f,
159                "- balance diff: {} → {}",
160                balance_diff.previous_value, balance_diff.new_value
161            )?;
162        }
163        // Print nonce diff if changed.
164        if let Some(nonce_diff) = &self.nonce_diff
165            && nonce_diff.previous_value != nonce_diff.new_value
166        {
167            writeln!(f, "- nonce diff: {} → {}", nonce_diff.previous_value, nonce_diff.new_value)?;
168        }
169        // Print state diff if any.
170        if !&self.state_diff.is_empty() {
171            writeln!(f, "- state diff:")?;
172            for (slot, slot_changes) in &self.state_diff {
173                writeln!(
174                    f,
175                    "@ {slot}: {} → {}",
176                    slot_changes.previous_value, slot_changes.new_value
177                )?;
178            }
179        }
180
181        Ok(())
182    }
183}

207impl Cheatcode for loadCall {
208    fn apply_stateful(&self, ccx: &mut CheatsCtxt) -> Result {
209        let Self { target, slot } = *self;
210        ccx.ensure_not_precompile(&target)?;
211        ccx.ecx.journaled_state.load_account(target)?;
212        let mut val = ccx.ecx.journaled_state.sload(target, slot.into())?;
213
214        if val.is_cold && val.data.is_zero() {
215            if ccx.state.has_arbitrary_storage(&target) {
216                // If storage slot is untouched and load from a target with arbitrary storage,
217                // then set random value for current slot.
218                let rand_value = ccx.state.rng().random();
219                ccx.state.arbitrary_storage.as_mut().unwrap().save(
220                    ccx.ecx,
221                    target,
222                    slot.into(),
223                    rand_value,
224                );
225                val.data = rand_value;
226            } else if ccx.state.is_arbitrary_storage_copy(&target) {
227                // If storage slot is untouched and load from a target that copies storage from
228                // a source address with arbitrary storage, then copy existing arbitrary value.
229                // If no arbitrary value generated yet, then the random one is saved and set.
230                let rand_value = ccx.state.rng().random();
231                val.data = ccx.state.arbitrary_storage.as_mut().unwrap().copy(
232                    ccx.ecx,
233                    target,
234                    slot.into(),
235                    rand_value,
236                );
237            }
238        }
239
240        Ok(val.abi_encode())
241    }
242}

244impl Cheatcode for loadAllocsCall {
245    fn apply_stateful(&self, ccx: &mut CheatsCtxt) -> Result {
246        let Self { pathToAllocsJson } = self;
247
248        let path = Path::new(pathToAllocsJson);
249        ensure!(path.exists(), "allocs file does not exist: {pathToAllocsJson}");
250
251        // Let's first assume we're reading a file with only the allocs.
252        let allocs: BTreeMap<Address, GenesisAccount> = match read_json_file(path) {
253            Ok(allocs) => allocs,
254            Err(_) => {
255                // Let's try and read from a genesis file, and extract allocs.
256                let genesis = read_json_file::<Genesis>(path)?;
257                genesis.alloc
258            }
259        };
260
261        // Then, load the allocs into the database.
262        let (db, journal, _) = ccx.ecx.as_db_env_and_journal();
263        db.load_allocs(&allocs, journal)
264            .map(|()| Vec::default())
265            .map_err(|e| fmt_err!("failed to load allocs: {e}"))
266    }
267}

283impl Cheatcode for dumpStateCall {
284    fn apply_stateful(&self, ccx: &mut CheatsCtxt) -> Result {
285        let Self { pathToStateJson } = self;
286        let path = Path::new(pathToStateJson);
287
288        // Do not include system account or empty accounts in the dump.
289        let skip = |key: &Address, val: &Account| {
290            key == &CHEATCODE_ADDRESS
291                || key == &CALLER
292                || key == &HARDHAT_CONSOLE_ADDRESS
293                || key == &TEST_CONTRACT_ADDRESS
294                || key == &ccx.caller
295                || key == &ccx.state.config.evm_opts.sender
296                || val.is_empty()
297        };
298
299        let alloc = ccx
300            .ecx
301            .journaled_state
302            .state()
303            .iter_mut()
304            .filter(|(key, val)| !skip(key, val))
305            .map(|(key, val)| (key, genesis_account(val)))
306            .collect::<BTreeMap<_, _>>();
307
308        write_json_file(path, &alloc)?;
309        Ok(Default::default())
310    }
311}

565impl Cheatcode for resetNonceCall {
566    fn apply_stateful(&self, ccx: &mut CheatsCtxt) -> Result {
567        let Self { account } = self;
568        let account = journaled_account(ccx.ecx, *account)?;
569        // Per EIP-161, EOA nonces start at 0, but contract nonces
570        // start at 1. Comparing by code_hash instead of code
571        // to avoid hitting the case where account's code is None.
572        let empty = account.info.code_hash == KECCAK_EMPTY;
573        let nonce = if empty { 0 } else { 1 };
574        account.info.nonce = nonce;
575        debug!(target: "cheatcodes", nonce, "reset");
576        Ok(Default::default())
577    }
578}

942impl Cheatcode for stopAndReturnDebugTraceRecordingCall {
943    fn apply_full(&self, ccx: &mut CheatsCtxt, executor: &mut dyn CheatcodesExecutor) -> Result {
944        let Some(tracer) = executor.tracing_inspector() else {
945            return Err(Error::from("no tracer initiated, consider adding -vvv flag"));
946        };
947
948        let Some(record_info) = ccx.state.record_debug_steps_info else {
949            return Err(Error::from("nothing recorded"));
950        };
951
952        // Use the trace nodes to flatten the call trace
953        let root = tracer.traces();
954        let steps = flatten_call_trace(0, root, record_info.start_node_idx);
955
956        let debug_steps: Vec<DebugStep> =
957            steps.iter().map(|&step| convert_call_trace_to_debug_step(step)).collect();
958        // Free up memory by clearing the steps if they are not recorded outside of cheatcode usage.
959        if !record_info.original_tracer_config.record_steps {
960            tracer.traces_mut().nodes_mut().iter_mut().for_each(|node| {
961                node.trace.steps = Vec::new();
962                node.logs = Vec::new();
963                node.ordering = Vec::new();
964            });
965        }
966
967        // Revert the tracer config to the one before recording
968        tracer.update_config(|_config| record_info.original_tracer_config);
969
970        // Clean up the recording info
971        ccx.state.record_debug_steps_info = None;
972
973        Ok(debug_steps.abi_encode())
974    }

1052fn inner_start_gas_snapshot(
1053    ccx: &mut CheatsCtxt,
1054    group: Option<String>,
1055    name: Option<String>,
1056) -> Result {
1057    // Revert if there is an active gas snapshot as we can only have one active snapshot at a time.
1058    if ccx.state.gas_metering.active_gas_snapshot.is_some() {
1059        let (group, name) = ccx.state.gas_metering.active_gas_snapshot.as_ref().unwrap().clone();
1060        bail!("gas snapshot was already started with group: {group} and name: {name}");
1061    }
1062
1063    let (group, name) = derive_snapshot_name(ccx, group, name);
1064
1065    ccx.state.gas_metering.gas_records.push(GasRecord {
1066        group: group.clone(),
1067        name: name.clone(),
1068        gas_used: 0,
1069        depth: ccx.ecx.journaled_state.depth(),
1070    });
1071
1072    ccx.state.gas_metering.active_gas_snapshot = Some((group, name));
1073
1074    ccx.state.gas_metering.start();
1075
1076    Ok(Default::default())
1077}

1079fn inner_stop_gas_snapshot(
1080    ccx: &mut CheatsCtxt,
1081    group: Option<String>,
1082    name: Option<String>,
1083) -> Result {
1084    // If group and name are not provided, use the last snapshot group and name.
1085    let (group, name) = group.zip(name).unwrap_or_else(|| {
1086        let (group, name) = ccx.state.gas_metering.active_gas_snapshot.as_ref().unwrap().clone();
1087        (group, name)
1088    });
1089
1090    if let Some(record) = ccx
1091        .state
1092        .gas_metering
1093        .gas_records
1094        .iter_mut()
1095        .find(|record| record.group == group && record.name == name)
1096    {
1097        // Calculate the gas used since the snapshot was started.
1098        // We subtract 171 from the gas used to account for gas used by the snapshot itself.
1099        let value = record.gas_used.saturating_sub(171);
1100
1101        ccx.state
1102            .gas_snapshots
1103            .entry(group.clone())
1104            .or_default()
1105            .insert(name.clone(), value.to_string());
1106
1107        // Stop the gas metering.
1108        ccx.state.gas_metering.stop();
1109
1110        // Remove the gas record.
1111        ccx.state
1112            .gas_metering
1113            .gas_records
1114            .retain(|record| record.group != group && record.name != name);
1115
1116        // Clear last snapshot cache if we have an exact match.
1117        if let Some((snapshot_group, snapshot_name)) = &ccx.state.gas_metering.active_gas_snapshot
1118            && snapshot_group == &group
1119            && snapshot_name == &name
1120        {
1121            ccx.state.gas_metering.active_gas_snapshot = None;
1122        }
1123
1124        Ok(value.abi_encode())
1125    } else {
1126        bail!("no gas snapshot was started with the name: {name} in group: {group}");
1127    }
1128}

1143/// Reads the current caller information and returns the current [CallerMode], `msg.sender` and
1144/// `tx.origin`.
1145///
1146/// Depending on the current caller mode, one of the following results will be returned:
1147/// - If there is an active prank:
1148///     - caller_mode will be equal to:
1149///         - [CallerMode::Prank] if the prank has been set with `vm.prank(..)`.
1150///         - [CallerMode::RecurrentPrank] if the prank has been set with `vm.startPrank(..)`.
1151///     - `msg.sender` will be equal to the address set for the prank.
1152///     - `tx.origin` will be equal to the default sender address unless an alternative one has been
1153///       set when configuring the prank.
1154///
1155/// - If there is an active broadcast:
1156///     - caller_mode will be equal to:
1157///         - [CallerMode::Broadcast] if the broadcast has been set with `vm.broadcast(..)`.
1158///         - [CallerMode::RecurrentBroadcast] if the broadcast has been set with
1159///           `vm.startBroadcast(..)`.
1160///     - `msg.sender` and `tx.origin` will be equal to the address provided when setting the
1161///       broadcast.
1162///
1163/// - If no caller modification is active:
1164///     - caller_mode will be equal to [CallerMode::None],
1165///     - `msg.sender` and `tx.origin` will be equal to the default sender address.
1166fn read_callers(state: &Cheatcodes, default_sender: &Address, call_depth: usize) -> Result {
1167    let mut mode = CallerMode::None;
1168    let mut new_caller = default_sender;
1169    let mut new_origin = default_sender;
1170    if let Some(prank) = state.get_prank(call_depth) {
1171        mode = if prank.single_call { CallerMode::Prank } else { CallerMode::RecurrentPrank };
1172        new_caller = &prank.new_caller;
1173        if let Some(new) = &prank.new_origin {
1174            new_origin = new;
1175        }
1176    } else if let Some(broadcast) = &state.broadcast {
1177        mode = if broadcast.single_call {
1178            CallerMode::Broadcast
1179        } else {
1180            CallerMode::RecurrentBroadcast
1181        };
1182        new_caller = &broadcast.new_origin;
1183        new_origin = &broadcast.new_origin;
1184    }
1185
1186    Ok((mode, new_caller, new_origin).abi_encode_params())
1187}

1239/// Helper function to returns state diffs recorded for each changed account.
1240fn get_recorded_state_diffs(ccx: &mut CheatsCtxt) -> BTreeMap<Address, AccountStateDiffs> {
1241    let mut state_diffs: BTreeMap<Address, AccountStateDiffs> = BTreeMap::default();
1242
1243    // First, collect all unique addresses we need to look up
1244    let mut addresses_to_lookup = HashSet::new();
1245    if let Some(records) = &ccx.state.recorded_account_diffs_stack {
1246        for account_access in records.iter().flatten() {
1247            if !account_access.storageAccesses.is_empty()
1248                || account_access.oldBalance != account_access.newBalance
1249            {
1250                addresses_to_lookup.insert(account_access.account);
1251                for storage_access in &account_access.storageAccesses {
1252                    if storage_access.isWrite && !storage_access.reverted {
1253                        addresses_to_lookup.insert(storage_access.account);
1254                    }
1255                }
1256            }
1257        }
1258    }
1259
1260    // Look up contract names for all addresses
1261    let mut contract_names = HashMap::new();
1262    for address in addresses_to_lookup {
1263        if let Some(name) = get_contract_name(ccx, address) {
1264            contract_names.insert(address, name);
1265        }
1266    }
1267
1268    // Now process the records
1269    if let Some(records) = &ccx.state.recorded_account_diffs_stack {
1270        records
1271            .iter()
1272            .flatten()
1273            .filter(|account_access| {
1274                !account_access.storageAccesses.is_empty()
1275                    || account_access.oldBalance != account_access.newBalance
1276                    || account_access.oldNonce != account_access.newNonce
1277            })
1278            .for_each(|account_access| {
1279                // Record account balance diffs.
1280                if account_access.oldBalance != account_access.newBalance {
1281                    let account_diff =
1282                        state_diffs.entry(account_access.account).or_insert_with(|| {
1283                            AccountStateDiffs {
1284                                label: ccx.state.labels.get(&account_access.account).cloned(),
1285                                contract: contract_names.get(&account_access.account).cloned(),
1286                                ..Default::default()
1287                            }
1288                        });
1289                    // Update balance diff. Do not overwrite the initial balance if already set.
1290                    if let Some(diff) = &mut account_diff.balance_diff {
1291                        diff.new_value = account_access.newBalance;
1292                    } else {
1293                        account_diff.balance_diff = Some(BalanceDiff {
1294                            previous_value: account_access.oldBalance,
1295                            new_value: account_access.newBalance,
1296                        });
1297                    }
1298                }
1299
1300                // Record account nonce diffs.
1301                if account_access.oldNonce != account_access.newNonce {
1302                    let account_diff =
1303                        state_diffs.entry(account_access.account).or_insert_with(|| {
1304                            AccountStateDiffs {
1305                                label: ccx.state.labels.get(&account_access.account).cloned(),
1306                                contract: contract_names.get(&account_access.account).cloned(),
1307                                ..Default::default()
1308                            }
1309                        });
1310                    // Update nonce diff. Do not overwrite the initial nonce if already set.
1311                    if let Some(diff) = &mut account_diff.nonce_diff {
1312                        diff.new_value = account_access.newNonce;
1313                    } else {
1314                        account_diff.nonce_diff = Some(NonceDiff {
1315                            previous_value: account_access.oldNonce,
1316                            new_value: account_access.newNonce,
1317                        });
1318                    }
1319                }
1320
1321                // Record account state diffs.
1322                for storage_access in &account_access.storageAccesses {
1323                    if storage_access.isWrite && !storage_access.reverted {
1324                        let account_diff = state_diffs
1325                            .entry(storage_access.account)
1326                            .or_insert_with(|| AccountStateDiffs {
1327                                label: ccx.state.labels.get(&storage_access.account).cloned(),
1328                                contract: contract_names.get(&storage_access.account).cloned(),
1329                                ..Default::default()
1330                            });
1331                        // Update state diff. Do not overwrite the initial value if already set.
1332                        match account_diff.state_diff.entry(storage_access.slot) {
1333                            Entry::Vacant(slot_state_diff) => {
1334                                slot_state_diff.insert(SlotStateDiff {
1335                                    previous_value: storage_access.previousValue,
1336                                    new_value: storage_access.newValue,
1337                                });
1338                            }
1339                            Entry::Occupied(mut slot_state_diff) => {
1340                                slot_state_diff.get_mut().new_value = storage_access.newValue;
1341                            }
1342                        }
1343                    }
1344                }
1345            });
1346    }
1347    state_diffs
1348}

1350/// Helper function to get the contract name from the deployed code.
1351fn get_contract_name(ccx: &mut CheatsCtxt, address: Address) -> Option<String> {
1352    // Check if we have available artifacts to match against
1353    let artifacts = ccx.state.config.available_artifacts.as_ref()?;
1354
1355    // Try to load the account and get its code
1356    let account = ccx.ecx.journaled_state.load_account(address).ok()?;
1357    let code = account.info.code.as_ref()?;
1358
1359    // Skip if code is empty
1360    if code.is_empty() {
1361        return None;
1362    }
1363
1364    // Try to find the artifact by deployed code
1365    let code_bytes = code.original_bytes();
1366    if let Some((artifact_id, _)) = artifacts.find_by_deployed_code_exact(&code_bytes) {
1367        return Some(artifact_id.identifier());
1368    }
1369
1370    // Fallback to fuzzy matching if exact match fails
1371    if let Some((artifact_id, _)) = artifacts.find_by_deployed_code(&code_bytes) {
1372        return Some(artifact_id.identifier());
1373    }
1374
1375    None
1376}

1378/// Helper function to set / unset cold storage slot of the target address.
1379fn set_cold_slot(ccx: &mut CheatsCtxt, target: Address, slot: U256, cold: bool) {
1380    if let Some(account) = ccx.ecx.journaled_state.state.get_mut(&target)
1381        && let Some(storage_slot) = account.storage.get_mut(&slot)
1382    {
1383        storage_slot.is_cold = cold;
1384    }
