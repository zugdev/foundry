67/// Cheatcode implementation.
68pub(crate) trait Cheatcode: CheatcodeDef + DynCheatcode {
69    /// Applies this cheatcode to the given state.
70    ///
71    /// Implement this function if you don't need access to the EVM data.
72    fn apply(&self, state: &mut Cheatcodes) -> Result {
73        let _ = state;
74        unimplemented!("{}", Self::CHEATCODE.func.id)
75    }
76
77    /// Applies this cheatcode to the given context.
78    ///
79    /// Implement this function if you need access to the EVM data.
80    #[inline(always)]
81    fn apply_stateful(&self, ccx: &mut CheatsCtxt) -> Result {
82        self.apply(ccx.state)
83    }
84
85    /// Applies this cheatcode to the given context and executor.
86    ///
87    /// Implement this function if you need access to the executor.
88    #[inline(always)]
89    fn apply_full(&self, ccx: &mut CheatsCtxt, executor: &mut dyn CheatcodesExecutor) -> Result {
90        let _ = executor;
91        self.apply_stateful(ccx)
92    }
93}

95pub(crate) trait DynCheatcode: 'static {
96    fn cheatcode(&self) -> &'static spec::Cheatcode<'static>;
97
98    fn as_debug(&self) -> &dyn std::fmt::Debug;
99
100    fn dyn_apply(&self, ccx: &mut CheatsCtxt, executor: &mut dyn CheatcodesExecutor) -> Result;
101}
102
103impl<T: Cheatcode> DynCheatcode for T {
104    #[inline]
105    fn cheatcode(&self) -> &'static spec::Cheatcode<'static> {
106        Self::CHEATCODE
107    }
108
109    #[inline]
110    fn as_debug(&self) -> &dyn std::fmt::Debug {
111        self
112    }
113
114    #[inline]
115    fn dyn_apply(&self, ccx: &mut CheatsCtxt, executor: &mut dyn CheatcodesExecutor) -> Result {
116        self.apply_full(ccx, executor)
117    }
118}

120impl dyn DynCheatcode {
121    pub(crate) fn name(&self) -> &'static str {
122        self.cheatcode().func.signature.split('(').next().unwrap()
123    }
124
125    pub(crate) fn id(&self) -> &'static str {
126        self.cheatcode().func.id
127    }
128
129    pub(crate) fn signature(&self) -> &'static str {
130        self.cheatcode().func.signature
131    }
132
133    pub(crate) fn status(&self) -> &Status<'static> {
134        &self.cheatcode().status
135    }
136}

138/// The cheatcode context, used in `Cheatcode`.
139pub struct CheatsCtxt<'cheats, 'evm, 'db, 'db2> {
140    /// The cheatcodes inspector state.
141    pub(crate) state: &'cheats mut Cheatcodes,
142    /// The EVM data.
143    pub(crate) ecx: &'evm mut EthEvmContext<&'db mut (dyn DatabaseExt + 'db2)>,
144    /// The original `msg.sender`.
145    pub(crate) caller: Address,
146    /// Gas limit of the current cheatcode call.
147    pub(crate) gas_limit: u64,
148}
149
150impl<'db, 'db2> std::ops::Deref for CheatsCtxt<'_, '_, 'db, 'db2> {
151    type Target = EthEvmContext<&'db mut (dyn DatabaseExt + 'db2)>;
152
153    #[inline(always)]
154    fn deref(&self) -> &Self::Target {
155        self.ecx
156    }
157}
158
159impl std::ops::DerefMut for CheatsCtxt<'_, '_, '_, '_> {
160    #[inline(always)]
161    fn deref_mut(&mut self) -> &mut Self::Target {
162        &mut *self.ecx
163    }
164}
165
166impl CheatsCtxt<'_, '_, '_, '_> {
167    #[inline]
168    pub(crate) fn ensure_not_precompile(&self, address: &Address) -> Result<()> {
169        if self.is_precompile(address) { Err(precompile_error(address)) } else { Ok(()) }
170    }
171
172    #[inline]
173    pub(crate) fn is_precompile(&self, address: &Address) -> bool {
174        self.ecx.journaled_state.inner.precompiles.contains(address)
175    }
176}
177
178#[cold]
179fn precompile_error(address: &Address) -> Error {
180    fmt_err!("cannot use precompile {address} as an argument")
